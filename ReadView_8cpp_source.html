38  explicit Impl(
39  DigestAwareReadView const& ledger,
40  std::unordered_set<uint256, beast::uhash<>> const& presets)
41  : presets_(presets)
42  {
43  auto const k = keylet::amendments();
44  digest_ = ledger.digest(k.key);
45  if (!digest_)
46  return;
47  auto const sle = ledger.read(k);
48  if (!sle)
49  {
50  // LogicError() ?
51  return;
52  }
53 
54  for (auto const& item : sle->getFieldV256(sfAmendments))
55  set_.insert(item);
56  }
57 
58  bool
59  enabled(uint256 const& feature) const
60  {
61  if (presets_.count(feature) > 0)
62  return true;
63  return set_.count(feature) > 0;
64  }
65 
66  bool
67  changed(DigestAwareReadView const& ledger) const
68  {
69  auto const digest = ledger.digest(keylet::amendments().key);
70  if (!digest && !digest_)
71  return false;
72  if (!digest || !digest_)
73  return true;
74  return *digest != *digest_;
75  }

77  bool
78  operator==(Impl const& other) const
79  {
80  if (!digest_ && !other.digest_)
81  return true;
82  if (!digest_ || !other.digest_)
83  return false;
84  return *digest_ == *other.digest_;
85  }
