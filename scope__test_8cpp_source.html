29  test_scope_exit()
30  {
31  // scope_exit always executes the functor on destruction,
32  // unless release() is called
33  int i = 0;
34  {
35  scope_exit x{[&i]() { i = 1; }};
36  }
37  BEAST_EXPECT(i == 1);
38  {
39  scope_exit x{[&i]() { i = 2; }};
40  x.release();
41  }
42  BEAST_EXPECT(i == 1);
43  {
44  scope_exit x{[&i]() { i += 2; }};
45  auto x2 = std::move(x);
46  }
47  BEAST_EXPECT(i == 3);
48  {
49  scope_exit x{[&i]() { i = 4; }};
50  x.release();
51  auto x2 = std::move(x);
52  }
53  BEAST_EXPECT(i == 3);

58  throw 1;
59  }

61  {
62  }
