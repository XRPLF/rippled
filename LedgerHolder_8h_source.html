39class LedgerHolder : public CountedObject<LedgerHolder>
40{
41public:
42 // Update the held ledger
43 void
44 set(std::shared_ptr<Ledger const> ledger)
45 {
46 if (!ledger)
47 LogicError("LedgerHolder::set with nullptr");
48 if (!ledger->isImmutable())
49 LogicError("LedgerHolder::set with mutable Ledger");
50 std::lock_guard sl(m_lock);
51 m_heldLedger = std::move(ledger);
52 }
53
54 // Return the (immutable) held ledger
55 std::shared_ptr<Ledger const>
56 get()
57 {
58 std::lock_guard sl(m_lock);
59 return m_heldLedger;
60 }
61
62 bool
63 empty()
64 {
65 std::lock_guard sl(m_lock);
66 return m_heldLedger == nullptr;
67 }
68
69private:
70 std::mutex m_lock;

72};
