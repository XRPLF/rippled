435  void
436  testGetKeys()
437  {
438  testcase("getKeys");
439 
440  ManifestCache cache;
441  auto const sk = randomSecretKey();
442  auto const pk = derivePublicKey(KeyType::ed25519, sk);
443 
444  // getSigningKey should return same key if there is no manifest
445  BEAST_EXPECT(cache.getSigningKey(pk) == pk);
446 
447  // getSigningKey should return the ephemeral public key
448  // for the listed validator master public key
449  // getMasterKey should return the listed validator master key
450  // for that ephemeral public key
451  auto const kp0 = randomKeyPair(KeyType::secp256k1);
452  BEAST_EXPECT(
453  ManifestDisposition::accepted ==
454  cache.applyManifest(makeManifest(
455  sk, KeyType::ed25519, kp0.second, KeyType::secp256k1, 0)));
456  BEAST_EXPECT(cache.getSigningKey(pk) == kp0.first);
457  BEAST_EXPECT(cache.getMasterKey(kp0.first) == pk);
458 
459  // getSigningKey should return the latest ephemeral public key
460  // for the listed validator master public key
461  // getMasterKey should only return a master key for the latest
462  // ephemeral public key
463  auto const kp1 = randomKeyPair(KeyType::secp256k1);
464  BEAST_EXPECT(
465  ManifestDisposition::accepted ==
466  cache.applyManifest(makeManifest(
467  sk, KeyType::ed25519, kp1.second, KeyType::secp256k1, 1)));
468  BEAST_EXPECT(cache.getSigningKey(pk) == kp1.first);
469  BEAST_EXPECT(cache.getMasterKey(kp1.first) == pk);
470  BEAST_EXPECT(cache.getMasterKey(kp0.first) == kp0.first);
471 
472  // getSigningKey and getMasterKey should return the same keys if
473  // a new manifest is applied with the same signing key but a higher
474  // sequence
475  BEAST_EXPECT(
476  ManifestDisposition::accepted ==
477  cache.applyManifest(makeManifest(
478  sk, KeyType::ed25519, kp1.second, KeyType::secp256k1, 2)));
479  BEAST_EXPECT(cache.getSigningKey(pk) == kp1.first);
480  BEAST_EXPECT(cache.getMasterKey(kp1.first) == pk);
481  BEAST_EXPECT(cache.getMasterKey(kp0.first) == kp0.first);
482 
483  // getSigningKey should return boost::none for a revoked master public
484  // key getMasterKey should return boost::none for an ephemeral public
485  // key from a revoked master public key
486  BEAST_EXPECT(
487  ManifestDisposition::accepted ==
488  cache.applyManifest(makeRevocation(sk, KeyType::ed25519)));
489  BEAST_EXPECT(cache.revoked(pk));
490  BEAST_EXPECT(cache.getSigningKey(pk) == pk);
491  BEAST_EXPECT(cache.getMasterKey(kp0.first) == kp0.first);
492  BEAST_EXPECT(cache.getMasterKey(kp1.first) == kp1.first);
493  }

590  void
591  testManifestDeserialization()
592  {
593  std::array<KeyType, 2> const keyTypes{
594  {KeyType::ed25519, KeyType::secp256k1}};
595 
596  std::uint32_t sequence = 0;
597 
598  // public key with invalid type
599  auto const ret = strUnHex(
600  "9930E7FC9D56BB25D6893BA3F317AE5BCF33B3291BD63DB32654A313222F7FD02"
601  "0");
602  auto const badKey = Slice{ret->data(), ret->size()};
603 
604  // short public key
605  auto const retShort = strUnHex("0330");
606  auto const shortKey = Slice{retShort->data(), retShort->size()};
607 
608  auto toString = [](STObject const& st) {
609  Serializer s;
610  st.add(s);
611 
612  return std::string(static_cast<char const*>(s.data()), s.size());
613  };
614 
615  for (auto const keyType : keyTypes)
616  {
617  auto const sk = generateSecretKey(keyType, randomSeed());
618  auto const pk = derivePublicKey(keyType, sk);
619 
620  for (auto const sKeyType : keyTypes)
621  {
622  auto const ssk = generateSecretKey(sKeyType, randomSeed());
623  auto const spk = derivePublicKey(sKeyType, ssk);
624 
625  auto buildManifestObject =
626  [&](std::uint32_t seq,
627  boost::optional<std::string> domain,
628  bool noSigningPublic = false,
629  bool noSignature = false) {
630  STObject st(sfGeneric);
631  st[sfSequence] = seq;
632  st[sfPublicKey] = pk;
633 
634  if (domain)
635  st[sfDomain] = makeSlice(*domain);
636 
637  if (!noSigningPublic)
638  st[sfSigningPubKey] = spk;
639 
640  sign(
641  st,
642  HashPrefix::manifest,
643  keyType,
644  sk,
645  sfMasterSignature);
646 
647  if (!noSignature)
648  sign(st, HashPrefix::manifest, sKeyType, ssk);
649 
650  return st;
651  };
652 
653  {
654  testcase << "deserializeManifest: normal manifest ("
655  << to_string(keyType) << " + "
656  << to_string(sKeyType) << ")";
657 
658  { // valid manifest without domain
659  auto const st =
660  buildManifestObject(++sequence, boost::none);
661 
662  auto const m = toString(st);
663  auto const manifest = deserializeManifest(m);
664 
665  BEAST_EXPECT(manifest);
666  BEAST_EXPECT(manifest->masterKey == pk);
667  BEAST_EXPECT(manifest->signingKey == spk);
668  BEAST_EXPECT(manifest->sequence == sequence);
669  BEAST_EXPECT(manifest->serialized == m);
670  BEAST_EXPECT(manifest->domain.empty());
671  BEAST_EXPECT(manifest->verify());
672  }
673 
674  { // invalid manifest (empty domain)
675  auto const st =
676  buildManifestObject(++sequence, std::string{});
677 
678  BEAST_EXPECT(!deserializeManifest(toString(st)));
679  }
680
