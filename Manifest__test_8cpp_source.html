107 std::string
108 makeManifestString(
109 PublicKey const& pk,
110 SecretKey const& sk,
111 PublicKey const& spk,
112 SecretKey const& ssk,
113 int seq)
114 {
115 STObject st(sfGeneric);
116 st[sfSequence] = seq;
117 st[sfPublicKey] = pk;
118 st[sfSigningPubKey] = spk;
119
120 sign(st, HashPrefix::manifest, *publicKeyType(spk), ssk);
121 sign(
122 st,
123 HashPrefix::manifest,
124 *publicKeyType(pk),
125 sk,
126 sfMasterSignature);
127
128 Serializer s;
129 st.add(s);
130
131 return base64_encode(
132 std::string(static_cast<char const*>(s.data()), s.size()));
133 }

135 std::string
136 makeRevocationString(
137 SecretKey const& sk,
138 KeyType type,
139 bool invalidSig = false)
140 {
141 auto const pk = derivePublicKey(type, sk);
142
143 STObject st(sfGeneric);
144 st[sfSequence] = std::numeric_limits<std::uint32_t>::max();
145 st[sfPublicKey] = pk;
146
147 sign(
148 st,
149 HashPrefix::manifest,
150 type,
151 invalidSig ? randomSecretKey() : sk,

153 BEAST_EXPECT(
154 invalidSig ^

157 Serializer s;
158 st.add(s);
159
160 return base64_encode(
161 std::string(static_cast<char const*>(s.data()), s.size()));

267 };

270 std::sort(
271 mv.begin(),
272 mv.end(),
273 [](Manifest const* lhs, Manifest const* rhs) {

331 sort(getPopulatedManifests(loaded)));

337 inManifests.end(),

344 {
345 fail();
346 }
347 }

351 std::vector<std::string> const emptyRevocation;

355 *dbCon,
356 "ValidatorManifests",
357 badManifest,
358 emptyRevocation));
359

361 auto const pk = derivePublicKey(KeyType::ed25519, sk);
362 auto const kp = randomKeyPair(KeyType::secp256k1);

379 "bad revocation"};
380 BEAST_EXPECT(!loaded.load(

382 "ValidatorManifests",
383 emptyManifest,
384 badRevocation));
385

387 auto const keyType = KeyType::ed25519;
388 auto const pk = derivePublicKey(keyType, sk);
389 auto const kp = randomKeyPair(KeyType::secp256k1);

448 void
449 testGetKeys()
450 {
451 testcase("getKeys");
452
453 ManifestCache cache;
454 auto const sk = randomSecretKey();
455 auto const pk = derivePublicKey(KeyType::ed25519, sk);
456
457 // getSigningKey should return same key if there is no manifest
458 BEAST_EXPECT(cache.getSigningKey(pk) == pk);
459
460 // getSigningKey should return the ephemeral public key
461 // for the listed validator master public key
462 // getMasterKey should return the listed validator master key
463 // for that ephemeral public key
464 auto const kp0 = randomKeyPair(KeyType::secp256k1);
465 BEAST_EXPECT(
466 ManifestDisposition::accepted ==
467 cache.applyManifest(makeManifest(
468 sk, KeyType::ed25519, kp0.second, KeyType::secp256k1, 0)));
469 BEAST_EXPECT(cache.getSigningKey(pk) == kp0.first);
470 BEAST_EXPECT(cache.getMasterKey(kp0.first) == pk);
471
472 // getSigningKey should return the latest ephemeral public key
473 // for the listed validator master public key
474 // getMasterKey should only return a master key for the latest
475 // ephemeral public key
476 auto const kp1 = randomKeyPair(KeyType::secp256k1);
477 BEAST_EXPECT(
478 ManifestDisposition::accepted ==
479 cache.applyManifest(makeManifest(
480 sk, KeyType::ed25519, kp1.second, KeyType::secp256k1, 1)));
481 BEAST_EXPECT(cache.getSigningKey(pk) == kp1.first);
482 BEAST_EXPECT(cache.getMasterKey(kp1.first) == pk);
483 BEAST_EXPECT(cache.getMasterKey(kp0.first) == kp0.first);
484
485 // getSigningKey and getMasterKey should fail if a new manifest is
486 // applied with the same signing key but a higher sequence
487 BEAST_EXPECT(
488 ManifestDisposition::badEphemeralKey ==
489 cache.applyManifest(makeManifest(
490 sk, KeyType::ed25519, kp1.second, KeyType::secp256k1, 2)));
491 BEAST_EXPECT(cache.getSigningKey(pk) == kp1.first);
492 BEAST_EXPECT(cache.getMasterKey(kp1.first) == pk);
493 BEAST_EXPECT(cache.getMasterKey(kp0.first) == kp0.first);
494
495 // getSigningKey should return std::nullopt for a revoked master public
496 // key getMasterKey should return std::nullopt for an ephemeral public
497 // key from a revoked master public key
498 BEAST_EXPECT(
499 ManifestDisposition::accepted ==
500 cache.applyManifest(makeRevocation(sk, KeyType::ed25519)));
501 BEAST_EXPECT(cache.revoked(pk));
502 BEAST_EXPECT(cache.getSigningKey(pk) == pk);
503 BEAST_EXPECT(cache.getMasterKey(kp0.first) == kp0.first);
504 BEAST_EXPECT(cache.getMasterKey(kp1.first) == kp1.first);
505 }

602 void
603 testManifestDeserialization()
604 {
605 std::array<KeyType, 2> const keyTypes{
606 {KeyType::ed25519, KeyType::secp256k1}};
607
608 std::uint32_t sequence = 0;
609
610 // public key with invalid type
611 std::array<std::uint8_t, 33> const badKey{
612 0x99, 0x30, 0xE7, 0xFC, 0x9D, 0x56, 0xBB, 0x25, 0xD6, 0x89, 0x3B,
613 0xA3, 0xF3, 0x17, 0xAE, 0x5B, 0xCF, 0x33, 0xB3, 0x29, 0x1B, 0xD6,
614 0x3D, 0xB3, 0x26, 0x54, 0xA3, 0x13, 0x22, 0x2F, 0x7F, 0xD0, 0x20};
615
616 // Short public key:
617 std::array<std::uint8_t, 16> const shortKey{
618 0x03,
619 0x30,
620 0xE7,
621 0xFC,
622 0x9D,
623 0x56,
624 0xBB,
625 0x25,
626 0xD6,
627 0x89,
628 0x3B,
629 0xA3,
630 0xF3,
631 0x17,
632 0xAE,
633 0x5B};
634
635 auto toString = [](STObject const& st) {
636 Serializer s;
637 st.add(s);
638
639 return std::string(static_cast<char const*>(s.data()), s.size());
640 };
641
642 for (auto const keyType : keyTypes)
643 {
644 auto const sk = generateSecretKey(keyType, randomSeed());
645 auto const pk = derivePublicKey(keyType, sk);
646
647 for (auto const sKeyType : keyTypes)
648 {
649 auto const ssk = generateSecretKey(sKeyType, randomSeed());
650 auto const spk = derivePublicKey(sKeyType, ssk);
651
652 auto buildManifestObject =
653 [&](std::uint32_t seq,
654 std::optional<std::string> domain,
655 bool noSigningPublic = false,
656 bool noSignature = false) {
657 STObject st(sfGeneric);
658 st[sfSequence] = seq;
659 st[sfPublicKey] = pk;
660
661 if (domain)
662 st[sfDomain] = makeSlice(*domain);
663
664 if (!noSigningPublic)
665 st[sfSigningPubKey] = spk;
666
667 sign(
668 st,
669 HashPrefix::manifest,
670 keyType,
671 sk,
672 sfMasterSignature);
673
674 if (!noSignature)
675 sign(st, HashPrefix::manifest, sKeyType, ssk);
676
677 return st;
678 };
679
680 {
681 testcase << "deserializeManifest: normal manifest ("
682 << to_string(keyType) << " + "
683 << to_string(sKeyType) << ")";
684
685 { // valid manifest without domain
686 auto const st =
687 buildManifestObject(++sequence, std::nullopt);
688
689 auto const m = toString(st);
690 auto const manifest = deserializeManifest(m);
691
692 BEAST_EXPECT(manifest);
693 BEAST_EXPECT(manifest->masterKey == pk);
694 BEAST_EXPECT(manifest->signingKey == spk);
695 BEAST_EXPECT(manifest->sequence == sequence);
696 BEAST_EXPECT(manifest->serialized == m);
697 BEAST_EXPECT(manifest->domain.empty());
698 BEAST_EXPECT(manifest->verify());
699 }
700
701 { // invalid manifest (empty domain)
702 auto const st =
703 buildManifestObject(++sequence, std::string{});
704
705 BEAST_EXPECT(!deserializeManifest(toString(st)));
706 }
707
708 { // invalid manifest (domain too short)
709 auto const st =
710 buildManifestObject(++sequence, std::string{"a.b"});
711 BEAST_EXPECT(!deserializeManifest(toString(st)));
712 }
713 { // invalid manifest (domain too long)
714 std::string s(254, 'a');
715 auto const st =
716 buildManifestObject(++sequence, s + ".example.com");
717 BEAST_EXPECT(!deserializeManifest(toString(st)));
718 }
719 { // invalid manifest (domain component too long)
720 std::string s(72, 'a');
721 auto const st =
722 buildManifestObject(++sequence, s + ".example.com");
723 BEAST_EXPECT(!deserializeManifest(toString(st)));
724 }
725
726 auto const st = buildManifestObject(
727 ++sequence, std::string{"example.com"});
728
729 {
730 // valid manifest with domain
731 auto const m = toString(st);
732 auto const manifest = deserializeManifest(m);
733
734 BEAST_EXPECT(manifest);
735 BEAST_EXPECT(manifest->masterKey == pk);
736 BEAST_EXPECT(manifest->signingKey == spk);

895 testcase("Manifest Domain Names");
896
897 auto const sk1 = generateSecretKey(KeyType::secp256k1, randomSeed());
898 auto const pk1 = derivePublicKey(KeyType::secp256k1, sk1);
899
900 auto const sk2 = generateSecretKey(KeyType::secp256k1, randomSeed());
901 auto const pk2 = derivePublicKey(KeyType::secp256k1, sk2);
902
903 auto test = [&](std::string domain) {
904 STObject st(sfGeneric);
905 st[sfSequence] = 7;
906 st[sfPublicKey] = pk1;
907 st[sfDomain] = makeSlice(domain);
908 st[sfSigningPubKey] = pk2;
909
910 sign(
911 st,
912 HashPrefix::manifest,
913 KeyType::secp256k1,
914 sk1,
915 sfMasterSignature);
916 sign(st, HashPrefix::manifest, KeyType::secp256k1, sk2);
917
918 Serializer s;
919 st.add(s);
920
921 return deserializeManifest(
922 std::string(static_cast<char const*>(s.data()), s.size()));
923 };
924
925 BEAST_EXPECT(test("example.com"));
926 BEAST_EXPECT(test("test.example.com"));
927 BEAST_EXPECT(test("example-domain.com"));
928 BEAST_EXPECT(test("xn--mxavchb.gr"));
929 BEAST_EXPECT(test("test.xn--mxavchb.gr"));
930 BEAST_EXPECT(test("123.gr"));
931 BEAST_EXPECT(test("x.yz"));
932 BEAST_EXPECT(test(std::string(63, 'a') + ".example.com"));
933 BEAST_EXPECT(test(std::string(63, 'a') + "." + std::string(63, 'b')));
934
935 // No period
936 BEAST_EXPECT(!test("example"));
937
938 // Leading period:
939 BEAST_EXPECT(!test(".com"));
940 BEAST_EXPECT(!test(".example.com"));
941
942 // A trailing period is technically valid but we don't allow it
943 BEAST_EXPECT(!test("example.com."));
944
945 // A component can't start or end with a dash
946 BEAST_EXPECT(!test("-example.com"));
947 BEAST_EXPECT(!test("example-.com"));
948
949 // Empty component:
950 BEAST_EXPECT(!test("double..periods.example.com"));
951
952 // TLD too short or too long:
953 BEAST_EXPECT(!test("example.x"));
954 BEAST_EXPECT(!test("example." + std::string(64, 'a')));
955
956 // Invalid characters:
957 BEAST_EXPECT(!test("example.com-org"));
958 BEAST_EXPECT(!test("bang!.com"));
959 BEAST_EXPECT(!test("bang!.example.com"));
960
961 // Too short
962 BEAST_EXPECT(!test("a.b"));
963
964 // Single component too long:
965 BEAST_EXPECT(!test(std::string(64, 'a') + ".com"));
966 BEAST_EXPECT(!test(std::string(64, 'a') + ".example.com"));
967
968 // Multiple components too long:
969 BEAST_EXPECT(!test(std::string(64, 'a') + "." + std::string(64, 'b')));
970 BEAST_EXPECT(!test(std::string(64, 'a') + "." + std::string(64, 'b')));
971
972 // Overall too long:
973 BEAST_EXPECT(!test(
974 std::string(63, 'a') + "." + std::string(63, 'b') +
975 ".example.com"));
