41 STAmount
42 roundSelf(STAmount const& amount)
43 {
44 if (amount.native())
45 return amount;
46
47 std::uint64_t mantissa = amount.mantissa();
48 std::uint64_t valueDigits = mantissa % 1000000000;
49
50 if (valueDigits == 1)
51 {
52 mantissa--;
53
54 if (mantissa < STAmount::cMinValue)
55 return {
56 amount.issue(),
57 mantissa,
58 amount.exponent(),
59 amount.negative()};
60
61 return {
62 amount.issue(),
63 mantissa,
64 amount.exponent(),
65 amount.negative(),
66 STAmount::unchecked{}};
67 }
68
69 if (valueDigits == 999999999)
70 {
71 mantissa++;
72
73 if (mantissa > STAmount::cMaxValue)
74 return {
75 amount.issue(),
76 mantissa,
77 amount.exponent(),
78 amount.negative()};
79
80 return {
81 amount.issue(),
82 mantissa,
83 amount.exponent(),
84 amount.negative(),
85 STAmount::unchecked{}};
86 }
87
88 return amount;
89 }

91 void
92 roundTest(int n, int d, int m)
93 {
94 // check STAmount rounding
95 STAmount num(noIssue(), n);
96 STAmount den(noIssue(), d);
97 STAmount mul(noIssue(), m);
98 STAmount quot = divide(STAmount(n), STAmount(d), noIssue());
99 STAmount res = roundSelf(multiply(quot, mul, noIssue()));
100
101 BEAST_EXPECT(!res.native());
102
103 STAmount cmp(noIssue(), (n * m) / d);
104
105 BEAST_EXPECT(!cmp.native());
106
107 BEAST_EXPECT(cmp.issue().currency == res.issue().currency);
108
109 if (res != cmp)
110 {
111 log << "(" << num.getText() << "/" << den.getText() << ") X "
112 << mul.getText() << " = " << res.getText() << " not "
113 << cmp.getText();
114 fail("Rounding");
115 return;
116 }
