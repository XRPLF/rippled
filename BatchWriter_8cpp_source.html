39void
40BatchWriter::store(std::shared_ptr<NodeObject> const& object)
41{
42 std::unique_lock<decltype(mWriteMutex)> sl(mWriteMutex);
43
44 // If the batch has reached its limit, we wait
45 // until the batch writer is finished
46 while (mWriteSet.size() >= batchWriteLimitSize)
47 mWriteCondition.wait(sl);
48
49 mWriteSet.push_back(object);
50
51 if (!mWritePending)
52 {
53 mWritePending = true;
54
55 m_scheduler.scheduleTask(*this);
56 }
57}

73void
74BatchWriter::writeBatch()
75{
76 for (;;)
77 {
78 std::vector<std::shared_ptr<NodeObject>> set;
79
80 set.reserve(batchWritePreallocationSize);
81
82 {
83 std::lock_guard sl(mWriteMutex);
84
85 mWriteSet.swap(set);
86 XRPL_ASSERT(
87 mWriteSet.empty(),
88 "ripple::NodeStore::BatchWriter::writeBatch : writes not set");
89 mWriteLoad = set.size();
90
91 if (set.empty())
92 {
93 mWritePending = false;
94 mWriteCondition.notify_all();
95
96 // VFALCO NOTE Fix this function to not return from the middle
97 return;
98 }
99 }
100
101 BatchWriteReport report;
102 report.writeCount = set.size();
103 auto const before = std::chrono::steady_clock::now();
104
105 m_callback.writeBatch(set);
106
107 report.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
108 std::chrono::steady_clock::now() - before);
109
110 m_scheduler.onBatchWrite(report);
111 }
112}
