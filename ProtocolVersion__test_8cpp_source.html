25class ProtocolVersion_test : public beast::unit_test::suite
26{
27private:
28 void
29 check(std::string const& s, std::string const& answer)
30 {
31 auto join = [](auto first, auto last) {
32 std::string result;
33 if (first != last)
34 {
35 result = to_string(*first++);
36
37 while (first != last)
38 result += "," + to_string(*first++);
39 }
40 return result;
41 };
42
43 auto const result = parseProtocolVersions(s);
44 BEAST_EXPECT(join(result.begin(), result.end()) == answer);
45 }
46
47public:
48 void
49 run() override
50 {
51 testcase("Convert protocol version to string");
52 BEAST_EXPECT(to_string(make_protocol(1, 3)) == "XRPL/1.3");
53 BEAST_EXPECT(to_string(make_protocol(2, 0)) == "XRPL/2.0");
54 BEAST_EXPECT(to_string(make_protocol(2, 1)) == "XRPL/2.1");
55 BEAST_EXPECT(to_string(make_protocol(10, 10)) == "XRPL/10.10");
56
57 {
58 testcase("Convert strings to protocol versions");
59
60 // Empty string
61 check("", "");
62
63 // clang-format off
64 check(
65 "RTXP/1.1,RTXP/1.2,RTXP/1.3,XRPL/2.1,XRPL/2.0,/XRPL/3.0",
66 "XRPL/2.0,XRPL/2.1");
67 check(
68 "RTXP/0.9,RTXP/1.01,XRPL/0.3,XRPL/2.01,websocket",
69 "");
70 check(
71 "XRPL/2.0,XRPL/2.0,XRPL/19.4,XRPL/7.89,XRPL/XRPL/3.0,XRPL/2.01",
72 "XRPL/2.0,XRPL/7.89,XRPL/19.4");
73 check(
74 "XRPL/2.0,XRPL/3.0,XRPL/4,XRPL/,XRPL,OPT XRPL/2.2,XRPL/5.67",
75 "XRPL/2.0,XRPL/3.0,XRPL/5.67");
76 // clang-format on
77 }
78
79 {
80 testcase("Protocol version negotiation");
81
82 BEAST_EXPECT(negotiateProtocolVersion("RTXP/1.2") == std::nullopt);
83 BEAST_EXPECT(
84 negotiateProtocolVersion("RTXP/1.2, XRPL/2.0, XRPL/2.1") ==
85 make_protocol(2, 1));
86 BEAST_EXPECT(
87 negotiateProtocolVersion("XRPL/2.2") == make_protocol(2, 2));
88 BEAST_EXPECT(
89 negotiateProtocolVersion(
90 "RTXP/1.2, XRPL/2.2, XRPL/2.3, XRPL/999.999") ==
91 make_protocol(2, 2));
92 BEAST_EXPECT(
93 negotiateProtocolVersion("XRPL/999.999, WebSocket/1.0") ==
94 std::nullopt);
95 BEAST_EXPECT(negotiateProtocolVersion("") == std::nullopt);
96 }
97 }
98};
