26std::string
27Status::codeString() const
28{
29 if (!*this)
30 return "";
31
32 if (type_ == Type::none)
33 return std::to_string(code_);
34
35 if (type_ == Status::Type::TER)
36 {
37 std::string s1, s2;
38
39 [[maybe_unused]] auto const success = transResultInfo(toTER(), s1, s2);
40 XRPL_ASSERT(success, "ripple::RPC::codeString : valid TER result");
41
42 return s1 + ": " + s2;
43 }
44
45 if (type_ == Status::Type::error_code_i)
46 {
47 auto info = get_error_info(toErrorCode());
48 std::ostringstream sStr;
49 sStr << info.token.c_str() << ": " << info.message.c_str();
50 return sStr.str();
51 }
52
53 UNREACHABLE("ripple::RPC::codeString : invalid type");
54 return "";
55}

57void
58Status::fillJson(Json::Value& value)
59{
60 if (!*this)
61 return;
62
63 auto& error = value[jss::error];
64 error[jss::code] = code_;
65 error[jss::message] = codeString();
66
67 // Are there any more messages?
68 if (!messages_.empty())
69 {
70 auto& messages = error[jss::data];
71 for (auto& i : messages_)
72 messages.append(i);
73 }
74}

76std::string
77Status::message() const
78{
79 std::string result;
80 for (auto& m : messages_)
81 {
82 if (!result.empty())
83 result += '/';
84 result += m;
85 }
86
87 return result;
88}
