28 std::string Status::codeString () const
29 {
30  if (!*this)
31  return "";
32 
33  if (type_ == Type::none)
34  return std::to_string (code_);
35 
36  if (type_ == Status::Type::TER)
37  {
38  std::string s1, s2;
39 
40  auto success = transResultInfo (toTER (), s1, s2);
41  assert (success);
42  (void) success;
43 
44  return s1 + ": " + s2;
45  }
46 
47  if (type_ == Status::Type::error_code_i)
48  {
49  auto info = get_error_info (toErrorCode ());
50  std::ostringstream sStr;
51  sStr << info.token.c_str() << ": " << info.message.c_str();
52  return sStr.str();
53  }
54 
55  assert (false);
56  return "";
57 }
