61 void
62 TransactionAcquire::done()
63 {
64  // We hold a PeerSet lock and so cannot do real work here
65 
66  if (failed_)
67  {
68  JLOG(journal_.warn()) << "Failed to acquire TX set " << hash_;
69  }
70  else
71  {
72  JLOG(journal_.debug()) << "Acquired TX set " << hash_;
73  mMap->setImmutable();
74 
75  uint256 const& hash(hash_);
76  std::shared_ptr<SHAMap> const& map(mMap);
77  auto const pap = &app_;
78  // Note that, when we're in the process of shutting down, addJob()
79  // may reject the request. If that happens then giveSet() will
80  // not be called. That's fine. According to David the giveSet() call
81  // just updates the consensus and related structures when we acquire
82  // a transaction set. No need to update them if we're shutting down.
83  app_.getJobQueue().addJob(
84  jtTXN_DATA, "completeAcquire", [pap, hash, map](Job&) {
85  pap->getInboundTransactions().giveSet(hash, map, true);
86  });
87  }
88 }

112 void
113 TransactionAcquire::trigger(std::shared_ptr<Peer> const& peer)
114 {
115  if (complete_)
116  {
117  JLOG(journal_.info()) << "trigger after complete";
118  return;
119  }
120  if (failed_)
121  {
122  JLOG(journal_.info()) << "trigger after fail";
123  return;
124  }
125 
126  if (!mHaveRoot)
127  {
128  JLOG(journal_.trace()) << "TransactionAcquire::trigger "
129  << (peer ? "havePeer" : "noPeer") << " no root";
130  protocol::TMGetLedger tmGL;
131  tmGL.set_ledgerhash(hash_.begin(), hash_.size());
132  tmGL.set_itype(protocol::liTS_CANDIDATE);
133  tmGL.set_querydepth(3); // We probably need the whole thing
134 
135  if (timeouts_ != 0)
136  tmGL.set_querytype(protocol::qtINDIRECT);
137 
138  *(tmGL.add_nodeids()) = SHAMapNodeID().getRawString();
139  mPeerSet->sendRequest(tmGL, peer);
140  }
141  else if (!mMap->isValid())
142  {
143  failed_ = true;
144  done();
145  }
146  else
147  {
148  ConsensusTransSetSF sf(app_, app_.getTempNodeCache());
149  auto nodes = mMap->getMissingNodes(256, &sf);
150 
151  if (nodes.empty())
152  {
153  if (mMap->isValid())
154  complete_ = true;
155  else
156  failed_ = true;
157 
158  done();
159  return;
160  }
161 
162  protocol::TMGetLedger tmGL;
163  tmGL.set_ledgerhash(hash_.begin(), hash_.size());
164  tmGL.set_itype(protocol::liTS_CANDIDATE);
165 
166  if (timeouts_ != 0)
167  tmGL.set_querytype(protocol::qtINDIRECT);
168 
169  for (auto const& node : nodes)
170  {
171  *tmGL.add_nodeids() = node.first.getRawString();
172  }
173  mPeerSet->sendRequest(tmGL, peer);
174  }
175 }

177 SHAMapAddNode
178 TransactionAcquire::takeNodes(
179  const std::list<SHAMapNodeID>& nodeIDs,
180  const std::list<Blob>& data,
181  std::shared_ptr<Peer> const& peer)
182 {
183  ScopedLockType sl(mtx_);
184 
185  if (complete_)
186  {
187  JLOG(journal_.trace()) << "TX set complete";
188  return SHAMapAddNode();
189  }
190 
191  if (failed_)
192  {
193  JLOG(journal_.trace()) << "TX set failed";
194  return SHAMapAddNode();
195  }
196 
197  try
198  {
199  if (nodeIDs.empty())
200  return SHAMapAddNode::invalid();
201 
202  std::list<SHAMapNodeID>::const_iterator nodeIDit = nodeIDs.begin();
203  std::list<Blob>::const_iterator nodeDatait = data.begin();
204  ConsensusTransSetSF sf(app_, app_.getTempNodeCache());
205 
206  while (nodeIDit != nodeIDs.end())
207  {
208  if (nodeIDit->isRoot())
209  {
210  if (mHaveRoot)
211  JLOG(journal_.debug())
212  << "Got root TXS node, already have it";
213  else if (!mMap->addRootNode(
214  SHAMapHash{hash_},
215  makeSlice(*nodeDatait),
216  nullptr)
217  .isGood())
218  {
219  JLOG(journal_.warn()) << "TX acquire got bad root node";
220  }
221  else
222  mHaveRoot = true;
223  }
224  else if (!mMap->addKnownNode(*nodeIDit, makeSlice(*nodeDatait), &sf)
225  .isGood())
226  {
227  JLOG(journal_.warn()) << "TX acquire got bad non-root node";
228  return SHAMapAddNode::invalid();
229  }
230 
231  ++nodeIDit;
232  ++nodeDatait;
233  }
234 
235  trigger(peer);
236  progress_ = true;
237  return SHAMapAddNode::useful();
238  }
239  catch (std::exception const&)
240  {
241  JLOG(journal_.error()) << "Peer sends us junky transaction node data";
242  return SHAMapAddNode::invalid();
243  }
244 }
