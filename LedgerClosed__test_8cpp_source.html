30 testMonitorRoot()

37 auto lc_result = env.rpc("ledger_closed")[jss::result];

42
43 env.close();
44 auto const ar_master = env.le(env.master);
45 BEAST_EXPECT(ar_master->getAccountID(sfAccount) == env.master.id());
46 BEAST_EXPECT((*ar_master)[sfBalance] == drops(99999989999999980));
47
48 auto const ar_alice = env.le(alice);
49 BEAST_EXPECT(ar_alice->getAccountID(sfAccount) == alice.id());
50 BEAST_EXPECT((*ar_alice)[sfBalance] == XRP(10000));
51
52 lc_result = env.rpc("ledger_closed")[jss::result];
53 BEAST_EXPECT(
54 lc_result[jss::ledger_hash] ==
55 "E86DE7F3D7A4D9CE17EF7C8BA08A8F4D8F643B9552F0D895A31CDA78F541DE4E");
56 BEAST_EXPECT(lc_result[jss::ledger_index] == 3);
57 }
