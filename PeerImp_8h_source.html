121 // Notes on thread locking:
122 //
123 // During an audit it was noted that some member variables that looked
124 // like they need thread protection were not receiving it. And, indeed,
125 // that was correct. But the multi-phase initialization of PeerImp
126 // makes such an audit difficult. A further audit suggests that the
127 // locking is now protecting variables that don't need it. We're
128 // leaving that locking in place (for now) as a form of future proofing.
129 //
130 // Here are the variables that appear to need locking currently:

132 // o closedLedgerHash_

134 // o minLedger_
135 // o maxLedger_

137 // o recentTxSets_
138 // o trackingTime_
139 // o latency_

143 // o name_
144 // o last_status_

639 Resource::Consumer usage,
640 PublicKey const& publicKey,
641 ProtocolVersion protocol,
642 id_t id,

645 , app_(app)
646 , id_(id)
647 , sink_(app_.journal("Peer"), makePrefix(id))
648 , p_sink_(app_.journal("Protocol"), makePrefix(id))
649 , journal_(sink_)
650 , p_journal_(p_sink_)
651 , stream_ptr_(std::move(stream_ptr))
652 , socket_(stream_ptr_->next_layer().socket())
653 , stream_(*stream_ptr_)
654 , strand_(socket_.get_executor())
655 , timer_(waitable_timer{socket_.get_executor()})
656 , remote_address_(slot->remote_endpoint())
657 , overlay_(overlay)
658 , inbound_(false)
659 , protocol_(protocol)
660 , tracking_(Tracking::unknown)
661 , trackingTime_(clock_type::now())
662 , publicKey_(publicKey)
663 , lastPingTime_(clock_type::now())
664 , creationTime_(clock_type::now())
665 , squelch_(app_.journal("Squelch"))
666 , usage_(usage)
667 , fee_(Resource::feeLightPeer)
668 , slot_(std::move(slot))
669 , response_(std::move(response))
670 , headers_(response_)
671 , compressionEnabled_(
672 peerFeatureEnabled(
673 headers_,
674 FEATURE_COMPR,
675 "lz4",
676 app_.config().COMPRESSION)
677 ? Compressed::On
678 : Compressed::Off)
679 , txReduceRelayEnabled_(peerFeatureEnabled(
680 headers_,
681 FEATURE_TXRR,
682 app_.config().TX_REDUCE_RELAY_ENABLE))
683 , vpReduceRelayEnabled_(peerFeatureEnabled(
684 headers_,
685 FEATURE_VPRR,
686 app_.config().VP_REDUCE_RELAY_ENABLE))
687 , ledgerReplayEnabled_(peerFeatureEnabled(
688 headers_,
689 FEATURE_LEDGER_REPLAY,
690 app_.config().LEDGER_REPLAY))
691 , ledgerReplayMsgHandler_(app, app.getLedgerReplayer())
692{
693 read_buffer_.commit(boost::asio::buffer_copy(
694 read_buffer_.prepare(boost::asio::buffer_size(buffers)), buffers));
695 JLOG(journal_.info()) << "compression enabled "
696 << (compressionEnabled_ == Compressed::On)
697 << " vp reduce-relay enabled "
698 << vpReduceRelayEnabled_
699 << " tx reduce-relay enabled "
700 << txReduceRelayEnabled_ << " on " << remote_address_
701 << " " << id_;
702}

704template <class FwdIt, class>
705void
706PeerImp::sendEndpoints(FwdIt first, FwdIt last)
707{
708 protocol::TMEndpoints tm;
709
710 while (first != last)
711 {
712 auto& tme2(*tm.add_endpoints_v2());
713 tme2.set_endpoint(first->address.to_string());
714 tme2.set_hops(first->hops);
715 first++;
716 }
717 tm.set_version(2);
718
719 send(std::make_shared<Message>(tm, protocol::mtENDPOINTS));
720}
