68 inline FeatureBitset
69 supported_amendments()
70 {
71  static const FeatureBitset ids = [] {
72  auto const& sa = ripple::detail::supportedAmendments();
73  std::vector<uint256> feats;
74  feats.reserve(sa.size());
75  for (auto const& s : sa)
76  {
77  if (auto const f = getRegisteredFeature(s))
78  feats.push_back(*f);
79  else
80  Throw<std::runtime_error>(
81  "Unknown feature: " + s + " in supportedAmendments.");
82  }
83  return FeatureBitset(feats);
84  }();
85  return ids;
86 }
