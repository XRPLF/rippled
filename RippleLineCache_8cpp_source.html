41std::shared_ptr<std::vector<PathFindTrustLine>>
42RippleLineCache::getRippleLines(
43 AccountID const& accountID,
44 LineDirection direction)
45{
46 auto const hash = hasher_(accountID);
47 AccountKey key(accountID, direction, hash);
48 AccountKey otherkey(
49 accountID,
50 direction == LineDirection::outgoing ? LineDirection::incoming
51 : LineDirection::outgoing,
52 hash);
53
54 std::lock_guard sl(mLock);
55
56 auto [it, inserted] = [&]() {
57 if (auto otheriter = lines_.find(otherkey); otheriter != lines_.end())
58 {
59 // The whole point of using the direction flag is to reduce the
60 // number of trust line objects held in memory. Ensure that there is
61 // only a single set of trustlines in the cache per account.
62 auto const size = otheriter->second ? otheriter->second->size() : 0;
63 JLOG(journal_.info())
64 << "Request for "
65 << (direction == LineDirection::outgoing ? "outgoing"
66 : "incoming")
67 << " trust lines for account " << accountID << " found " << size
68 << (direction == LineDirection::outgoing ? " incoming"
69 : " outgoing")
70 << " trust lines. "
71 << (direction == LineDirection::outgoing
72 ? "Deleting the subset of incoming"
73 : "Returning the superset of outgoing")
74 << " trust lines. ";
75 if (direction == LineDirection::outgoing)
76 {
77 // This request is for the outgoing set, but there is already a
78 // subset of incoming lines in the cache. Erase that subset
79 // to be replaced by the full set. The full set will be built
80 // below, and will be returned, if needed, on subsequent calls
81 // for either value of outgoing.
82 XRPL_ASSERT(
83 size <= totalLineCount_,
84 "ripple::RippleLineCache::getRippleLines : maximum lines");
85 totalLineCount_ -= size;
86 lines_.erase(otheriter);
87 }
88 else
89 {
90 // This request is for the incoming set, but there is
91 // already a superset of the outgoing trust lines in the cache.
92 // The path finding engine will disregard the non-rippling trust
93 // lines, so to prevent them from being stored twice, return the
94 // outgoing set.
95 key = otherkey;
96 return std::pair{otheriter, false};
97 }
98 }
99 return lines_.emplace(key, nullptr);
100 }();
101
102 if (inserted)
103 {
104 XRPL_ASSERT(
105 it->second == nullptr,
106 "ripple::RippleLineCache::getRippleLines : null lines");
107 auto lines =
108 PathFindTrustLine::getItems(accountID, *ledger_, direction);
109 if (lines.size())
110 {
111 it->second = std::make_shared<std::vector<PathFindTrustLine>>(
112 std::move(lines));
113 totalLineCount_ += it->second->size();
114 }
115 }
116
117 XRPL_ASSERT(
118 !it->second || (it->second->size() > 0),
119 "ripple::RippleLineCache::getRippleLines : null or nonempty lines");
120 auto const size = it->second ? it->second->size() : 0;
121 JLOG(journal_.trace()) << "getRippleLines for ledger "
122 << ledger_->info().seq << " found " << size
123 << (key.direction_ == LineDirection::outgoing
124 ? " outgoing"
125 : " incoming")
126 << " lines for " << (inserted ? "new " : "existing ")
127 << accountID << " out of a total of "
128 << lines_.size() << " accounts and "
129 << totalLineCount_ << " trust lines";
130
131 return it->second;
132}
