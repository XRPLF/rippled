81 toAmount<IOUAmount>(STAmount const& amt)
82 {
83  assert(amt.mantissa() < std::numeric_limits<std::int64_t>::max());
84  bool const isNeg = amt.negative();
85  std::int64_t const sMant =
86  isNeg ? -std::int64_t(amt.mantissa()) : amt.mantissa();
87 
88  assert(!isXRP(amt));
