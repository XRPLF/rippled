82 class Base
83 {
84 protected:
85 class Child
86 {
87 private:
88 Base& base_;
89
90 public:
91 explicit Child(Base& base) : base_(base)
92 {
93 }
94
95 virtual ~Child()
96 {
97 base_.remove(this);
98 }
99
100 virtual void
101 close() = 0;
102 };
103
104 private:
105 std::mutex mutex_;
106 std::condition_variable cond_;
107 std::map<Child*, std::weak_ptr<Child>> list_;
108 bool closed_ = false;
109
110 public:
111 ~Base()
112 {
113 // Derived class must call wait() in the destructor
114 assert(list_.empty());
115 }
116
117 void
118 add(std::shared_ptr<Child> const& child)
119 {
120 std::lock_guard lock(mutex_);
121 list_.emplace(child.get(), child);
122 }
123
124 void
125 remove(Child* child)
126 {
127 std::lock_guard lock(mutex_);
128 list_.erase(child);
129 if (list_.empty())
130 cond_.notify_one();
131 }
132
133 void
134 close()
135 {
136 std::vector<std::shared_ptr<Child>> v;
137 {
138 std::lock_guard lock(mutex_);
139 v.reserve(list_.size());
140 if (closed_)
141 return;
142 closed_ = true;
143 for (auto const& c : list_)
144 {
145 if (auto p = c.second.lock())
146 {
147 p->close();
148 // Must destroy shared_ptr outside the
149 // lock otherwise deadlock from the
150 // managed object's destructor.
151 v.emplace_back(std::move(p));
152 }
153 }
154 }
155 }
156
157 void
158 wait()
159 {
160 std::unique_lock<std::mutex> lock(mutex_);
161 while (!list_.empty())
162 cond_.wait(lock);
163 }
164 };

354 void
355 on_read(error_code ec, std::size_t bytes_transferred)
356 {
357 if (ec == boost::asio::error::eof)

359 server_.test_.log << "[server] read: EOF" << std::endl;
360 return stream_.async_shutdown(bind_executor(

365 std::placeholders::_1)));

368 return fail("read", ec);

370 buf_.commit(bytes_transferred);
371 buf_.consume(bytes_transferred);
372 write(buf_, "BYE\n");
373 boost::asio::async_write(
374 stream_,
375 buf_.data(),
376 bind_executor(
377 strand_,
378 std::bind(
379 &Connection::on_write,
380 shared_from_this(),
381 std::placeholders::_1,
382 std::placeholders::_2)));
383 }

392 strand_,
393 std::bind(
394 &Connection::on_shutdown,
395 shared_from_this(),
396 std::placeholders::_1)));
397 }
398
399 void
400 on_shutdown(error_code ec)
401 {
402 if (ec)
403 return fail("shutdown", ec);
404 socket_.close();

406 }
