82  {
83  protected:
84  class Child
85  {
86  private:
87  Base& base_;
88 
89  public:
90  explicit Child(Base& base) : base_(base)
91  {
92  }
93 
94  virtual ~Child()
95  {
96  base_.remove(this);
97  }
98 
99  virtual void
100  close() = 0;
101  };
102 
103  private:
104  std::mutex mutex_;
105  std::condition_variable cond_;
106  std::map<Child*, std::weak_ptr<Child>> list_;
107  bool closed_ = false;
108 
109  public:
110  ~Base()
111  {
112  // Derived class must call wait() in the destructor
113  assert(list_.empty());
114  }
115 
116  void
117  add(std::shared_ptr<Child> const& child)
118  {
119  std::lock_guard lock(mutex_);
120  list_.emplace(child.get(), child);
121  }
122 
123  void
124  remove(Child* child)
125  {
126  std::lock_guard lock(mutex_);
127  list_.erase(child);
128  if (list_.empty())
129  cond_.notify_one();
130  }
131 
132  void
133  close()
134  {
135  std::vector<std::shared_ptr<Child>> v;
136  {
137  std::lock_guard lock(mutex_);
138  v.reserve(list_.size());
139  if (closed_)
140  return;
141  closed_ = true;
142  for (auto const& c : list_)
143  {
144  if (auto p = c.second.lock())
145  {
146  p->close();
147  // Must destroy shared_ptr outside the
148  // lock otherwise deadlock from the
149  // managed object's destructor.
150  v.emplace_back(std::move(p));
151  }
152  }
153  }
154  }
155 
156  void
157  wait()
158  {
159  std::unique_lock<std::mutex> lock(mutex_);
160  while (!list_.empty())
161  cond_.wait(lock);
162  }
163  };
