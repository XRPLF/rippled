25bool
26doVacuumDB(DatabaseCon::Setup const& setup, beast::Journal j)
27{
28 boost::filesystem::path dbPath = setup.dataDir / TxDBName;
29
30 uintmax_t const dbSize = file_size(dbPath);
31 XRPL_ASSERT(
32 dbSize != static_cast<uintmax_t>(-1),
33 "ripple:doVacuumDB : file_size succeeded");
34
35 if (auto available = space(dbPath.parent_path()).available;
36 available < dbSize)
37 {
38 std::cerr << "The database filesystem must have at least as "
39 "much free space as the size of "
40 << dbPath.string() << ", which is " << dbSize
41 << " bytes. Only " << available << " bytes are available.\n";
42 return false;
43 }
44
45 auto txnDB = std::make_unique<DatabaseCon>(
46 setup, TxDBName, setup.txPragma, TxDBInit, j);
47 auto& session = txnDB->getSession();
48 std::uint32_t pageSize;
49
50 // Only the most trivial databases will fit in memory on typical
51 // (recommended) hardware. Force temp files to be written to disk
52 // regardless of the config settings.
53 session << boost::format(CommonDBPragmaTemp) % "file";
54 session << "PRAGMA page_size;", soci::into(pageSize);
55
56 std::cout << "VACUUM beginning. page_size: " << pageSize << std::endl;
57
58 session << "VACUUM;";
59 XRPL_ASSERT(
60 setup.globalPragma, "ripple:doVacuumDB : non-null global pragma");
61 for (auto const& p : *setup.globalPragma)
62 session << p;
63 session << "PRAGMA page_size;", soci::into(pageSize);
64
65 std::cout << "VACUUM finished. page_size: " << pageSize << std::endl;
66
67 return true;
68}
