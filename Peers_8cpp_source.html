31Json::Value
32doPeers(RPC::JsonContext& context)
33{
34 Json::Value jvResult(Json::objectValue);
35
36 jvResult[jss::peers] = context.app.overlay().json();
37
38 // Legacy support
39 if (context.apiVersion == 1)
40 {
41 for (auto& p : jvResult[jss::peers])
42 {
43 if (p.isMember(jss::track))
44 {
45 auto const s = p[jss::track].asString();
46
47 if (s == "diverged")
48 p["sanity"] = "insane";
49 else if (s == "unknown")
50 p["sanity"] = "unknown";
51 }
52 }
53 }
54
55 auto const now = context.app.timeKeeper().now();
56 auto const self = context.app.nodeIdentity().first;
57
58 Json::Value& cluster = (jvResult[jss::cluster] = Json::objectValue);
59 std::uint32_t ref = context.app.getFeeTrack().getLoadBase();
60
61 context.app.cluster().for_each(
62 [&cluster, now, ref, &self](ClusterNode const& node) {
63 if (node.identity() == self)
64 return;
65
66 Json::Value& json =
67 cluster[toBase58(TokenType::NodePublic, node.identity())];
68
69 if (!node.name().empty())
70 json[jss::tag] = node.name();
71
72 if ((node.getLoadFee() != ref) && (node.getLoadFee() != 0))
73 json[jss::fee] = static_cast<double>(node.getLoadFee()) / ref;
74
75 if (node.getReportTime() != NetClock::time_point{})
76 json[jss::age] = (node.getReportTime() >= now)
77 ? 0
78 : (now - node.getReportTime()).count();
79 });
80
81 return jvResult;
82}
