50 template <class FwdIter>
51 float
52 distribution_factor(FwdIter first, FwdIter last)
53 {
54  using value_type = typename FwdIter::value_type;
55  static_assert(std::is_unsigned<value_type>::value, "");
56 
57  const unsigned nbits = CHAR_BIT * sizeof(std::size_t);
58  const unsigned rows = nbits / 4;
59  unsigned counts[rows][16] = {};
60  std::for_each(first, last, [&](typename FwdIter::value_type h) {
61  std::size_t mask = 0xF;
62  for (unsigned i = 0; i < rows; ++i, mask <<= 4)
63  counts[i][(h & mask) >> 4 * i] += 1;
64  });
65  float mean_rows[rows] = {0};
66  float mean_cols[16] = {0};
67  for (unsigned i = 0; i < rows; ++i)
68  {
69  for (unsigned j = 0; j < 16; ++j)
70  {
71  mean_rows[i] += counts[i][j];
72  mean_cols[j] += counts[i][j];
73  }
74  }
75  for (unsigned i = 0; i < rows; ++i)
76  mean_rows[i] /= 16;
77  for (unsigned j = 0; j < 16; ++j)
78  mean_cols[j] /= rows;
79  std::pair<float, float> dev[rows][16];
80  for (unsigned i = 0; i < rows; ++i)
81  {
82  for (unsigned j = 0; j < 16; ++j)
83  {
84  dev[i][j].first =
85  std::abs(counts[i][j] - mean_rows[i]) / mean_rows[i];
86  dev[i][j].second =
87  std::abs(counts[i][j] - mean_cols[j]) / mean_cols[j];
88  }
89  }
90  float max_err = 0;
91  for (unsigned i = 0; i < rows; ++i)
92  {
93  for (unsigned j = 0; j < 16; ++j)
94  {
95  if (max_err < dev[i][j].first)
96  max_err = dev[i][j].first;
97  if (max_err < dev[i][j].second)
98  max_err = dev[i][j].second;
99  }
100  }
101  return max_err;
102 }

115 double
116 score(int const* bins, std::size_t const bincount, double const k)
117 {
118  double const n = bincount;
119  // compute rms^2 value
120  double rms_sq = 0;
121  for (std::size_t i = 0; i < bincount; ++i)
122  rms_sq += sqr(bins[i]);
123  ;
124  rms_sq /= n;
125  // compute fill factor
126  double const f = (sqr(k) - 1) / (n * rms_sq - k);
127  // rescale to (0,1) with 0 = good, 1 = bad
128  return 1 - (f / n);
129 }

131 template <class T>
132 std::uint32_t
133 window(T* blob, int start, int count)
134 {
135  std::size_t const len = sizeof(T);
136  static_assert((len & 3) == 0, "");
137  if (count == 0)
138  return 0;
139  int const nbits = len * CHAR_BIT;
140  start %= nbits;
141  int ndwords = len / 4;
142  std::uint32_t const* k =
143  static_cast<std::uint32_t const*>(static_cast<void const*>(blob));
144  int c = start & (32 - 1);
145  int d = start / 32;
146  if (c == 0)
147  return (k[d] & ((1 << count) - 1));
148  int ia = (d + 1) % ndwords;
149  int ib = (d + 0) % ndwords;
150  std::uint32_t a = k[ia];
151  std::uint32_t b = k[ib];
152  std::uint32_t t = (a << (32 - c)) | (b >> c);
153  t &= ((1 << count) - 1);
154  return t;
155 }

162 template <class FwdIter>
163 double
164 windowed_score(FwdIter first, FwdIter last)
165 {
166  auto const size(std::distance(first, last));
167  int maxwidth = 20;
168  // We need at least 5 keys per bin to reliably test distribution biases
169  // down to 1%, so don't bother to test sparser distributions than that
170  while (static_cast<double>(size) / (1ull << maxwidth) < 5.0)
171  maxwidth--;
172  double worst = 0;
173  std::vector<int> bins(1ull << maxwidth);
174  int const hashbits = sizeof(std::size_t) * CHAR_BIT;
175  for (int start = 0; start < hashbits; ++start)
176  {
177  int width = maxwidth;
178  bins.assign(1ull << width, 0);
179  for (auto iter(first); iter != last; ++iter)
180  ++bins[detail::window(&*iter, start, width)];
181  // Test the distribution, then fold the bins in half,
182  // repeat until we're down to 256 bins
183  while (bins.size() >= 256)
184  {
185  double score(detail::score(bins.data(), bins.size(), size));
186  worst = std::max(score, worst);
187  if (--width < 8)
188  break;
189  for (std::size_t i = 0, j = bins.size() / 2; j < bins.size();
190  ++i, ++j)
191  bins[i] += bins[j];
192  bins.resize(bins.size() / 2);
193  }
194  }
195  return worst;
196 }
