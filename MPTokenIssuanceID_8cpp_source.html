31bool
32canHaveMPTokenIssuanceID(
33 std::shared_ptr<STTx const> const& serializedTx,
34 TxMeta const& transactionMeta)
35{
36 if (!serializedTx)
37 return false;
38
39 TxType const tt = serializedTx->getTxnType();
40 if (tt != ttMPTOKEN_ISSUANCE_CREATE)
41 return false;
42
43 // if the transaction failed nothing could have been delivered.
44 if (transactionMeta.getResultTER() != tesSUCCESS)
45 return false;
46
47 return true;
48}

50std::optional<uint192>
51getIDFromCreatedIssuance(TxMeta const& transactionMeta)
52{
53 for (STObject const& node : transactionMeta.getNodes())
54 {
55 if (node.getFieldU16(sfLedgerEntryType) != ltMPTOKEN_ISSUANCE ||
56 node.getFName() != sfCreatedNode)
57 continue;
58
59 auto const& mptNode =
60 node.peekAtField(sfNewFields).downcast<STObject>();
61 return makeMptID(
62 mptNode.getFieldU32(sfSequence), mptNode.getAccountID(sfIssuer));
63 }
64
65 return std::nullopt;
66}

68void
69insertMPTokenIssuanceID(
70 Json::Value& response,
71 std::shared_ptr<STTx const> const& transaction,
72 TxMeta const& transactionMeta)
73{
74 if (!canHaveMPTokenIssuanceID(transaction, transactionMeta))
75 return;
76
77 std::optional<uint192> result = getIDFromCreatedIssuance(transactionMeta);
78 if (result)
79 response[jss::mpt_issuance_id] = to_string(result.value());
80}
