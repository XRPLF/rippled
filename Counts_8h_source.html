251 private:
252  // Adjusts counts based on the specified slot, in the direction indicated.
253  void
254  adjust(Slot const& s, int const n)
255  {
256  if (s.fixed())
257  m_fixed += n;
258 
259  if (s.reserved())
260  m_reserved += n;
261 
262  switch (s.state())
263  {
264  case Slot::accept:
265  assert(s.inbound());
266  m_acceptCount += n;
267  break;
268 
269  case Slot::connect:
270  case Slot::connected:
271  assert(!s.inbound());
272  m_attempts += n;
273  break;
274 
275  case Slot::active:
276  if (s.fixed())
277  m_fixed_active += n;
278  if (!s.fixed() && !s.reserved())
279  {
280  if (s.inbound())
281  m_in_active += n;
282  else
283  m_out_active += n;
284  }
285  m_active += n;
286  break;
287 
288  case Slot::closing:
289  m_closingCount += n;
290  break;
291 
292  default:
293  assert(false);
294  break;
295  };
296  }
