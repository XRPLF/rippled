31template <class Generator>
32void
33rngfill(void* buffer, std::size_t bytes, Generator& g)
34{
35 using result_type = typename Generator::result_type;
36
37 while (bytes >= sizeof(result_type))
38 {
39 auto const v = g();
40 std::memcpy(buffer, &v, sizeof(v));
41 buffer = reinterpret_cast<std::uint8_t*>(buffer) + sizeof(v);
42 bytes -= sizeof(v);
43 }
44
45 XRPL_ASSERT(
46 bytes < sizeof(result_type), "beast::rngfill(void*) : maximum bytes");
47
48#ifdef __GNUC__
49 // gcc 11.1 (falsely) warns about an array-bounds overflow in release mode.
50#pragma GCC diagnostic push
51#pragma GCC diagnostic ignored "-Warray-bounds"
52#endif
53
54 if (bytes > 0)
55 {
56 auto const v = g();
57 std::memcpy(buffer, &v, bytes);
58 }
59
60#ifdef __GNUC__
61#pragma GCC diagnostic pop
62#endif
63}

66 class Generator,
67 std::size_t N,
68 class = std::enable_if_t<N % sizeof(typename Generator::result_type) == 0>>
69void
70rngfill(std::array<std::uint8_t, N>& a, Generator& g)
71{
72 using result_type = typename Generator::result_type;
73 auto i = N / sizeof(result_type);
74 result_type* p = reinterpret_cast<result_type*>(a.data());
75 while (i--)
76 *p++ = g();
77}
