29// Used to construct received proposals
30RCLCxPeerPos::RCLCxPeerPos(
31 PublicKey const& publicKey,
32 Slice const& signature,
33 uint256 const& suppression,
34 Proposal&& proposal)
35 : publicKey_(publicKey)
36 , suppression_(suppression)
37 , proposal_(std::move(proposal))
38{
39 // The maximum allowed size of a signature is 72 bytes; we verify
40 // this elsewhere, but we want to be extra careful here:
41 XRPL_ASSERT(
42 signature.size() != 0 && signature.size() <= signature_.capacity(),
43 "ripple::RCLCxPeerPos::RCLCxPeerPos : valid signature size");
44
45 if (signature.size() != 0 && signature.size() <= signature_.capacity())
46 signature_.assign(signature.begin(), signature.end());
47}

67uint256
68proposalUniqueId(
69 uint256 const& proposeHash,
70 uint256 const& previousLedger,
71 std::uint32_t proposeSeq,
72 NetClock::time_point closeTime,
73 Slice const& publicKey,
74 Slice const& signature)
75{
76 Serializer s(512);
77 s.addBitString(proposeHash);
78 s.addBitString(previousLedger);
79 s.add32(proposeSeq);
80 s.add32(closeTime.time_since_epoch().count());
81 s.addVL(publicKey);
82 s.addVL(signature);
83
84 return s.getSHA512Half();
85}
