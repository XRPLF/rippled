52template <class... Cs>
53class Collectors
54{
55 std::tuple<Cs&...> cs;
56
57 template <class C, class E>
58 static void
59 apply(C& c, PeerID who, SimTime when, E e)
60 {
61 c.on(who, when, e);
62 }
63
64 template <std::size_t... Is, class E>
65 static void
66 apply(
67 std::tuple<Cs&...>& cs,
68 PeerID who,
69 SimTime when,
70 E e,
71 std::index_sequence<Is...>)
72 {
73 (..., apply(std::get<Is>(cs), who, when, e));
74 }
75
76public:
81 Collectors(Cs&... cs_) : cs(std::tie(cs_...))
82 {
83 }
84
85 template <class E>
86 void
87 on(PeerID who, SimTime when, E e)
88 {
89 apply(cs, who, when, e, std::index_sequence_for<Cs...>{});
90 }
91};

173struct TxCollector
174{
175 // Counts
176 std::size_t submitted{0};
177 std::size_t accepted{0};
178 std::size_t validated{0};
179
180 struct Tracker
181 {
182 Tx tx;
183 SimTime submitted;
184 std::optional<SimTime> accepted;
185 std::optional<SimTime> validated;
186
187 Tracker(Tx tx_, SimTime submitted_) : tx{tx_}, submitted{submitted_}
188 {
189 }
190 };
191
192 hash_map<Tx::ID, Tracker> txs;
193
194 using Hist = Histogram<SimTime::duration>;
195 Hist submitToAccept;
196 Hist submitToValidate;
197
198 // Ignore most events by default
199 template <class E>
200 void
201 on(PeerID, SimTime when, E const& e)
202 {
203 }
204
205 void
206 on(PeerID who, SimTime when, SubmitTx const& e)
207 {
208 // save first time it was seen
209 if (txs.emplace(e.tx.id(), Tracker{e.tx, when}).second)
210 {
211 submitted++;
212 }
213 }
214
215 void
216 on(PeerID who, SimTime when, AcceptLedger const& e)
217 {
218 for (auto const& tx : e.ledger.txs())
219 {
220 auto it = txs.find(tx.id());
221 if (it != txs.end() && !it->second.accepted)
222 {
223 Tracker& tracker = it->second;
224 tracker.accepted = when;
225 accepted++;
226
227 submitToAccept.insert(*tracker.accepted - tracker.submitted);
228 }
229 }
230 }
231
232 void
233 on(PeerID who, SimTime when, FullyValidateLedger const& e)
234 {
235 for (auto const& tx : e.ledger.txs())
236 {
237 auto it = txs.find(tx.id());
238 if (it != txs.end() && !it->second.validated)
239 {
240 Tracker& tracker = it->second;
241 // Should only validated a previously accepted Tx
242 assert(tracker.accepted);
243
244 tracker.validated = when;
245 validated++;
246 submitToValidate.insert(*tracker.validated - tracker.submitted);
247 }
248 }
249 }
250
251 // Returns the number of txs which were never accepted
252 std::size_t
253 orphaned() const
254 {
255 return std::count_if(txs.begin(), txs.end(), [](auto const& it) {
256 return !it.second.accepted;
257 });
258 }
259
260 // Returns the number of txs which were never validated
261 std::size_t
262 unvalidated() const
263 {
264 return std::count_if(txs.begin(), txs.end(), [](auto const& it) {
265 return !it.second.validated;
266 });
267 }

426struct LedgerCollector
427{
428 std::size_t accepted{0};
429 std::size_t fullyValidated{0};
430
431 struct Tracker
432 {
433 SimTime accepted;
434 std::optional<SimTime> fullyValidated;
435
436 Tracker(SimTime accepted_) : accepted{accepted_}
437 {
438 }
439 };
440
441 hash_map<Ledger::ID, Tracker> ledgers_;
442
443 using Hist = Histogram<SimTime::duration>;
444 Hist acceptToFullyValid;
445 Hist acceptToAccept;
446 Hist fullyValidToFullyValid;
447
448 // Ignore most events by default
449 template <class E>
450 void
451 on(PeerID, SimTime, E const& e)
452 {
453 }
454
455 void
456 on(PeerID who, SimTime when, AcceptLedger const& e)
457 {
458 // First time this ledger accepted
459 if (ledgers_.emplace(e.ledger.id(), Tracker{when}).second)
460 {
461 ++accepted;
462 // ignore jumps?
463 if (e.prior.id() == e.ledger.parentID())
464 {
465 auto const it = ledgers_.find(e.ledger.parentID());
466 if (it != ledgers_.end())
467 {
468 acceptToAccept.insert(when - it->second.accepted);
469 }
470 }
471 }
472 }
473
474 void
475 on(PeerID who, SimTime when, FullyValidateLedger const& e)
476 {
477 // ignore jumps
478 if (e.prior.id() == e.ledger.parentID())
479 {
480 auto const it = ledgers_.find(e.ledger.id());
481 assert(it != ledgers_.end());
482 auto& tracker = it->second;
483 // first time fully validated
484 if (!tracker.fullyValidated)
485 {
486 ++fullyValidated;
487 tracker.fullyValidated = when;
488 acceptToFullyValid.insert(when - tracker.accepted);
489
490 auto const parentIt = ledgers_.find(e.ledger.parentID());
491 if (parentIt != ledgers_.end())
492 {
493 auto& parentTracker = parentIt->second;
494 if (parentTracker.fullyValidated)
495 {
496 fullyValidToFullyValid.insert(
497 when - *parentTracker.fullyValidated);
498 }
499 }
500 }
501 }
502 }
503
504 std::size_t
505 unvalidated() const
506 {
507 return std::count_if(
508 ledgers_.begin(), ledgers_.end(), [](auto const& it) {
509 return !it.second.fullyValidated;
510 });
511 }

679struct JumpCollector
680{

682 {
683 PeerID id;
684 SimTime when;
685 Ledger from;
686 Ledger to;

688
689 std::vector<Jump> closeJumps;

700 on(PeerID who, SimTime when, AcceptLedger const& e)

710 // Not a direct child -> parent switch

712 fullyValidatedJumps.emplace_back(
713 Jump{who, when, e.prior, e.ledger});
