24 std::string
25 getFileContents(
26  boost::system::error_code& ec,
27  boost::filesystem::path const& sourcePath,
28  boost::optional<std::size_t> maxSize)
29 {
30  using namespace boost::filesystem;
31  using namespace boost::system::errc;
32 
33  path fullPath{canonical(sourcePath, ec)};
34  if (ec)
35  return {};
36 
37  if (maxSize && (file_size(fullPath, ec) > *maxSize || ec))
38  {
39  if (!ec)
40  ec = make_error_code(file_too_large);
41  return {};
42  }
43 
44  ifstream fileStream(fullPath, std::ios::in);
45 
46  if (!fileStream)
47  {
48  ec = make_error_code(static_cast<errc_t>(errno));
49  return {};
50  }
51 
52  const std::string result{
53  std::istreambuf_iterator<char>{fileStream},
54  std::istreambuf_iterator<char>{}};
55 
56  if (fileStream.bad())
57  {
58  ec = make_error_code(static_cast<errc_t>(errno));
59  return {};
60  }
61 
62  return result;
63 }
