76 template <class T>
77 error_code_i
78 conditionMet(Condition condition_required, T& context)
79 {
80  if ((condition_required & NEEDS_NETWORK_CONNECTION) &&
81  (context.netOps.getOperatingMode() < OperatingMode::SYNCING))
82  {
83  JLOG(context.j.info()) << "Insufficient network mode for RPC: "
84  << context.netOps.strOperatingMode();
85 
86  return rpcNO_NETWORK;
87  }
88 
89  if (context.app.getOPs().isAmendmentBlocked() &&
90  (condition_required & NEEDS_CURRENT_LEDGER ||
91  condition_required & NEEDS_CLOSED_LEDGER))
92  {
93  return rpcAMENDMENT_BLOCKED;
94  }
95 
96  if (!context.app.config().standalone() &&
97  condition_required & NEEDS_CURRENT_LEDGER)
98  {
99  if (context.ledgerMaster.getValidatedLedgerAge() >
100  Tuning::maxValidatedLedgerAge)
101  {
102  return rpcNO_CURRENT;
103  }
104 
105  auto const cID = context.ledgerMaster.getCurrentLedgerIndex();
106  auto const vID = context.ledgerMaster.getValidLedgerIndex();
107 
108  if (cID + 10 < vID)
109  {
110  JLOG(context.j.debug())
111  << "Current ledger ID(" << cID
112  << ") is less than validated ledger ID(" << vID << ")";
113  return rpcNO_CURRENT;
114  }
115  }
116 
117  if ((condition_required & NEEDS_CLOSED_LEDGER) &&
118  !context.ledgerMaster.getClosedLedger())
119  {
120  return rpcNO_CLOSED;
121  }
122 
123  return rpcSUCCESS;
124 }
