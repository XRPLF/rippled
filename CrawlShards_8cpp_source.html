43 doCrawlShards(RPC::JsonContext& context)
44 {
45  if (context.role != Role::ADMIN)
46  return rpcError(rpcNO_PERMISSION);
47 
48  std::uint32_t hops{0};
49  if (auto const& jv = context.params[jss::limit])
50  {
51  if (!(jv.isUInt() || (jv.isInt() && jv.asInt() >= 0)))
52  {
53  return RPC::expected_field_error(jss::limit, "unsigned integer");
54  }
55 
56  hops = std::min(jv.asUInt(), csHopLimit);
57  }
58 
59  bool const pubKey{
60  context.params.isMember(jss::public_key) &&
61  context.params[jss::public_key].asBool()};
62 
63  // Collect shard info from peers connected to this server
64  Json::Value jvResult{context.app.overlay().crawlShards(pubKey, hops)};
65 
66  // Collect shard info from this server
67  if (auto shardStore = context.app.getShardStore())
68  {
69  if (pubKey)
70  jvResult[jss::public_key] = toBase58(
71  TokenType::NodePublic, context.app.nodeIdentity().first);
72  jvResult[jss::complete_shards] = shardStore->getCompleteShards();
73  }
74 
75  if (hops == 0)
76  context.loadType = Resource::feeMediumBurdenRPC;
77  else
78  context.loadType = Resource::feeHighBurdenRPC;
79 
80  return jvResult;
81 }
