117 class SField
118 {
119 public:
120  enum {
121  sMD_Never = 0x00,
122  sMD_ChangeOrig = 0x01, // original value when it changes
123  sMD_ChangeNew = 0x02, // new value when it changes
124  sMD_DeleteFinal = 0x04, // final value when it is deleted
125  sMD_Create = 0x08, // value when it's created
126  sMD_Always = 0x10, // value when node containing it is affected at all
127  sMD_Default =
128  sMD_ChangeOrig | sMD_ChangeNew | sMD_DeleteFinal | sMD_Create
129  };

131  enum class IsSigning : unsigned char { no, yes };
132  static IsSigning const notSigning = IsSigning::no;
133 
134  int const fieldCode; // (type<<16)|index
135  SerializedTypeID const fieldType; // STI_*
136  int const fieldValue; // Code number for protocol
137  std::string const fieldName;
138  int const fieldMeta;
139  int const fieldNum;
140  IsSigning const signingField;
141  Json::StaticString const jsonName;
142 
143  SField(SField const&) = delete;
144  SField&
145  operator=(SField const&) = delete;
146  SField(SField&&) = delete;
147  SField&
148  operator=(SField&&) = delete;
149 
150 public:
151  struct private_access_tag_t; // public, but still an implementation detail
152 
153  // These constructors can only be called from SField.cpp
154  SField(
155  private_access_tag_t,
156  SerializedTypeID tid,
157  int fv,
158  const char* fn,
159  int meta = sMD_Default,
160  IsSigning signing = IsSigning::yes);
161  explicit SField(private_access_tag_t, int fc);
162 
163  static const SField&
164  getField(int fieldCode);
165  static const SField&
166  getField(std::string const& fieldName);
167  static const SField&
168  getField(int type, int value)
169  {
170  return getField(field_code(type, value));
171  }
172 
173  static const SField&
174  getField(SerializedTypeID type, int value)
175  {
176  return getField(field_code(type, value));
177  }
178 
179  std::string const&
180  getName() const
181  {
182  return fieldName;
183  }
184 
185  bool
186  hasName() const
187  {
188  return fieldCode > 0;
189  }
190 
191  Json::StaticString const&
192  getJsonName() const
193  {
194  return jsonName;
195  }
196 
197  bool
198  isInvalid() const
199  {
200  return fieldCode == -1;
201  }
202 
203  bool
204  isUseful() const
205  {
206  return fieldCode > 0;
207  }
208 
209  bool
210  isBinary() const
211  {
212  return fieldValue < 256;
213  }
214 
215  // A discardable field is one that cannot be serialized, and
216  // should be discarded during serialization,like 'hash'.
217  // You cannot serialize an object's hash inside that object,
218  // but you can have it in the JSON representation.
219  bool
220  isDiscardable() const
221  {
222  return fieldValue > 256;
223  }
224 
225  int
226  getCode() const
227  {
228  return fieldCode;
229  }
230  int
231  getNum() const
232  {
233  return fieldNum;
234  }
235  static int
236  getNumFields()
237  {
238  return num;
239  }
240 
241  bool
242  shouldMeta(int c) const
243  {
244  return (fieldMeta & c) != 0;
245  }
246 
247  bool
248  shouldInclude(bool withSigningField) const
249  {
250  return (fieldValue < 256) &&
251  (withSigningField || (signingField == IsSigning::yes));
252  }
253 
254  bool
255  operator==(const SField& f) const
256  {
257  return fieldCode == f.fieldCode;
258  }
259 
260  bool
261  operator!=(const SField& f) const
262  {
263  return fieldCode != f.fieldCode;
264  }
265 
266  static int
267  compare(const SField& f1, const SField& f2);
268 
269 private:
270  static int num;
271  static std::map<int, SField const*> knownCodeToField;
272 };
