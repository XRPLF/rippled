28template <class FwdIt>
29std::string
30strHex(FwdIt begin, FwdIt end)
31{
32 static_assert(
33 std::is_convertible<
34 typename std::iterator_traits<FwdIt>::iterator_category,
35 std::forward_iterator_tag>::value,
36 "FwdIt must be a forward iterator");
37 std::string result;
38 result.reserve(2 * std::distance(begin, end));
39 boost::algorithm::hex(begin, end, std::back_inserter(result));
40 return result;
41}
