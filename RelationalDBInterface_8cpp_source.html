39 std::unique_ptr<RelationalDBInterface>
40 RelationalDBInterface::init(
41  Application& app,
42  Config const& config,
43  JobQueue& jobQueue)
44 {
45  bool use_sqlite = false;
46  bool use_postgres = false;
47 
48  if (config.reporting())
49  {
50  use_postgres = true;
51  }
52  else
53  {
54  const Section& rdb_section{config.section(SECTION_RELATIONAL_DB)};
55  if (!rdb_section.empty())
56  {
57  if (boost::iequals(get(rdb_section, "backend"), "sqlite"))
58  {
59  use_sqlite = true;
60  }
61  else
62  {
63  Throw<std::runtime_error>(
64  "Invalid rdb_section backend value: " +
65  get(rdb_section, "backend"));
66  }
67  }
68  else
69  {
70  use_sqlite = true;
71  }
72  }
73 
74  if (use_sqlite)
75  {
76  return getRelationalDBInterfaceSqlite(app, config, jobQueue);
77  }
78  else if (use_postgres)
79  {
80  return getRelationalDBInterfacePostgres(app, config, jobQueue);
81  }
82 
83  return std::unique_ptr<RelationalDBInterface>();
84 }
