182 io_list::work::destroy()
183 {
184  if (!ios_)
185  return;
186  std::function<void(void)> f;
187  {

189  ios_->map_.erase(this);

195  }

233 template <class Finisher>
234 void
235 io_list::close(Finisher&& f)
236 {
237  std::unique_lock<std::mutex> lock(m_);
238  if (closed_)
239  return;
240  closed_ = true;
241  auto map = std::move(map_);
242  if (!map.empty())
243  {
244  f_ = std::forward<Finisher>(f);
245  lock.unlock();
246  for (auto const& p : map)
247  if (auto sp = p.second.lock())
248  sp->close();
249  }
250  else
251  {
252  lock.unlock();
253  f();
254  }
