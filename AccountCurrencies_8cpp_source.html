39 for (auto const& rspEntry : *lines)

41 auto& saBalance = rspEntry.getBalance();
42
43 // Filter out non
44 if (saBalance > beast::zero
45 // Have IOUs to send.
46 ||
47 (rspEntry.getLimitPeer()

53 }
54 }
