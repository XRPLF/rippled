40void
41flags::operator()(Env& env) const
42{
43 auto const sle = env.le(account_);
44 if (!sle)
45 env.test.fail();
46 else if (sle->isFieldPresent(sfFlags))
47 env.test.expect((sle->getFieldU32(sfFlags) & mask_) == mask_);

49 env.test.expect(mask_ == 0);
