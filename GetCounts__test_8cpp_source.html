28 class GetCounts_test : public beast::unit_test::suite
29 {
30  void
31  testGetCounts()
32  {
33  using namespace test::jtx;
34  Env env(*this);
35 
36  Json::Value result;
37  {
38  // check counts with no transactions posted
39  result = env.rpc("get_counts")[jss::result];
40  BEAST_EXPECT(result[jss::status] == "success");
41  BEAST_EXPECT(!result.isMember("Transaction"));
42  BEAST_EXPECT(!result.isMember("STObject"));
43  BEAST_EXPECT(!result.isMember("HashRouterEntry"));
44  BEAST_EXPECT(
45  result.isMember(jss::uptime) &&
46  !result[jss::uptime].asString().empty());
47  BEAST_EXPECT(
48  result.isMember(jss::dbKBTotal) &&
49  result[jss::dbKBTotal].asInt() > 0);
50  }
51 
52  // create some transactions
53  env.close();
54  Account alice{"alice"};
55  Account bob{"bob"};
56  env.fund(XRP(10000), alice, bob);
57  env.trust(alice["USD"](1000), bob);
58  for (auto i = 0; i < 20; ++i)
59  {
60  env(pay(alice, bob, alice["USD"](5)));
61  env.close();
62  }
63 
64  {
65  // check counts, default params
66  result = env.rpc("get_counts")[jss::result];
67  BEAST_EXPECT(result[jss::status] == "success");
68  // compare with values reported by CountedObjects
69  auto const& objectCounts =
70  CountedObjects::getInstance().getCounts(10);
71  for (auto const& it : objectCounts)
72  {
73  BEAST_EXPECTS(result.isMember(it.first), it.first);
74  BEAST_EXPECTS(result[it.first].asInt() == it.second, it.first);
75  }
76  BEAST_EXPECT(!result.isMember(jss::local_txs));
77  }
78 
79  {
80  // make request with min threshold 100 and verify
81  // that only STObject and NodeObject are reported
82  result = env.rpc("get_counts", "100")[jss::result];
83  BEAST_EXPECT(result[jss::status] == "success");
84 
85  // compare with values reported by CountedObjects
86  auto const& objectCounts =
87  CountedObjects::getInstance().getCounts(100);
88  for (auto const& it : objectCounts)
89  {
90  BEAST_EXPECTS(result.isMember(it.first), it.first);
91  BEAST_EXPECTS(result[it.first].asInt() == it.second, it.first);
92  }
93  BEAST_EXPECT(!result.isMember("Transaction"));
94  BEAST_EXPECT(!result.isMember("STTx"));
95  BEAST_EXPECT(!result.isMember("STArray"));
96  BEAST_EXPECT(!result.isMember("HashRouterEntry"));
97  BEAST_EXPECT(!result.isMember("STLedgerEntry"));
98  }
99 
100  {
101  // local_txs field will exist when there are open Txs
102  env(pay(alice, bob, alice["USD"](5)));
103  result = env.rpc("get_counts")[jss::result];
104  // deliberately don't call close so we have open Tx
105  BEAST_EXPECT(
106  result.isMember(jss::local_txs) &&
107  result[jss::local_txs].asInt() > 0);
108  }
109  }
