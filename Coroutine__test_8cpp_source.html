39  public:
40  // Thread safe, blocks until signaled or period expires.
41  // Returns `true` if signaled.
42  template <class Rep, class Period>
43  bool
44  wait_for(std::chrono::duration<Rep, Period> const& rel_time)
45  {
46  std::unique_lock<std::mutex> lk(mutex_);
47  auto b = cv_.wait_for(lk, rel_time, [=]{ return signaled_; });
48  signaled_ = false;
49  return b;
50  }
51 
52  void
53  signal()
54  {
55  std::lock_guard lk(mutex_);
56  signaled_ = true;
57  cv_.notify_all();
58  }
59  };
