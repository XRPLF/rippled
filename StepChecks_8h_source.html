63 checkNoRipple (
64  ReadView const& view,
65  AccountID const& prev,
66  AccountID const& cur,
67  // This is the account whose constraints we are checking
68  AccountID const& next,
69  Currency const& currency,
70  beast::Journal j)
71 {
72  // fetch the ripple lines into and out of this node
73  auto sleIn = view.read (keylet::line (prev, cur, currency));
74  auto sleOut = view.read (keylet::line (cur, next, currency));
75 
76  if (!sleIn || !sleOut)
77  return terNO_LINE;
78 
79  if ((*sleIn)[sfFlags] &
80  ((cur > prev) ? lsfHighNoRipple : lsfLowNoRipple) &&
81  (*sleOut)[sfFlags] &
82  ((cur > next) ? lsfHighNoRipple : lsfLowNoRipple))
83  {
84  JLOG (j.info()) << "Path violates noRipple constraint between " << prev
85  << ", " << cur << " and " << next;
86 
87  return terNO_RIPPLE;
