61void
62Job::doJob()
63{
64 beast::setCurrentThreadName("doJob: " + mName);
65 m_loadEvent->start();
66 m_loadEvent->setName(mName);
67
68 mJob();
69
70 // Destroy the lambda, otherwise we won't include
71 // its duration in the time measurement
72 mJob = nullptr;
73}
