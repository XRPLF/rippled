32// {
33// ip: <string>,
34// port: <number>
35// }
36// XXX Might allow domain for manual connections.
37Json::Value
38doConnect(RPC::JsonContext& context)
39{
40 if (context.app.config().standalone())
41 {
42 return RPC::make_error(rpcNOT_SYNCED);
43 }
44
45 if (!context.params.isMember(jss::ip))
46 return RPC::missing_field_error(jss::ip);
47
48 if (context.params.isMember(jss::port) &&
49 !context.params[jss::port].isConvertibleTo(Json::intValue))
50 {
51 return rpcError(rpcINVALID_PARAMS);
52 }
53
54 int iPort;
55
56 if (context.params.isMember(jss::port))
57 iPort = context.params[jss::port].asInt();
58 else
59 iPort = DEFAULT_PEER_PORT;
60
61 auto const ip_str = context.params[jss::ip].asString();
62 auto ip = beast::IP::Endpoint::from_string(ip_str);
63
64 if (!is_unspecified(ip))
65 context.app.overlay().connect(ip.at_port(iPort));
66
67 return RPC::makeObjectValue(
68 "attempting connection to IP:" + ip_str +
69 " port: " + std::to_string(iPort));
70}
