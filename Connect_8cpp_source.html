38 doConnect(RPC::JsonContext& context)
39 {
40  if (context.app.config().standalone())
41  return "cannot connect in standalone mode";
42 
43  if (!context.params.isMember(jss::ip))
44  return RPC::missing_field_error(jss::ip);
45 
46  if (context.params.isMember(jss::port) &&
47  !context.params[jss::port].isConvertibleTo(Json::intValue))
48  {
49  return rpcError(rpcINVALID_PARAMS);
50  }
51 
52  int iPort;
53 
54  if (context.params.isMember(jss::port))
55  iPort = context.params[jss::port].asInt();
56  else
57  iPort = DEFAULT_PEER_PORT;
58 
59  auto ip =
60  beast::IP::Endpoint::from_string(context.params[jss::ip].asString());
61 
62  if (!is_unspecified(ip))
63  context.app.overlay().connect(ip.at_port(iPort));
64 
65  return RPC::makeObjectValue("connecting");
66 }
