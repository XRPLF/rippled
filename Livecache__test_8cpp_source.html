114  c.insert(ep1);
115  BEAST_EXPECT(c.size() == 1);
116  c.expire();
117  BEAST_EXPECT(c.size() == 1);
118  // verify that advancing to 1 sec before expiration
119  // leaves our entry intact
120  clock_.advance(Tuning::liveCacheSecondsToLive - 1s);
121  c.expire();

123  // now advance to the point of expiration

129  void testHistogram ()
130  {
131  testcase ("Histogram");
132  constexpr auto num_eps = 40;
133  Livecache <> c (clock_, journal_);
134  for (auto i = 0; i < num_eps; ++i)
135  add(
136  beast::IP::randomEP(true),

139  auto h = c.hops.histogram();
140  if(! BEAST_EXPECT(! h.empty()))
141  return;
142  std::vector <std::string> v;
143  boost::split (v, h, boost::algorithm::is_any_of (","));
144  auto sum = 0;
145  for (auto const& n : v)
146  {
147  auto val = boost::lexical_cast<int>(boost::trim_copy(n));
148  sum += val;
149  BEAST_EXPECT(val >= 0);
150  }
151  BEAST_EXPECT(sum == num_eps);
152  }

155  void testShuffle ()
156  {
157  testcase ("Shuffle");
158  Livecache <> c (clock_, journal_);
159  for (auto i = 0; i < 100; ++i)
160  add(
161  beast::IP::randomEP(true),
162  c,
163  ripple::rand_int(0, safe_cast<int>(Tuning::maxHops + 1)));
164 
165  using at_hop = std::vector <ripple::PeerFinder::Endpoint>;
166  using all_hops = std::array <at_hop, 1 + Tuning::maxHops + 1>;
167 
168  auto cmp_EP = [](Endpoint const& a, Endpoint const& b) {
169  return (b.hops < a.hops || (b.hops == a.hops && b.address < a.address));
170  };
171  all_hops before;
172  all_hops before_sorted;
173  for (auto i = std::make_pair(0, c.hops.begin());
174  i.second != c.hops.end(); ++i.first, ++i.second)
175  {
176  std::copy ((*i.second).begin(), (*i.second).end(),
177  std::back_inserter (before[i.first]));
178  std::copy ((*i.second).begin(), (*i.second).end(),
179  std::back_inserter (before_sorted[i.first]));
180  std::sort(
181  before_sorted[i.first].begin(),
182  before_sorted[i.first].end(),
183  cmp_EP);
184  }
185 
186  c.hops.shuffle();
187 
188  all_hops after;
189  all_hops after_sorted;
190  for (auto i = std::make_pair(0, c.hops.begin());
191  i.second != c.hops.end(); ++i.first, ++i.second)
192  {
193  std::copy ((*i.second).begin(), (*i.second).end(),
194  std::back_inserter (after[i.first]));
195  std::copy ((*i.second).begin(), (*i.second).end(),
196  std::back_inserter (after_sorted[i.first]));
197  std::sort(
198  after_sorted[i.first].begin(),
199  after_sorted[i.first].end(),
200  cmp_EP);
201  }
202 
203  // each hop bucket should contain the same items
204  // before and after sort, albeit in different order
205  bool all_match = true;
206  for (auto i = 0; i < before.size(); ++i)
207  {
208  BEAST_EXPECT(before[i].size() == after[i].size());
209  all_match = all_match && (before[i] == after[i]);
210  BEAST_EXPECT(before_sorted[i] == after_sorted[i]);
211  }
212  BEAST_EXPECT(! all_match);
213  }
