49inline constexpr auto kPROPAGATE_EXCEPTIONS = [](std::exception_ptr ePtr) {
50 if (ePtr)
51 {

87spawn(Ctx&& ctx, F&& func)
88{
89 if constexpr (impl::IsStrand<Ctx>)

91 boost::asio::spawn(

99 boost::asio::make_strand(
