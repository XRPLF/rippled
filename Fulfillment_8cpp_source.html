33 match(Fulfillment const& f, Condition const& c)
34 {
35  // Fast check: the fulfillment's type must match the
36  // conditions's type:
37  if (f.type() != c.type)
38  return false;
39 
40  // Derive the condition from the given fulfillment
41  // and ensure that it matches the given condition.
42  return c == f.condition();
43 }

57 std::unique_ptr<Fulfillment>
58 Fulfillment::deserialize(Slice s, std::error_code& ec)
59 {
60  // Per the RFC, in a fulfillment we choose a type based
61  // on the tag of the item we contain:
62  //
63  // Fulfillment ::= CHOICE {
64  // preimageSha256 [0] PreimageFulfillment ,
65  // prefixSha256 [1] PrefixFulfillment,
66  // thresholdSha256 [2] ThresholdFulfillment,
67  // rsaSha256 [3] RsaSha256Fulfillment,
68  // ed25519Sha256 [4] Ed25519Sha512Fulfillment
69  // }
70 
71  if (s.empty())
72  {
73  ec = error::buffer_empty;
74  return nullptr;
75  }
76 
77  using namespace der;
78 
79  auto const p = parsePreamble(s, ec);
80  if (ec)
81  return nullptr;
82 
83  // All fulfillments are context-specific, constructed types
84  if (!isConstructed(p) || !isContextSpecific(p))
85  {
86  ec = error::malformed_encoding;
87  return nullptr;
88  }
89 
90  if (p.length > s.size())
91  {
92  ec = error::buffer_underfull;
93  return {};
94  }
95 
96  if (p.length < s.size())
97  {
98  ec = error::buffer_overfull;
99  return {};
100  }
101 
102  if (p.length > maxSerializedFulfillment)
103  {
104  ec = error::large_size;
105  return {};
106  }
107 
108  std::unique_ptr<Fulfillment> f;
109 
110  using TagType = decltype(p.tag);
111  switch (p.tag)
112  {
113  case safe_cast<TagType>(Type::preimageSha256):
114  f = PreimageSha256::deserialize(Slice(s.data(), p.length), ec);
115  if (ec)
116  return {};
117  s += p.length;
118  break;
119 
120  case safe_cast<TagType>(Type::prefixSha256):
121  ec = error::unsupported_type;
122  return {};
123  break;
124 
125  case safe_cast<TagType>(Type::thresholdSha256):
126  ec = error::unsupported_type;
127  return {};
128  break;
129 
130  case safe_cast<TagType>(Type::rsaSha256):
131  ec = error::unsupported_type;
132  return {};
133  break;
134 
135  case safe_cast<TagType>(Type::ed25519Sha256):
136  ec = error::unsupported_type;
137  return {};
138 
139  default:
140  ec = error::unknown_type;
141  return {};
142  }
143 
144  if (!s.empty())
145  {
146  ec = error::trailing_garbage;
147  return {};
148  }
149 
150  return f;
151 }
