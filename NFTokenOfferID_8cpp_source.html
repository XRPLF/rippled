25bool
26canHaveNFTokenOfferID(
27 std::shared_ptr<STTx const> const& serializedTx,
28 TxMeta const& transactionMeta)
29{
30 if (!serializedTx)
31 return false;
32
33 TxType const tt = serializedTx->getTxnType();
34 if (!(tt == ttNFTOKEN_MINT && serializedTx->isFieldPresent(sfAmount)) &&
35 tt != ttNFTOKEN_CREATE_OFFER)
36 return false;
37
38 // if the transaction failed nothing could have been delivered.
39 if (transactionMeta.getResultTER() != tesSUCCESS)
40 return false;
41
42 return true;
43}

45std::optional<uint256>
46getOfferIDFromCreatedOffer(TxMeta const& transactionMeta)
47{
48 for (STObject const& node : transactionMeta.getNodes())
49 {
50 if (node.getFieldU16(sfLedgerEntryType) != ltNFTOKEN_OFFER ||
51 node.getFName() != sfCreatedNode)
52 continue;
53
54 return node.getFieldH256(sfLedgerIndex);
55 }
56 return std::nullopt;
57}

59void
60insertNFTokenOfferID(
61 Json::Value& response,
62 std::shared_ptr<STTx const> const& transaction,
63 TxMeta const& transactionMeta)
64{
65 if (!canHaveNFTokenOfferID(transaction, transactionMeta))
66 return;
67
68 std::optional<uint256> result = getOfferIDFromCreatedOffer(transactionMeta);
69
70 if (result.has_value())
71 response[jss::offer_id] = to_string(result.value());
72}
