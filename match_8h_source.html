17// Predicate for implementing matches
18class selector
19{
20public:
21 enum mode_t {
22 // Run all tests except manual ones
23 all,
24
25 // Run tests that match in any field
26 automatch,
27
28 // Match on suite
29 suite,
30
31 // Match on library
32 library,
33
34 // Match on module (used internally)
35 module,
36
37 // Match nothing (used internally)
38 none
39 };
40
41private:
42 mode_t mode_;
43 std::string pat_;
44 std::string library_;
45
46public:
47 template <class = void>
48 explicit selector(mode_t mode, std::string const& pattern = "");
49
50 template <class = void>
51 bool
52 operator()(suite_info const& s);
53};

55//------------------------------------------------------------------------------
56
57template <class>
58selector::selector(mode_t mode, std::string const& pattern)
59 : mode_(mode), pat_(pattern)
60{
61 if (mode_ == automatch && pattern.empty())
62 mode_ = all;
63}
64
65template <class>
66bool
67selector::operator()(suite_info const& s)
68{
69 switch (mode_)
70 {
71 case automatch:
72 // suite or full name
73 if (s.name() == pat_ || s.full_name() == pat_)
74 {
75 mode_ = none;
76 return true;
77 }
78
79 // check module
80 if (pat_ == s.module())
81 {
82 mode_ = module;
83 library_ = s.library();
84 return !s.manual();
85 }
86
87 // check library
88 if (pat_ == s.library())
89 {
90 mode_ = library;
91 return !s.manual();
92 }
93
94 // check start of name
95 if (s.name().starts_with(pat_) || s.full_name().starts_with(pat_))
96 {
97 // Don't change the mode so that the partial pattern can match
98 // more than once
99 return !s.manual();
100 }
101
102 return false;
103
104 case suite:
105 return pat_ == s.name();
106
107 case module:
108 return pat_ == s.module() && !s.manual();
109
110 case library:
111 return pat_ == s.library() && !s.manual();
112
113 case none:
114 return false;
115
116 case all:
117 default:
118 break;
119 };
120
121 return !s.manual();
122}
