29class Peers_test : public beast::unit_test::suite
30{
31 void
32 testRequest()
33 {
34 testcase("Basic request");
35 using namespace test::jtx;
36 Env env{*this};
37
38 // without modification of the cluster, expect an empty set
39 // from this request
40 auto peers = env.rpc("peers")[jss::result];
41 BEAST_EXPECT(
42 peers.isMember(jss::cluster) && peers[jss::cluster].size() == 0);
43 BEAST_EXPECT(peers.isMember(jss::peers) && peers[jss::peers].isNull());
44
45 // insert some nodes in to the cluster
46 std::unordered_map<std::string, std::string> nodes;
47 for (auto i = 0; i < 3; ++i)
48 {
49 auto kp = generateKeyPair(
50 KeyType::secp256k1, generateSeed("seed" + std::to_string(i)));
51
52 std::string name = "Node " + std::to_string(i);
53
54 using namespace std::chrono_literals;
55 env.app().cluster().update(
56 kp.first, name, 200, env.timeKeeper().now() - 10s);
57 nodes.insert(std::make_pair(
58 toBase58(TokenType::NodePublic, kp.first), name));
59 }
60
61 // make request, verify nodes we created match
62 // what is reported
63 peers = env.rpc("peers")[jss::result];
64 if (!BEAST_EXPECT(peers.isMember(jss::cluster)))
65 return;
66 if (!BEAST_EXPECT(peers[jss::cluster].size() == nodes.size()))
67 return;
68 for (auto it = peers[jss::cluster].begin();
69 it != peers[jss::cluster].end();
70 ++it)
71 {
72 auto key = it.key().asString();
73 auto search = nodes.find(key);
74 if (!BEAST_EXPECTS(search != nodes.end(), key))
75 continue;
76 if (!BEAST_EXPECT((*it).isMember(jss::tag)))
77 continue;
78 auto tag = (*it)[jss::tag].asString();
79 BEAST_EXPECTS((*it)[jss::tag].asString() == nodes[key], key);
80 }
81 BEAST_EXPECT(peers.isMember(jss::peers) && peers[jss::peers].isNull());
82 }
83
84public:
85 void
86 run() override
87 {
88 testRequest();
89 }
90};
