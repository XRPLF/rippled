31  void testRequest()
32  {
33  testcase("Basic request");
34  using namespace test::jtx;
35  Env env {*this};
36 
37  // without modification of the cluster, expect an empty set
38  // from this request
39  auto peers = env.rpc("peers")[jss::result];
40  BEAST_EXPECT(peers.isMember(jss::cluster) &&
41  peers[jss::cluster].size() == 0 );
42  BEAST_EXPECT(peers.isMember(jss::peers) &&
43  peers[jss::peers].isNull());
44 
45  // insert some nodes in to the cluster
46  std::unordered_map<std::string, std::string> nodes;
47  for(auto i =0; i < 3; ++i)
48  {
49 
50  auto kp = generateKeyPair (KeyType::secp256k1,
51  generateSeed("seed" + std::to_string(i)));
52 
53  std::string name = "Node " + std::to_string(i);
54 
55  using namespace std::chrono_literals;
56  env.app().cluster().update(
57  kp.first,
58  name,
59  200,
60  env.timeKeeper().now() - 10s);
61  nodes.insert( std::make_pair(
62  toBase58(TokenType::NodePublic, kp.first), name));
63  }
64 
65  // make request, verify nodes we created match
66  // what is reported
67  peers = env.rpc("peers")[jss::result];
68  if(! BEAST_EXPECT(peers.isMember(jss::cluster)))
69  return;
70  if(! BEAST_EXPECT(peers[jss::cluster].size() == nodes.size()))
71  return;
72  for(auto it = peers[jss::cluster].begin();
73  it != peers[jss::cluster].end();
74  ++it)
75  {
76  auto key = it.key().asString();
77  auto search = nodes.find(key);
78  if(! BEAST_EXPECTS(search != nodes.end(), key))
79  continue;
80  if(! BEAST_EXPECT((*it).isMember(jss::tag)))
81  continue;
82  auto tag = (*it)[jss::tag].asString();
83  BEAST_EXPECTS((*it)[jss::tag].asString() == nodes[key], key);
84  }
85  BEAST_EXPECT(peers.isMember(jss::peers) && peers[jss::peers].isNull());
86  }
