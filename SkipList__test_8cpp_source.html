29 class SkipList_test : public beast::unit_test::suite
30 {
31  void
32  testSkipList()
33  {
34  jtx::Env env(*this);
35  std::vector<std::shared_ptr<Ledger>> history;
36  {
37  Config config;
38  auto prev = std::make_shared<Ledger>(
39  create_genesis, config,
40  std::vector<uint256>{}, env.app().family());
41  history.push_back(prev);
42  for (auto i = 0; i < 1023; ++i)
43  {
44  auto next = std::make_shared<Ledger>(
45  *prev,
46  env.app().timeKeeper().closeTime());
47  next->updateSkipList();
48  history.push_back(next);
49  prev = next;
50  }
51  }
52 
53  {
54  auto l = *(std::next(std::begin(history)));
55  BEAST_EXPECT((*std::begin(history))->info().seq <
56  l->info().seq);
57  BEAST_EXPECT(hashOfSeq(*l, l->info().seq + 1,
58  env.journal) == boost::none);
59  BEAST_EXPECT(hashOfSeq(*l, l->info().seq,
60  env.journal) == l->info().hash);
61  BEAST_EXPECT(hashOfSeq(*l, l->info().seq - 1,
62  env.journal) == l->info().parentHash);
63  BEAST_EXPECT(hashOfSeq(*history.back(),
64  l->info().seq, env.journal) == boost::none);
65  }
66 
67  // ledger skip lists store up to the previous 256 hashes
68  for (auto i = history.crbegin();
69  i != history.crend(); i += 256)
70  {
71  for (auto n = i;
72  n != std::next(i,
73  (*i)->info().seq - 256 > 1 ? 257 : 256);
74  ++n)
75  {
76  BEAST_EXPECT(hashOfSeq(**i,
77  (*n)->info().seq, env.journal) ==
78  (*n)->info().hash);
79  }
80 
81  // edge case accessing beyond 256
82  BEAST_EXPECT(hashOfSeq(**i,
83  (*i)->info().seq - 258, env.journal) ==
84  boost::none);
85  }
86 
87  // every 256th hash beyond the first 256 is stored
88  for (auto i = history.crbegin();
89  i != std::next(history.crend(), -512);
90  i += 256)
91  {
92  for (auto n = std::next(i, 512);
93  n != history.crend();
94  n += 256)
95  {
96  BEAST_EXPECT(hashOfSeq(**i,
97  (*n)->info().seq, env.journal) ==
98  (*n)->info().hash);
99  }
100  }
101  }

106  }
