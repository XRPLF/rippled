51  void
52  cmpAmount(
53  const org::xrpl::rpc::v1::CurrencyAmount& proto_amount,
54  STAmount amount)
55  {
56  if (amount.native())
57  {
58  if (!BEAST_EXPECT(proto_amount.has_xrp_amount()))
59  return;
60  BEAST_EXPECT(
61  proto_amount.xrp_amount().drops() == amount.xrp().drops());
62  }
63  else
64  {
65  if (!BEAST_EXPECT(proto_amount.has_issued_currency_amount()))
66  return;
67 
68  org::xrpl::rpc::v1::IssuedCurrencyAmount issuedCurrency =
69  proto_amount.issued_currency_amount();
70  Issue const& issue = amount.issue();
71  Currency currency = issue.currency;
72  BEAST_EXPECT(
73  issuedCurrency.currency().name() == to_string(currency));
74  BEAST_EXPECT(
75  issuedCurrency.currency().code() == toByteString(currency));
76  BEAST_EXPECT(issuedCurrency.value() == to_string(amount.iou()));
77  BEAST_EXPECT(
78  issuedCurrency.issuer().address() == toBase58(issue.account));
79  }
80  }
