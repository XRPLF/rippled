54 Sim::synchronized(PeerGroup const & g) const
55 {
56  if (g.size() < 1)
57  return true;
58  Peer const * ref = g[0];
59  return std::all_of(g.begin(), g.end(), [&ref](Peer const* p) {
60  return p->lastClosedLedger.id() ==
61  ref->lastClosedLedger.id() &&
62  p->fullyValidatedLedger.id() ==
63  ref->fullyValidatedLedger.id();
64  });
65 }
