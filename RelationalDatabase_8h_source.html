228template <class T, class C>
229T
230rangeCheckedCast(C c)
231{
232 if ((c > std::numeric_limits<T>::max()) ||
233 (!std::numeric_limits<T>::is_signed && c < 0) ||
234 (std::numeric_limits<T>::is_signed &&
235 std::numeric_limits<C>::is_signed &&
236 c < std::numeric_limits<T>::lowest()))
237 {
238 /* This should never happen */
239 UNREACHABLE("ripple::rangeCheckedCast : domain error");
240 JLOG(debugLog().error())
241 << "rangeCheckedCast domain error:" << " value = " << c
242 << " min = " << std::numeric_limits<T>::lowest()
243 << " max: " << std::numeric_limits<T>::max();
244 }
245
246 return static_cast<T>(c);
247}
