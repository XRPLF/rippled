78 bool
79 updateLedgerDBs(
80  soci::session& txsession,
81  soci::session& lgrsession,
82  std::shared_ptr<Ledger const> const& ledger,
83  std::uint32_t index,
84  std::atomic<bool>& stop,
85  beast::Journal j)
86 {
87  auto const ledgerSeq{ledger->info().seq};
88 
89  // Update the transactions database
90  {
91  auto& session{txsession};
92  soci::transaction tr(session);
93 
94  session << "DELETE FROM Transactions "
95  "WHERE LedgerSeq = :seq;",
96  soci::use(ledgerSeq);
97  session << "DELETE FROM AccountTransactions "
98  "WHERE LedgerSeq = :seq;",
99  soci::use(ledgerSeq);
100 
101  if (ledger->info().txHash.isNonZero())
102  {
103  auto const sSeq{std::to_string(ledgerSeq)};
104  if (!ledger->txMap().isValid())
105  {
106  JLOG(j.error())
107  << "shard " << index << " has an invalid transaction map"
108  << " on sequence " << sSeq;
109  return false;
110  }
111 
112  for (auto const& item : ledger->txs)
113  {
114  if (stop)
115  return false;
116 
117  auto const txID{item.first->getTransactionID()};
118  auto const sTxID{to_string(txID)};
119  auto const txMeta{std::make_shared<TxMeta>(
120  txID, ledger->seq(), *item.second)};
121 
122  session << "DELETE FROM AccountTransactions "
123  "WHERE TransID = :txID;",
124  soci::use(sTxID);
125 
126  auto const& accounts = txMeta->getAffectedAccounts(j);
127  if (!accounts.empty())
128  {
129  auto const sTxnSeq{std::to_string(txMeta->getIndex())};
130  auto const s{boost::str(
131  boost::format("('%s','%s',%s,%s)") % sTxID % "%s" %
132  sSeq % sTxnSeq)};
133  std::string sql;
134  sql.reserve((accounts.size() + 1) * 128);
135  sql =
136  "INSERT INTO AccountTransactions "
137  "(TransID, Account, LedgerSeq, TxnSeq) VALUES ";
138  sql += boost::algorithm::join(
139  accounts |
140  boost::adaptors::transformed(
141  [&](AccountID const& accountID) {
142  return boost::str(
143  boost::format(s) %
144  ripple::toBase58(accountID));
145  }),
146  ",");
147  sql += ';';
148  session << sql;
149 
150  JLOG(j.trace())
151  << "shard " << index << " account transaction: " << sql;
152  }
153  else
154  {
155  JLOG(j.warn())
156  << "shard " << index << " transaction in ledger "
157  << sSeq << " affects no accounts";
158  }
159 
160  Serializer s;
161  item.second->add(s);
162  session
163  << (STTx::getMetaSQLInsertReplaceHeader() +
164  item.first->getMetaSQL(
165  ledgerSeq, sqlBlobLiteral(s.modData())) +
166  ';');
167  }
168  }
169 
170  tr.commit();
171  }
172 
173  auto const sHash{to_string(ledger->info().hash)};
174 
175  // Update the ledger database
176  {
177  auto& session{lgrsession};
178  soci::transaction tr(session);
179 
180  auto const sParentHash{to_string(ledger->info().parentHash)};
181  auto const sDrops{to_string(ledger->info().drops)};
182  auto const sAccountHash{to_string(ledger->info().accountHash)};
183  auto const sTxHash{to_string(ledger->info().txHash)};
184 
185  session << "DELETE FROM Ledgers "
186  "WHERE LedgerSeq = :seq;",
187  soci::use(ledgerSeq);
188  session << "INSERT OR REPLACE INTO Ledgers ("
189  "LedgerHash, LedgerSeq, PrevHash, TotalCoins, ClosingTime,"
190  "PrevClosingTime, CloseTimeRes, CloseFlags, AccountSetHash,"
191  "TransSetHash)"
192  "VALUES ("
193  ":ledgerHash, :ledgerSeq, :prevHash, :totalCoins,"
194  ":closingTime, :prevClosingTime, :closeTimeRes,"
195  ":closeFlags, :accountSetHash, :transSetHash);",
196  soci::use(sHash), soci::use(ledgerSeq), soci::use(sParentHash),
197  soci::use(sDrops),
198  soci::use(ledger->info().closeTime.time_since_epoch().count()),
199  soci::use(
200  ledger->info().parentCloseTime.time_since_epoch().count()),
201  soci::use(ledger->info().closeTimeResolution.count()),
202  soci::use(ledger->info().closeFlags), soci::use(sAccountHash),
203  soci::use(sTxHash);
204 
205  tr.commit();
206  }
207 
208  return true;
209 }

234 std::pair<bool, std::optional<std::string>>
235 selectAcquireDBLedgerSeqs(soci::session& session, std::uint32_t index)
236 {
237  // resIndex and must be boost::optional (not std) because that's
238  // what SOCI expects in its interface.
239  boost::optional<std::uint32_t> resIndex;
240  soci::blob sociBlob(session);
241  soci::indicator blobPresent;
242 
243  session << "SELECT ShardIndex, StoredLedgerSeqs "
244  "FROM Shard "
245  "WHERE ShardIndex = :index;",
246  soci::into(resIndex), soci::into(sociBlob, blobPresent),
247  soci::use(index);
248 
249  if (!resIndex || index != resIndex)
250  return {false, {}};
251 
252  if (blobPresent != soci::i_ok)
253  return {true, {}};
254 
255  std::string s;
256  convert(sociBlob, s);
257 
258  return {true, s};
259 }

261 std::pair<bool, AcquireShardSeqsHash>
262 selectAcquireDBLedgerSeqsHash(soci::session& session, std::uint32_t index)
263 {
264  // resIndex and sHash0 must be boost::optional (not std) because that's
265  // what SOCI expects in its interface.
266  boost::optional<std::uint32_t> resIndex;
267  boost::optional<std::string> sHash0;
268  soci::blob sociBlob(session);
269  soci::indicator blobPresent;
270 
271  session << "SELECT ShardIndex, LastLedgerHash, StoredLedgerSeqs "
272  "FROM Shard "
273  "WHERE ShardIndex = :index;",
274  soci::into(resIndex), soci::into(sHash0),
275  soci::into(sociBlob, blobPresent), soci::use(index);
276 
277  std::optional<std::string> sHash =
278  (sHash0 ? *sHash0 : std::optional<std::string>());
279 
280  if (!resIndex || index != resIndex)
281  return {false, {{}, {}}};
282 
283  if (blobPresent != soci::i_ok)
284  return {true, {{}, sHash}};
285 
286  std::string s;
287  convert(sociBlob, s);
288 
289  return {true, {s, sHash}};
290 }

292 void
293 updateAcquireDB(
294  soci::session& session,
295  std::shared_ptr<Ledger const> const& ledger,
296  std::uint32_t index,
297  std::uint32_t lastSeq,
298  std::optional<std::string> const& seqs)
299 {
300  soci::blob sociBlob(session);
301  auto const sHash{to_string(ledger->info().hash)};
302 
303  if (seqs)
304  convert(*seqs, sociBlob);
305 
306  if (ledger->info().seq == lastSeq)
307  {
308  // Store shard's last ledger hash
309  session << "UPDATE Shard "
310  "SET LastLedgerHash = :lastLedgerHash,"
311  "StoredLedgerSeqs = :storedLedgerSeqs "
312  "WHERE ShardIndex = :shardIndex;",
313  soci::use(sHash), soci::use(sociBlob), soci::use(index);
314  }
315  else
316  {
317  session << "UPDATE Shard "
318  "SET StoredLedgerSeqs = :storedLedgerSeqs "
319  "WHERE ShardIndex = :shardIndex;",
320  soci::use(sociBlob), soci::use(index);
321  }
322 }
