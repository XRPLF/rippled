69  void
70  testPrevented()
71  {
72  using namespace jtx;
73  Env env(*this);
74 
75  for (auto const& stx : getPseudoTxs(env.closed()->seq() + 1))
76  {
77  std::string reason;
78  BEAST_EXPECT(isPseudoTx(stx));

80  BEAST_EXPECT(reason == "Cannot submit pseudo transactions.");
81  env.app().openLedger().modify(
82  [&](OpenView& view, beast::Journal j) {

85  BEAST_EXPECT(!result.second && result.first == temINVALID);
