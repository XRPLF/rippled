173 void
174 changePosition(
175 Position_t const& newPosition,
176 NetClock::time_point newCloseTime,
177 NetClock::time_point now)
178 {
179 signingHash_.reset();
180 position_ = newPosition;
181 closeTime_ = newCloseTime;
182 time_ = now;
183 if (proposeSeq_ != seqLeave)
184 ++proposeSeq_;
185 }

201 std::string

206 << " proposal_seq: " << proposeSeq_ << " position: " << position_

209 << " node_id: " << nodeID_;

214 Json::Value

216 {
217 using std::to_string;

219 Json::Value ret = Json::objectValue;

222 if (!isBowOut())

224 ret[jss::transaction_hash] = to_string(position());

235 uint256 const&
236 signingHash() const
237 {
238 if (!signingHash_)
239 {
240 signingHash_ = sha512Half(
241 HashPrefix::proposal,
242 std::uint32_t(proposeSeq()),
243 closeTime().time_since_epoch().count(),
244 prevLedger(),
245 position());
246 }
247
248 return signingHash_.value();
249 }
