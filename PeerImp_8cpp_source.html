150 void
151 PeerImp::run()
152 {
153  if (!strand_.running_in_this_thread())
154  return post(strand_, std::bind(&PeerImp::run, shared_from_this()));
155 
156  auto parseLedgerHash =
157  [](std::string const& value) -> std::optional<uint256> {
158  if (uint256 ret; ret.parseHex(value))
159  return ret;
160 
161  if (auto const s = base64_decode(value); s.size() == uint256::size())
162  return uint256{s};
163 
164  return std::nullopt;
165  };
166 
167  std::optional<uint256> closed;
168  std::optional<uint256> previous;
169 
170  if (auto const iter = headers_.find("Closed-Ledger");
171  iter != headers_.end())
172  {
173  closed = parseLedgerHash(iter->value().to_string());
174 
175  if (!closed)
176  fail("Malformed handshake data (1)");
177  }
178 
179  if (auto const iter = headers_.find("Previous-Ledger");
180  iter != headers_.end())
181  {
182  previous = parseLedgerHash(iter->value().to_string());
183 
184  if (!previous)
185  fail("Malformed handshake data (2)");
186  }
187 
188  if (previous && !closed)
189  fail("Malformed handshake data (3)");
190 
191  {
192  std::lock_guard<std::mutex> sl(recentLock_);
193  if (closed)
194  closedLedgerHash_ = *closed;
195  if (previous)
196  previousLedgerHash_ = *previous;
197  }
198 
199  if (inbound_)
200  doAccept();
201  else
202  doProtocolStart();
203 
204  // Anything else that needs to be done with the connection should be
205  // done in doProtocolStart
206 }

235 PeerImp::send(std::shared_ptr<Message> const& m)
236 {
237  if (!strand_.running_in_this_thread())
238  return post(strand_, std::bind(&PeerImp::send, shared_from_this(), m));
239  if (gracefulClose_)
240  return;
241  if (detaching_)
242  return;
243 
244  auto validator = m->getValidatorKey();
245  if (validator && !squelch_.expireSquelch(*validator))
246  return;
247 
248  overlay_.reportTraffic(
249  safe_cast<TrafficCount::category>(m->getCategory()),
250  false,
251  static_cast<int>(m->getBuffer(compressionEnabled_).size()));
252 
253  auto sendq_size = send_queue_.size();
254 
255  if (sendq_size < Tuning::targetSendQueue)
256  {
257  // To detect a peer that does not read from their
258  // side of the connection, we expect a peer to have
259  // a small senq periodically
260  large_sendq_ = 0;
261  }
262  else if (auto sink = journal_.debug();
263  sink && (sendq_size % Tuning::sendQueueLogFreq) == 0)
264  {
265  std::string const n = name();
266  sink << (n.empty() ? remote_address_.to_string() : n)
267  << " sendq: " << sendq_size;
268  }
269 
270  send_queue_.push(m);
271 
272  if (sendq_size != 0)
273  return;
274 
275  boost::asio::async_write(
276  stream_,
277  boost::asio::buffer(
278  send_queue_.front()->getBuffer(compressionEnabled_)),
279  bind_executor(
280  strand_,
281  std::bind(
282  &PeerImp::onWriteMessage,
283  shared_from_this(),
284  std::placeholders::_1,
285  std::placeholders::_2)));

382  break;
383 
384  case Tracking::converged:
385  // Nothing to do here
386  break;
387  }
388 
389  uint256 closedLedgerHash;
390  protocol::TMStatusChange last_status;
391  {
392  std::lock_guard sl(recentLock_);
393  closedLedgerHash = closedLedgerHash_;

395  }
396

398  ret[jss::ledger] = to_string(closedLedgerHash);

404  case protocol::nsCONNECTING:
405  ret[jss::status] = "connecting";
406  break;
407 
408  case protocol::nsCONNECTED:
409  ret[jss::status] = "connected";
410  break;
411 
412  case protocol::nsMONITORING:
413  ret[jss::status] = "monitoring";
414  break;
415 
416  case protocol::nsVALIDATING:
417  ret[jss::status] = "validating";
418  break;
419 
420  case protocol::nsSHUTTING:
421  ret[jss::status] = "shutting";

653 void
654 PeerImp::onTimer(error_code const& ec)
655 {
656  if (!socket_.is_open())
657  return;
658 
659  if (ec == boost::asio::error::operation_aborted)
660  return;
661 
662  if (ec)
663  {
664  // This should never happen
665  JLOG(journal_.error()) << "onTimer: " << ec.message();
666  return close();
667  }
668 
669  if (large_sendq_++ >= Tuning::sendqIntervals)
670  {
671  fail("Large send queue");
672  return;
673  }
674 
675  if (auto const t = tracking_.load(); !inbound_ && t != Tracking::converged)
676  {
677  clock_type::duration duration;
678 
679  {
680  std::lock_guard sl(recentLock_);
681  duration = clock_type::now() - trackingTime_;
682  }
683 
684  if ((t == Tracking::diverged &&
685  (duration > app_.config().MAX_DIVERGED_TIME)) ||
686  (t == Tracking::unknown &&
687  (duration > app_.config().MAX_UNKNOWN_TIME)))
688  {
689  overlay_.peerFinder().on_failure(slot_);
690  fail("Not useful");
691  return;
692  }
693  }
694 
695  // Already waiting for PONG
696  if (lastPingSeq_)
697  {
698  fail("Ping Timeout");
699  return;
700  }
701 
702  lastPingTime_ = clock_type::now();
703  lastPingSeq_ = rand_int<std::uint32_t>();
704 
705  protocol::TMPing message;
706  message.set_type(protocol::TMPing::ptPING);
707  message.set_seq(*lastPingSeq_);
708 
709  send(std::make_shared<Message>(message, protocol::mtPING));
710 
711  setTimer();
712 }

729 //------------------------------------------------------------------------------
730 void
731 PeerImp::doAccept()
732 {
733  assert(read_buffer_.size() == 0);
734 
735  JLOG(journal_.debug()) << "doAccept: " << remote_address_;
736 
737  auto const sharedValue = makeSharedValue(*stream_ptr_, journal_);
738 
739  // This shouldn't fail since we already computed
740  // the shared value successfully in OverlayImpl
741  if (!sharedValue)
742  return fail("makeSharedValue: Unexpected failure");
743 
744  JLOG(journal_.info()) << "Protocol: " << to_string(protocol_);
745  JLOG(journal_.info()) << "Public Key: "
746  << toBase58(TokenType::NodePublic, publicKey_);
747 
748  if (auto member = app_.cluster().member(publicKey_))
749  {
750  {
751  std::unique_lock lock{nameMutex_};
752  name_ = *member;
753  }
754  JLOG(journal_.info()) << "Cluster name: " << *member;
755  }
756 
757  overlay_.activate(shared_from_this());
758 
759  // XXX Set timer: connection is in grace period to be useful.
760  // XXX Set timer: connection idle (idle may vary depending on connection
761  // type.)
762 
763  auto write_buffer = std::make_shared<boost::beast::multi_buffer>();
764 
765  boost::beast::ostream(*write_buffer) << makeResponse(
766  !overlay_.peerFinder().config().peerPrivate,
767  request_,
768  overlay_.setup().public_ip,
