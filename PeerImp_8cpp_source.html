155void
156PeerImp::run()
157{
158 if (!strand_.running_in_this_thread())
159 return post(strand_, std::bind(&PeerImp::run, shared_from_this()));
160
161 auto parseLedgerHash =
162 [](std::string_view value) -> std::optional<uint256> {
163 if (uint256 ret; ret.parseHex(value))
164 return ret;
165
166 if (auto const s = base64_decode(value); s.size() == uint256::size())
167 return uint256{s};
168
169 return std::nullopt;
170 };
171
172 std::optional<uint256> closed;
173 std::optional<uint256> previous;
174
175 if (auto const iter = headers_.find("Closed-Ledger");
176 iter != headers_.end())
177 {
178 closed = parseLedgerHash(iter->value());
179
180 if (!closed)
181 fail("Malformed handshake data (1)");
182 }
183
184 if (auto const iter = headers_.find("Previous-Ledger");
185 iter != headers_.end())
186 {
187 previous = parseLedgerHash(iter->value());
188
189 if (!previous)
190 fail("Malformed handshake data (2)");
191 }
192
193 if (previous && !closed)
194 fail("Malformed handshake data (3)");
195
196 {
197 std::lock_guard<std::mutex> sl(recentLock_);
198 if (closed)
199 closedLedgerHash_ = *closed;
200 if (previous)
201 previousLedgerHash_ = *previous;
202 }
203
204 if (inbound_)
205 doAccept();
206 else
207 doProtocolStart();
208
209 // Anything else that needs to be done with the connection should be
210 // done in doProtocolStart
211}

213void
214PeerImp::stop()
215{
216 if (!strand_.running_in_this_thread())
217 return post(strand_, std::bind(&PeerImp::stop, shared_from_this()));
218 if (socket_.is_open())
219 {
220 // The rationale for using different severity levels is that
221 // outbound connections are under our control and may be logged
222 // at a higher level, but inbound connections are more numerous and
223 // uncontrolled so to prevent log flooding the severity is reduced.
224 //
225 if (inbound_)
226 {
227 JLOG(journal_.debug()) << "Stop";
228 }
229 else
230 {
231 JLOG(journal_.info()) << "Stop";
232 }
233 }
234 close();
235}

239void
240PeerImp::send(std::shared_ptr<Message> const& m)
241{
242 if (!strand_.running_in_this_thread())
243 return post(strand_, std::bind(&PeerImp::send, shared_from_this(), m));
244 if (gracefulClose_)
245 return;
246 if (detaching_)
247 return;
248
249 auto validator = m->getValidatorKey();
250 if (validator && !squelch_.expireSquelch(*validator))
251 return;
252
253 overlay_.reportTraffic(
254 safe_cast<TrafficCount::category>(m->getCategory()),
255 false,
256 static_cast<int>(m->getBuffer(compressionEnabled_).size()));
257
258 auto sendq_size = send_queue_.size();
259
260 if (sendq_size < Tuning::targetSendQueue)
261 {
262 // To detect a peer that does not read from their
263 // side of the connection, we expect a peer to have
264 // a small senq periodically
265 large_sendq_ = 0;
266 }
267 else if (auto sink = journal_.debug();
268 sink && (sendq_size % Tuning::sendQueueLogFreq) == 0)
269 {
270 std::string const n = name();
271 sink << (n.empty() ? remote_address_.to_string() : n)
272 << " sendq: " << sendq_size;
273 }
274
275 send_queue_.push(m);
276
277 if (sendq_size != 0)
278 return;
279
280 boost::asio::async_write(
281 stream_,
282 boost::asio::buffer(
283 send_queue_.front()->getBuffer(compressionEnabled_)),
284 bind_executor(
285 strand_,
286 std::bind(
287 &PeerImp::onWriteMessage,
288 shared_from_this(),
289 std::placeholders::_1,
290 std::placeholders::_2)));
291}

293void
294PeerImp::sendTxQueue()
295{
296 if (!strand_.running_in_this_thread())
297 return post(
298 strand_, std::bind(&PeerImp::sendTxQueue, shared_from_this()));
299
300 if (!txQueue_.empty())
301 {
302 protocol::TMHaveTransactions ht;
303 std::for_each(txQueue_.begin(), txQueue_.end(), [&](auto const& hash) {
304 ht.add_hashes(hash.data(), hash.size());
305 });
306 JLOG(p_journal_.trace()) << "sendTxQueue " << txQueue_.size();
307 txQueue_.clear();
308 send(std::make_shared<Message>(ht, protocol::mtHAVE_TRANSACTIONS));
309 }
310}

378Json::Value
379PeerImp::json()
380{
381 Json::Value ret(Json::objectValue);
382
383 ret[jss::public_key] = toBase58(TokenType::NodePublic, publicKey_);
384 ret[jss::address] = remote_address_.to_string();
385
386 if (inbound_)
387 ret[jss::inbound] = true;
388
389 if (cluster())
390 {
391 ret[jss::cluster] = true;
392
393 if (auto const n = name(); !n.empty())
394 // Could move here if Json::Value supported moving from a string
395 ret[jss::name] = n;
396 }
397
398 if (auto const d = domain(); !d.empty())
399 ret[jss::server_domain] = std::string{d};
400
401 if (auto const nid = headers_["Network-ID"]; !nid.empty())
402 ret[jss::network_id] = std::string{nid};
403
404 ret[jss::load] = usage_.balance();
405
406 if (auto const version = getVersion(); !version.empty())
407 ret[jss::version] = std::string{version};
408
409 ret[jss::protocol] = to_string(protocol_);
410
411 {
412 std::lock_guard sl(recentLock_);
413 if (latency_)
414 ret[jss::latency] = static_cast<Json::UInt>(latency_->count());
415 }
416
417 ret[jss::uptime] = static_cast<Json::UInt>(
418 std::chrono::duration_cast<std::chrono::seconds>(uptime()).count());
419
420 std::uint32_t minSeq, maxSeq;
421 ledgerRange(minSeq, maxSeq);
422
423 if ((minSeq != 0) || (maxSeq != 0))
424 ret[jss::complete_ledgers] =
425 std::to_string(minSeq) + " - " + std::to_string(maxSeq);
426
427 switch (tracking_.load())
428 {
429 case Tracking::diverged:
430 ret[jss::track] = "diverged";
431 break;
432
433 case Tracking::unknown:
434 ret[jss::track] = "unknown";
435 break;
436
437 case Tracking::converged:
438 // Nothing to do here
439 break;
440 }
441
442 uint256 closedLedgerHash;
443 protocol::TMStatusChange last_status;
444 {
445 std::lock_guard sl(recentLock_);
446 closedLedgerHash = closedLedgerHash_;
447 last_status = last_status_;
448 }
449
450 if (closedLedgerHash != beast::zero)
451 ret[jss::ledger] = to_string(closedLedgerHash);
452
453 if (last_status.has_newstatus())
454 {
455 switch (last_status.newstatus())
456 {
457 case protocol::nsCONNECTING:
458 ret[jss::status] = "connecting";
459 break;
460
461 case protocol::nsCONNECTED:
462 ret[jss::status] = "connected";
463 break;
464
465 case protocol::nsMONITORING:
466 ret[jss::status] = "monitoring";
467 break;
468
469 case protocol::nsVALIDATING:
470 ret[jss::status] = "validating";
471 break;
472
473 case protocol::nsSHUTTING:
474 ret[jss::status] = "shutting";
475 break;
476
477 default:
478 JLOG(p_journal_.warn())
479 << "Unknown status: " << last_status.newstatus();
480 }
481 }
482
483 ret[jss::metrics] = Json::Value(Json::objectValue);
484 ret[jss::metrics][jss::total_bytes_recv] =
485 std::to_string(metrics_.recv.total_bytes());
486 ret[jss::metrics][jss::total_bytes_sent] =
487 std::to_string(metrics_.sent.total_bytes());
488 ret[jss::metrics][jss::avg_bps_recv] =
489 std::to_string(metrics_.recv.average_bytes());
490 ret[jss::metrics][jss::avg_bps_sent] =
491 std::to_string(metrics_.sent.average_bytes());
492
493 return ret;
494}

679void
680PeerImp::onTimer(error_code const& ec)
681{
682 if (!socket_.is_open())
683 return;
684
685 if (ec == boost::asio::error::operation_aborted)
686 return;
687
688 if (ec)
689 {
690 // This should never happen
691 JLOG(journal_.error()) << "onTimer: " << ec.message();
692 return close();
693 }
694
695 if (large_sendq_++ >= Tuning::sendqIntervals)
696 {
697 fail("Large send queue");
698 return;
699 }
700
701 if (auto const t = tracking_.load(); !inbound_ && t != Tracking::converged)
702 {
703 clock_type::duration duration;
704
705 {
706 std::lock_guard sl(recentLock_);
707 duration = clock_type::now() - trackingTime_;
708 }
709
710 if ((t == Tracking::diverged &&
711 (duration > app_.config().MAX_DIVERGED_TIME)) ||
712 (t == Tracking::unknown &&
713 (duration > app_.config().MAX_UNKNOWN_TIME)))
714 {
715 overlay_.peerFinder().on_failure(slot_);
716 fail("Not useful");
717 return;
718 }
719 }
720
721 // Already waiting for PONG
722 if (lastPingSeq_)
723 {
724 fail("Ping Timeout");
725 return;
726 }
727
728 lastPingTime_ = clock_type::now();
729 lastPingSeq_ = rand_int<std::uint32_t>();
730
731 protocol::TMPing message;
732 message.set_type(protocol::TMPing::ptPING);
733 message.set_seq(*lastPingSeq_);
734
735 send(std::make_shared<Message>(message, protocol::mtPING));
736
737 setTimer();
738}

756void
757PeerImp::doAccept()
758{
759 XRPL_ASSERT(
760 read_buffer_.size() == 0,
761 "ripple::PeerImp::doAccept : empty read buffer");
762
763 JLOG(journal_.debug()) << "doAccept: " << remote_address_;
764
765 auto const sharedValue = makeSharedValue(*stream_ptr_, journal_);
766
767 // This shouldn't fail since we already computed
768 // the shared value successfully in OverlayImpl
769 if (!sharedValue)
770 return fail("makeSharedValue: Unexpected failure");
771
772 JLOG(journal_.info()) << "Protocol: " << to_string(protocol_);
773 JLOG(journal_.info()) << "Public Key: "
774 << toBase58(TokenType::NodePublic, publicKey_);
775
776 if (auto member = app_.cluster().member(publicKey_))
777 {
778 {
779 std::unique_lock lock{nameMutex_};
780 name_ = *member;
781 }
782 JLOG(journal_.info()) << "Cluster name: " << *member;
783 }
784
785 overlay_.activate(shared_from_this());
786
787 // XXX Set timer: connection is in grace period to be useful.
788 // XXX Set timer: connection idle (idle may vary depending on connection
789 // type.)
790
791 auto write_buffer = std::make_shared<boost::beast::multi_buffer>();
792
793 boost::beast::ostream(*write_buffer) << makeResponse(
794 !overlay_.peerFinder().config().peerPrivate,
795 request_,
796 overlay_.setup().public_ip,
797 remote_address_.address(),
798 *sharedValue,
799 overlay_.setup().networkID,
800 protocol_,
801 app_);
802
803 // Write the whole buffer and only start protocol when that's done.
804 boost::asio::async_write(
805 stream_,
806 write_buffer->data(),
807 boost::asio::transfer_all(),
808 bind_executor(
809 strand_,
810 [this, write_buffer, self = shared_from_this()](
811 error_code ec, std::size_t bytes_transferred) {
812 if (!socket_.is_open())
813 return;
814 if (ec == boost::asio::error::operation_aborted)
815 return;
816 if (ec)
817 return fail("onWriteResponse", ec);
818 if (write_buffer->size() == bytes_transferred)
819 return doProtocolStart();
820 return fail("Failed to write header");
821 }));
822}

878// Called repeatedly with protocol message data
879void
880PeerImp::onReadMessage(error_code ec, std::size_t bytes_transferred)
881{
882 if (!socket_.is_open())
883 return;
884 if (ec == boost::asio::error::operation_aborted)
885 return;
886 if (ec == boost::asio::error::eof)
887 {
888 JLOG(journal_.info()) << "EOF";
889 return gracefulClose();
890 }
891 if (ec)
892 return fail("onReadMessage", ec);
893 if (auto stream = journal_.trace())
894 {
895 if (bytes_transferred > 0)
896 stream << "onReadMessage: " << bytes_transferred << " bytes";
897 else
898 stream << "onReadMessage";
899 }
900
901 metrics_.recv.add_message(bytes_transferred);
902
903 read_buffer_.commit(bytes_transferred);
904
905 auto hint = Tuning::readBufferBytes;
906
907 while (read_buffer_.size() > 0)
908 {
909 std::size_t bytes_consumed;
910
911 using namespace std::chrono_literals;
912 std::tie(bytes_consumed, ec) = perf::measureDurationAndLog(
913 [&]() {
914 return invokeProtocolMessage(read_buffer_.data(), *this, hint);
915 },
916 "invokeProtocolMessage",
917 350ms,
918 journal_);
919
920 if (ec)
921 return fail("onReadMessage", ec);
922 if (!socket_.is_open())
923 return;
924 if (gracefulClose_)
925 return;
926 if (bytes_consumed == 0)
927 break;
928 read_buffer_.consume(bytes_consumed);
929 }
930
931 // Timeout on writes only
932 stream_.async_read_some(
933 read_buffer_.prepare(std::max(Tuning::readBufferBytes, hint)),
934 bind_executor(
935 strand_,
936 std::bind(
937 &PeerImp::onReadMessage,
938 shared_from_this(),
939 std::placeholders::_1,
940 std::placeholders::_2)));
941}

945{
946 if (!socket_.is_open())
947 return;
948 if (ec == boost::asio::error::operation_aborted)
949 return;
950 if (ec)
951 return fail("onWriteMessage", ec);
952 if (auto stream = journal_.trace())

1106PeerImp::onMessage(std::shared_ptr<protocol::TMCluster> const& m)
1107{
1108 // VFALCO NOTE I think we should drop the peer immediately
1109 if (!cluster())
1110 {
1111 fee_ = Resource::feeUnwantedData;
1112 return;
1113 }
1114
1115 for (int i = 0; i < m->clusternodes().size(); ++i)
1116 {
1117 protocol::TMClusterNode const& node = m->clusternodes(i);
1118
1119 std::string name;
1120 if (node.has_nodename())
1121 name = node.nodename();
1122
1123 auto const publicKey =
1124 parseBase58<PublicKey>(TokenType::NodePublic, node.publickey());
1125
1126 // NIKB NOTE We should drop the peer immediately if
1127 // they send us a public key we can't parse
1128 if (publicKey)
1129 {
1130 auto const reportTime =
1131 NetClock::time_point{NetClock::duration{node.reporttime()}};
1132
1133 app_.cluster().update(
1134 *publicKey, name, node.nodeload(), reportTime);
1135 }
1136 }
1137
1138 int loadSources = m->loadsources().size();
1139 if (loadSources != 0)
1140 {
1141 Resource::Gossip gossip;
1142 gossip.items.reserve(loadSources);
1143 for (int i = 0; i < m->loadsources().size(); ++i)
1144 {
1145 protocol::TMLoadSource const& node = m->loadsources(i);
1146 Resource::Gossip::Item item;
1147 item.address = beast::IP::Endpoint::from_string(node.name());
1148 item.balance = node.cost();
1149 if (item.address != beast::IP::Endpoint())
1150 gossip.items.push_back(item);
1151 }
1152 overlay_.resourceManager().importConsumers(name(), gossip);
1153 }
1154
1155 // Calculate the cluster fee:
1156 auto const thresh = app_.timeKeeper().now() - 90s;
1157 std::uint32_t clusterFee = 0;
1158
1159 std::vector<std::uint32_t> fees;
1160 fees.reserve(app_.cluster().size());
1161
1162 app_.cluster().for_each([&fees, thresh](ClusterNode const& status) {
1163 if (status.getReportTime() >= thresh)
1164 fees.push_back(status.getLoadFee());
1165 });
1166
1167 if (!fees.empty())
1168 {
1169 auto const index = fees.size() / 2;
1170 std::nth_element(fees.begin(), fees.begin() + index, fees.end());
1171 clusterFee = fees[index];
1172 }
1173
1174 app_.getFeeTrack().setClusterFee(clusterFee);

1191 }

1230void
1231PeerImp::handleTransaction(
1232 std::shared_ptr<protocol::TMTransaction> const& m,
1233 bool eraseTxQueue)
1234{
1235 if (tracking_.load() == Tracking::diverged)
1236 return;
1237
1238 if (app_.getOPs().isNeedNetworkLedger())
1239 {
1240 // If we've never been in synch, there's nothing we can do
1241 // with a transaction
1242 JLOG(p_journal_.debug())
1243 << "Ignoring incoming transaction: " << "Need network ledger";
1244 return;
1245 }
1246
1247 SerialIter sit(makeSlice(m->rawtransaction()));
1248
1249 try
1250 {
1251 auto stx = std::make_shared<STTx const>(sit);
1252 uint256 txID = stx->getTransactionID();
1253
1254 int flags;
1255 constexpr std::chrono::seconds tx_interval = 10s;
1256
1257 if (!app_.getHashRouter().shouldProcess(txID, id_, flags, tx_interval))
1258 {
1259 // we have seen this transaction recently
1260 if (flags & SF_BAD)
1261 {
1262 fee_ = Resource::feeInvalidSignature;
1263 JLOG(p_journal_.debug()) << "Ignoring known bad tx " << txID;
1264 }
1265
1266 // Erase only if the server has seen this tx. If the server has not
1267 // seen this tx then the tx could not has been queued for this peer.
1268 else if (eraseTxQueue && txReduceRelayEnabled())
1269 removeTxQueue(txID);
1270
1271 return;
1272 }
1273
1274 JLOG(p_journal_.debug()) << "Got tx " << txID;
1275
1276 bool checkSignature = true;
1277 if (cluster())
1278 {
1279 if (!m->has_deferred() || !m->deferred())
1280 {
1281 // Skip local checks if a server we trust
1282 // put the transaction in its open ledger
1283 flags |= SF_TRUSTED;
1284 }
1285
1286 // for non-validator nodes only -- localPublicKey is set for
1287 // validators only
1288 if (!app_.getValidationPublicKey())
1289 {
1290 // For now, be paranoid and have each validator
1291 // check each transaction, regardless of source
1292 checkSignature = false;
1293 }
1294 }
1295
1296 if (app_.getLedgerMaster().getValidatedLedgerAge() > 4min)
1297 {
1298 JLOG(p_journal_.trace())
1299 << "No new transactions until synchronized";
1300 }
1301 else if (
1302 app_.getJobQueue().getJobCount(jtTRANSACTION) >
1303 app_.config().MAX_TRANSACTIONS)
1304 {
1305 overlay_.incJqTransOverflow();
1306 JLOG(p_journal_.info()) << "Transaction queue is full";
1307 }
1308 else
1309 {
1310 app_.getJobQueue().addJob(
1311 jtTRANSACTION,
1312 "recvTransaction->checkTransaction",
1313 [weak = std::weak_ptr<PeerImp>(shared_from_this()),
1314 flags,
1315 checkSignature,
1316 stx]() {
1317 if (auto peer = weak.lock())
1318 peer->checkTransaction(flags, checkSignature, stx);
1319 });
1320 }
1321 }
1322 catch (std::exception const& ex)
1323 {
1324 JLOG(p_journal_.warn())
1325 << "Transaction invalid: " << strHex(m->rawtransaction())
1326 << ". Exception: " << ex.what();
1327 }
1328}

1608void
1609PeerImp::onMessage(std::shared_ptr<protocol::TMProposeSet> const& m)
1610{
1611 protocol::TMProposeSet& set = *m;
1612
1613 auto const sig = makeSlice(set.signature());
1614
1615 // Preliminary check for the validity of the signature: A DER encoded
1616 // signature can't be longer than 72 bytes.
1617 if ((std::clamp<std::size_t>(sig.size(), 64, 72) != sig.size()) ||
1618 (publicKeyType(makeSlice(set.nodepubkey())) != KeyType::secp256k1))
1619 {
1620 JLOG(p_journal_.warn()) << "Proposal: malformed";
1621 fee_ = Resource::feeInvalidSignature;
1622 return;
1623 }
1624
1625 if (!stringIsUint256Sized(set.currenttxhash()) ||
1626 !stringIsUint256Sized(set.previousledger()))
1627 {
1628 JLOG(p_journal_.warn()) << "Proposal: malformed";
1629 fee_ = Resource::feeInvalidRequest;
1630 return;
1631 }
1632
1633 // RH TODO: when isTrusted = false we should probably also cache a key
1634 // suppression for 30 seconds to avoid doing a relatively expensive lookup
1635 // every time a spam packet is received
1636 PublicKey const publicKey{makeSlice(set.nodepubkey())};
1637 auto const isTrusted = app_.validators().trusted(publicKey);
1638
1639 // If the operator has specified that untrusted proposals be dropped then
1640 // this happens here I.e. before further wasting CPU verifying the signature
1641 // of an untrusted key
1642 if (!isTrusted && app_.config().RELAY_UNTRUSTED_PROPOSALS == -1)
1643 return;
1644
1645 uint256 const proposeHash{set.currenttxhash()};
1646 uint256 const prevLedger{set.previousledger()};
1647
1648 NetClock::time_point const closeTime{NetClock::duration{set.closetime()}};
1649
1650 uint256 const suppression = proposalUniqueId(
1651 proposeHash,
1652 prevLedger,
1653 set.proposeseq(),
1654 closeTime,
1655 publicKey.slice(),
1656 sig);
1657
1658 if (auto [added, relayed] =
1659 app_.getHashRouter().addSuppressionPeerWithStatus(suppression, id_);
1660 !added)
1661 {
1662 // Count unique messages (Slots has it's own 'HashRouter'), which a peer
1663 // receives within IDLED seconds since the message has been relayed.
1664 if (reduceRelayReady() && relayed &&
1665 (stopwatch().now() - *relayed) < reduce_relay::IDLED)
1666 overlay_.updateSlotAndSquelch(
1667 suppression, publicKey, id_, protocol::mtPROPOSE_LEDGER);
1668 JLOG(p_journal_.trace()) << "Proposal: duplicate";
1669 return;
1670 }
1671
1672 if (!isTrusted)
1673 {
1674 if (tracking_.load() == Tracking::diverged)
1675 {
1676 JLOG(p_journal_.debug())
1677 << "Proposal: Dropping untrusted (peer divergence)";
1678 return;
1679 }
1680
1681 if (!cluster() && app_.getFeeTrack().isLoadedLocal())
1682 {
1683 JLOG(p_journal_.debug()) << "Proposal: Dropping untrusted (load)";
1684 return;
1685 }
1686 }
1687
1688 JLOG(p_journal_.trace())
1689 << "Proposal: " << (isTrusted ? "trusted" : "untrusted");
1690
1691 auto proposal = RCLCxPeerPos(
1692 publicKey,
1693 sig,
1694 suppression,
1695 RCLCxPeerPos::Proposal{
1696 prevLedger,
1697 set.proposeseq(),
1698 proposeHash,
1699 closeTime,
1700 app_.timeKeeper().closeTime(),
1701 calcNodeID(app_.validatorManifests().getMasterKey(publicKey))});
1702
1703 std::weak_ptr<PeerImp> weak = shared_from_this();
1704 app_.getJobQueue().addJob(
1705 isTrusted ? jtPROPOSAL_t : jtPROPOSAL_ut,
1706 "recvPropose->checkPropose",
1707 [weak, isTrusted, m, proposal]() {
1708 if (auto peer = weak.lock())
1709 peer->checkPropose(isTrusted, m, proposal);
1710 });
1711}

1955void
1956PeerImp::onValidatorListMessage(
1957 std::string const& messageType,
1958 std::string const& manifest,
1959 std::uint32_t version,
1960 std::vector<ValidatorBlobInfo> const& blobs)
1961{
1962 // If there are no blobs, the message is malformed (possibly because of
1963 // ValidatorList class rules), so charge accordingly and skip processing.
1964 if (blobs.empty())
1965 {
1966 JLOG(p_journal_.warn()) << "Ignored malformed " << messageType
1967 << " from peer " << remote_address_;
1968 // This shouldn't ever happen with a well-behaved peer
1969 fee_ = Resource::feeHighBurdenPeer;
1970 return;
1971 }
1972
1973 auto const hash = sha512Half(manifest, blobs, version);
1974
1975 JLOG(p_journal_.debug())
1976 << "Received " << messageType << " from " << remote_address_.to_string()
1977 << " (" << id_ << ")";
1978
1979 if (!app_.getHashRouter().addSuppressionPeer(hash, id_))
1980 {
1981 JLOG(p_journal_.debug())
1982 << messageType << ": received duplicate " << messageType;
1983 // Charging this fee here won't hurt the peer in the normal
1984 // course of operation (ie. refresh every 5 minutes), but
1985 // will add up if the peer is misbehaving.
1986 fee_ = Resource::feeUnwantedData;
1987 return;
1988 }
1989
1990 auto const applyResult = app_.validators().applyListsAndBroadcast(
1991 manifest,
1992 version,
1993 blobs,
1994 remote_address_.to_string(),
1995 hash,
1996 app_.overlay(),
1997 app_.getHashRouter(),
1998 app_.getOPs());
1999
2000 JLOG(p_journal_.debug())
2001 << "Processed " << messageType << " version " << version << " from "
2002 << (applyResult.publisherKey ? strHex(*applyResult.publisherKey)
2003 : "unknown or invalid publisher")
2004 << " from " << remote_address_.to_string() << " (" << id_
2005 << ") with best result " << to_string(applyResult.bestDisposition());
2006
2007 // Act based on the best result
2008 switch (applyResult.bestDisposition())
2009 {
2010 // New list
2011 case ListDisposition::accepted:
2012 // Newest list is expired, and that needs to be broadcast, too
2013 case ListDisposition::expired:
2014 // Future list
2015 case ListDisposition::pending: {
2016 std::lock_guard<std::mutex> sl(recentLock_);
2017
2018 XRPL_ASSERT(
2019 applyResult.publisherKey,
2020 "ripple::PeerImp::onValidatorListMessage : publisher key is "
2021 "set");
2022 auto const& pubKey = *applyResult.publisherKey;
2023#ifndef NDEBUG
2024 if (auto const iter = publisherListSequences_.find(pubKey);
2025 iter != publisherListSequences_.end())
2026 {
2027 XRPL_ASSERT(
2028 iter->second < applyResult.sequence,
2029 "ripple::PeerImp::onValidatorListMessage : lower sequence");
2030 }
2031#endif
2032 publisherListSequences_[pubKey] = applyResult.sequence;
2033 }
2034 break;
2035 case ListDisposition::same_sequence:
2036 case ListDisposition::known_sequence:
2037#ifndef NDEBUG
2038 {
2039 std::lock_guard<std::mutex> sl(recentLock_);
2040 XRPL_ASSERT(
2041 applyResult.sequence && applyResult.publisherKey,
2042 "ripple::PeerImp::onValidatorListMessage : nonzero sequence "
2043 "and set publisher key");
2044 XRPL_ASSERT(
2045 publisherListSequences_[*applyResult.publisherKey] <=
2046 applyResult.sequence,
2047 "ripple::PeerImp::onValidatorListMessage : maximum sequence");
2048 }
2049#endif // !NDEBUG
2050
2051 break;
2052 case ListDisposition::stale:
2053 case ListDisposition::untrusted:
2054 case ListDisposition::invalid:
2055 case ListDisposition::unsupported_version:
2056 break;
2057 default:
2058 UNREACHABLE(
2059 "ripple::PeerImp::onValidatorListMessage : invalid best list "
2060 "disposition");
2061 }
2062
2063 // Charge based on the worst result
2064 switch (applyResult.worstDisposition())
2065 {
2066 case ListDisposition::accepted:
2067 case ListDisposition::expired:
2068 case ListDisposition::pending:
2069 // No charges for good data
2070 break;
2071 case ListDisposition::same_sequence:
2072 case ListDisposition::known_sequence:
2073 // Charging this fee here won't hurt the peer in the normal
2074 // course of operation (ie. refresh every 5 minutes), but
2075 // will add up if the peer is misbehaving.
2076 fee_ = Resource::feeUnwantedData;
2077 break;
2078 case ListDisposition::stale:
2079 // There are very few good reasons for a peer to send an
2080 // old list, particularly more than once.
2081 fee_ = Resource::feeBadData;
2082 break;
2083 case ListDisposition::untrusted:
2084 // Charging this fee here won't hurt the peer in the normal
2085 // course of operation (ie. refresh every 5 minutes), but
2086 // will add up if the peer is misbehaving.
2087 fee_ = Resource::feeUnwantedData;
2088 break;
2089 case ListDisposition::invalid:
2090 // This shouldn't ever happen with a well-behaved peer
2091 fee_ = Resource::feeInvalidSignature;
2092 break;
2093 case ListDisposition::unsupported_version:
2094 // During a version transition, this may be legitimate.
2095 // If it happens frequently, that's probably bad.
2096 fee_ = Resource::feeBadData;
2097 break;
2098 default:
2099 UNREACHABLE(
2100 "ripple::PeerImp::onValidatorListMessage : invalid worst list "
2101 "disposition");
2102 }
2103
2104 // Log based on all the results.
2105 for (auto const& [disp, count] : applyResult.dispositions)
2106 {
2107 switch (disp)
2108 {
2109 // New list
2110 case ListDisposition::accepted:
2111 JLOG(p_journal_.debug())
2112 << "Applied " << count << " new " << messageType
2113 << "(s) from peer " << remote_address_;
2114 break;
2115 // Newest list is expired, and that needs to be broadcast, too
2116 case ListDisposition::expired:
2117 JLOG(p_journal_.debug())
2118 << "Applied " << count << " expired " << messageType
2119 << "(s) from peer " << remote_address_;
2120 break;
2121 // Future list
2122 case ListDisposition::pending:
2123 JLOG(p_journal_.debug())
2124 << "Processed " << count << " future " << messageType
2125 << "(s) from peer " << remote_address_;
2126 break;
2127 case ListDisposition::same_sequence:
2128 JLOG(p_journal_.warn())
2129 << "Ignored " << count << " " << messageType
2130 << "(s) with current sequence from peer "
2131 << remote_address_;
2132 break;
2133 case ListDisposition::known_sequence:
2134 JLOG(p_journal_.warn())
2135 << "Ignored " << count << " " << messageType
2136 << "(s) with future sequence from peer " << remote_address_;
2137 break;
2138 case ListDisposition::stale:
2139 JLOG(p_journal_.warn())
2140 << "Ignored " << count << "stale " << messageType
2141 << "(s) from peer " << remote_address_;
2142 break;
2143 case ListDisposition::untrusted:
2144 JLOG(p_journal_.warn())
2145 << "Ignored " << count << " untrusted " << messageType
2146 << "(s) from peer " << remote_address_;
2147 break;
2148 case ListDisposition::unsupported_version:
2149 JLOG(p_journal_.warn())
2150 << "Ignored " << count << "unsupported version "
2151 << messageType << "(s) from peer " << remote_address_;
2152 break;
2153 case ListDisposition::invalid:
2154 JLOG(p_journal_.warn())
2155 << "Ignored " << count << "invalid " << messageType
2156 << "(s) from peer " << remote_address_;
2157 break;
2158 default:
2159 UNREACHABLE(
2160 "ripple::PeerImp::onValidatorListMessage : invalid list "
2161 "disposition");
2162 }
2163 }
