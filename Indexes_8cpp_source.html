99 uint256
100 getQualityIndex(uint256 const& uBase, const std::uint64_t uNodeDir)
101 {
102  // Indexes are stored in big endian format: they print as hex as stored.
103  // Most significant bytes are first. Least significant bytes represent
104  // adjacent entries. We place uNodeDir in the 8 right most bytes to be
105  // adjacent. Want uNodeDir in big endian format so ++ goes to the next
106  // entry for indexes.
107  uint256 uNode(uBase);
108 
109  // TODO(tom): there must be a better way.
110  // VFALCO [base_uint] This assumes a certain storage format
111  ((std::uint64_t*)uNode.end())[-1] = boost::endian::native_to_big(uNodeDir);
112 
113  return uNode;
114 }

154 uint256
155 getSignerListIndex(AccountID const& account)
156 {
157  // We are prepared for there to be multiple SignerLists in the future,
158  // but we don't have them yet. In anticipation of multiple SignerLists
159  // We supply a 32-bit ID to locate the SignerList. Until we actually
160  // *have* multiple signer lists, we can default that ID to zero.
161  return sha512Half(
162  std::uint16_t(spaceSignerList),
163  account,
164  std::uint32_t(0)); // 0 == default SignerList ID.
165 }
