81Json::Value
82ledgerEntry(
83 jtx::Env& env,
84 jtx::Account const& subject,
85 jtx::Account const& issuer,
86 std::string_view credType)
87{
88 Json::Value jvParams;
89 jvParams[jss::ledger_index] = jss::validated;
90 jvParams[jss::credential][jss::subject] = subject.human();
91 jvParams[jss::credential][jss::issuer] = issuer.human();
92 jvParams[jss::credential][jss::credential_type] = strHex(credType);
93 return env.rpc("json", "ledger_entry", to_string(jvParams));
94}
