120  for (std::size_t j = 0; j < 32; j++)
121  {
122  uint256 digest;

124 
125  auto sig = signDigest(pk, sk, digest);
126 
127  BEAST_EXPECT(sig.size() != 0);
128  BEAST_EXPECT(verifyDigest(pk, digest, sig, true));
129 
130  // Wrong digest:
131  BEAST_EXPECT(!verifyDigest(pk, ~digest, sig, true));
132 
133  // Slightly change the signature:
134  if (auto ptr = sig.data())
135  ptr[j % sig.size()]++;
136 
137  // Wrong signature:
138  BEAST_EXPECT(!verifyDigest(pk, digest, sig, true));
139 
140  // Wrong digest and signature:
141  BEAST_EXPECT(!verifyDigest(pk, ~digest, sig, true));

146  void
147  testSigning(KeyType type)
148  {
149  testcase << to_string(type) << ": signing & verification";
150 
151  for (std::size_t i = 0; i < 32; i++)
152  {
153  auto const [pk, sk] = randomKeyPair(type);
154 
155  BEAST_EXPECT(pk == derivePublicKey(type, sk));
156  BEAST_EXPECT(*publicKeyType(pk) == type);
157 
158  for (std::size_t j = 0; j < 32; j++)
159  {
160  std::vector<std::uint8_t> data(64 + (8 * i) + j);
161  beast::rngfill(data.data(), data.size(), crypto_prng());
162 
163  auto sig = sign(pk, sk, makeSlice(data));
164 
165  BEAST_EXPECT(sig.size() != 0);
166  BEAST_EXPECT(verify(pk, makeSlice(data), sig, true));
167 
168  // Construct wrong data:
169  auto badData = data;
170 
171  // swaps the smallest and largest elements in buffer
172  std::iter_swap(
173  std::min_element(badData.begin(), badData.end()),
174  std::max_element(badData.begin(), badData.end()));
175 
176  // Wrong data: should fail
177  BEAST_EXPECT(!verify(pk, makeSlice(badData), sig, true));
178 
179  // Slightly change the signature:
180  if (auto ptr = sig.data())
181  ptr[j % sig.size()]++;
182 
183  // Wrong signature: should fail
184  BEAST_EXPECT(!verify(pk, makeSlice(data), sig, true));
185 
186  // Wrong data and signature: should fail
187  BEAST_EXPECT(!verify(pk, makeSlice(badData), sig, true));
188  }
189  }

192  void
193  testBase58()
194  {
195  testcase("Base58");
196 
197  // Ensure that parsing some well-known secret keys works
198  {
199  auto const sk1 = generateSecretKey(
200  KeyType::secp256k1, generateSeed("masterpassphrase"));
201 
202  auto const sk2 = parseBase58<SecretKey>(
203  TokenType::NodePrivate,
204  "pnen77YEeUd4fFKG7iycBWcwKpTaeFRkW2WFostaATy1DSupwXe");
205  BEAST_EXPECT(sk2);
206 
207  BEAST_EXPECT(sk1 == *sk2);
208  }
209 
210  {
211  auto const sk1 = generateSecretKey(
212  KeyType::ed25519, generateSeed("masterpassphrase"));
213 
214  auto const sk2 = parseBase58<SecretKey>(
215  TokenType::NodePrivate,
216  "paKv46LztLqK3GaKz1rG2nQGN6M4JLyRtxFBYFTw4wAVHtGys36");
217  BEAST_EXPECT(sk2);
218 
219  BEAST_EXPECT(sk1 == *sk2);
220  }
221 
222  // Try converting short, long and malformed data
223  BEAST_EXPECT(!parseBase58<SecretKey>(TokenType::NodePrivate, ""));
224  BEAST_EXPECT(!parseBase58<SecretKey>(TokenType::NodePrivate, " "));
225  BEAST_EXPECT(!parseBase58<SecretKey>(
226  TokenType::NodePrivate, "!35gty9mhju8nfjl"));
227 
228  auto const good = toBase58(TokenType::NodePrivate, randomSecretKey());
229 
230  // Short (non-empty) strings
231  {
232  auto s = good;
233 
234  // Remove all characters from the string in random order:
235  std::hash<std::string> r;
236 
237  while (!s.empty())
238  {
239  s.erase(r(s) % s.size(), 1);
240  BEAST_EXPECT(
241  !parseBase58<SecretKey>(TokenType::NodePrivate, s));
242  }
243  }
244 
245  // Long strings
246  for (std::size_t i = 1; i != 16; i++)
247  {
248  auto s = good;
249  s.resize(s.size() + i, s[i % s.size()]);
250  BEAST_EXPECT(!parseBase58<SecretKey>(TokenType::NodePrivate, s));
251  }
252 
253  // Strings with invalid Base58 characters
254  for (auto c : std::string("0IOl"))
255  {
256  for (std::size_t i = 0; i != good.size(); ++i)
257  {
258  auto s = good;
259  s[i % s.size()] = c;
260  BEAST_EXPECT(
261  !parseBase58<SecretKey>(TokenType::NodePrivate, s));
262  }
263  }
264 
265  // Strings with incorrect prefix
266  {
267  auto s = good;
268 
269  for (auto c : std::string("ansrJqtv7"))
270  {
271  s[0] = c;
272  BEAST_EXPECT(
273  !parseBase58<SecretKey>(TokenType::NodePrivate, s));
274  }
275  }
276 
277  // Try some random secret keys
278  std::array<SecretKey, 32> keys;
279 
280  for (std::size_t i = 0; i != keys.size(); ++i)
281  keys[i] = randomSecretKey();
282 
283  for (std::size_t i = 0; i != keys.size(); ++i)
284  {
285  auto const si = toBase58(TokenType::NodePrivate, keys[i]);
286  BEAST_EXPECT(!si.empty());
287 
288  auto const ski = parseBase58<SecretKey>(TokenType::NodePrivate, si);
289  BEAST_EXPECT(ski && keys[i] == *ski);
290 
291  for (std::size_t j = i; j != keys.size(); ++j)
292  {
293  BEAST_EXPECT((keys[i] == keys[j]) == (i == j));
294 
295  auto const sj = toBase58(TokenType::NodePrivate, keys[j]);
296 
297  BEAST_EXPECT((si == sj) == (i == j));
298 
299  auto const skj =
300  parseBase58<SecretKey>(TokenType::NodePrivate, sj);
301  BEAST_EXPECT(skj && keys[j] == *skj);
302 
303  BEAST_EXPECT((*ski == *skj) == (i == j));
304  }
305  }
306  }
