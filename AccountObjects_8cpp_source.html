49 Json::Value doAccountObjects (RPC::JsonContext& context)
50 {
51  auto const& params = context.params;
52  if (! params.isMember (jss::account))
53  return RPC::missing_field_error (jss::account);
54 
55  std::shared_ptr<ReadView const> ledger;
56  auto result = RPC::lookupLedger (ledger, context);
57  if (ledger == nullptr)
58  return result;
59 
60  AccountID accountID;
61  {
62  auto const strIdent = params[jss::account].asString ();
63  if (auto jv = RPC::accountFromString (accountID, strIdent))
64  {
65  for (auto it = jv.begin (); it != jv.end (); ++it)
66  result[it.memberName ()] = *it;
67 
68  return result;
69  }
70  }
71 
72  if (! ledger->exists(keylet::account (accountID)))
73  return rpcError (rpcACT_NOT_FOUND);
74 
75  boost::optional<std::vector<LedgerEntryType>> typeFilter;
76 
77  if (params.isMember(jss::deletion_blockers_only) &&
78  params[jss::deletion_blockers_only].asBool())
79  {
80  struct {
81  Json::StaticString name;
82  LedgerEntryType type;
83  } static constexpr deletionBlockers[] = {
84  { jss::check, ltCHECK },
85  { jss::escrow, ltESCROW },
86  { jss::payment_channel, ltPAYCHAN },
87  { jss::state, ltRIPPLE_STATE }
88  };
89 
90  typeFilter.emplace();
91  typeFilter->reserve(std::size(deletionBlockers));
92 
93  for (auto [name, type] : deletionBlockers)
94  {
95  if (params.isMember(jss::type) &&
96  name != params[jss::type])
97  {
98  continue;
99  }
100 
101  typeFilter->push_back(type);
102  }
103  }
104  else
105  {
106  auto [rpcStatus, type] = RPC::chooseLedgerEntryType(params);
107 
108  if (rpcStatus)
109  {
110  result.clear();
111  rpcStatus.inject(result);
112  return result;
113  }
114  else if (type != ltINVALID)
115  {
116  typeFilter = std::vector<LedgerEntryType>({ type });
117  }
118  }
119 
120  unsigned int limit;
121  if (auto err = readLimitField(limit, RPC::Tuning::accountObjects, context))
122  return *err;
123 
124  uint256 dirIndex;
125  uint256 entryIndex;
126  if (params.isMember (jss::marker))
127  {
128  auto const& marker = params[jss::marker];
129  if (! marker.isString ())
130  return RPC::expected_field_error (jss::marker, "string");
131 
132  std::stringstream ss (marker.asString ());
133  std::string s;
134  if (!std::getline(ss, s, ','))
135  return RPC::invalid_field_error (jss::marker);
136 
137  if (! dirIndex.SetHex (s))
138  return RPC::invalid_field_error (jss::marker);
139 
140  if (! std::getline (ss, s, ','))
141  return RPC::invalid_field_error (jss::marker);
142 
143  if (! entryIndex.SetHex (s))
144  return RPC::invalid_field_error (jss::marker);
145  }
146 
147  if (! RPC::getAccountObjects (*ledger, accountID, typeFilter,
148  dirIndex, entryIndex, limit, result))
149  {
150  result[jss::account_objects] = Json::arrayValue;
151  }
152 
153  result[jss::account] = context.app.accountIDCache().toBase58 (accountID);
154  context.loadType = Resource::feeMediumBurdenRPC;
155  return result;
156 }
