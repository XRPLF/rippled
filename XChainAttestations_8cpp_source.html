710template <class TAttestation>
711XChainAttestationsBase<TAttestation>::XChainAttestationsBase(
712 Json::Value const& v)
713{
714 if (!v.isObject())
715 {
716 Throw<std::runtime_error>(
717 "XChainAttestationsBase can only be specified with an 'object' "
718 "Json value");
719 }
720
721 attestations_ = [&] {
722 auto const jAtts = v[jss::attestations];
723
724 if (jAtts.size() > maxAttestations)
725 Throw<std::runtime_error>(
726 "XChainAttestationsBase exceeded max number of attestations");
727
728 std::vector<TAttestation> r;
729 r.reserve(jAtts.size());
730 for (auto const& a : jAtts)
731 r.emplace_back(a);
732 return r;
733 }();
734}
735
736template <class TAttestation>
737XChainAttestationsBase<TAttestation>::XChainAttestationsBase(STArray const& arr)
738{
739 if (arr.size() > maxAttestations)
740 Throw<std::runtime_error>(
741 "XChainAttestationsBase exceeded max number of attestations");
742
743 attestations_.reserve(arr.size());
744 for (auto const& o : arr)
745 attestations_.emplace_back(o);
746}

753 for (auto const& e : attestations_)
