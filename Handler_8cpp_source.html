30 template <typename Function>
31 Handler::Method<Json::Value>
32 byRef(Function const& f)
33 {
34  return [f](JsonContext& context, Json::Value& result) {
35  result = f(context);
36  if (result.type() != Json::objectValue)
37  {
38  assert(false);
39  result = RPC::makeObjectValue(result);
40  }
41 
42  return Status();
43  };

46 template <class Object, class HandlerImpl>
47 Status
48 handle(JsonContext& context, Object& object)
49 {
50  HandlerImpl handler(context);
51 
52  auto status = handler.check();
53  if (status)
54  status.inject(object);
55  else
56  handler.writeResult(object);
57  return status;
58 };

165 class HandlerTable
166 {
167 private:
168  template <std::size_t N>
169  explicit HandlerTable(const Handler (&entries)[N])
170  {
171  for (std::size_t i = 0; i < N; ++i)
172  {
173  auto const& entry = entries[i];
174  assert(table_.find(entry.name_) == table_.end());
175  table_[entry.name_] = entry;
176  }
177 
178  // This is where the new-style handlers are added.
179  addHandler<LedgerHandler>();
180  addHandler<VersionHandler>();
181  }
182 
183 public:
184  static HandlerTable const&
185  instance()
186  {
187  static HandlerTable const handlerTable(handlerArray);
188  return handlerTable;
189  }
190 
191  Handler const*
192  getHandler(unsigned version, bool betaEnabled, std::string const& name)
193  const
194  {
195  if (version < RPC::apiMinimumSupportedVersion ||
196  version > (betaEnabled ? RPC::apiBetaVersion
197  : RPC::apiMaximumSupportedVersion))
198  return nullptr;
199  auto i = table_.find(name);
200  return i == table_.end() ? nullptr : &i->second;
201  }
202 
203  std::vector<char const*>
204  getHandlerNames() const
205  {
206  std::vector<char const*> ret;
207  ret.reserve(table_.size());
208  for (auto const& i : table_)
209  ret.push_back(i.second.name_);
210  return ret;
211  }
212 
213 private:
214  std::map<std::string, Handler> table_;
215 
216  template <class HandlerImpl>
217  void
218  addHandler()
219  {
220  assert(table_.find(HandlerImpl::name()) == table_.end());
221 
222  Handler h;
223  h.name_ = HandlerImpl::name();
224  h.valueMethod_ = &handle<Json::Value, HandlerImpl>;
225  h.role_ = HandlerImpl::role();
226  h.condition_ = HandlerImpl::condition();
227 
228  table_[HandlerImpl::name()] = h;
229  }
230 };
