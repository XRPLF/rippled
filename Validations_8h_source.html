144 inline bool
145 isCurrent(
146  ValidationParms const& p,
147  NetClock::time_point now,
148  NetClock::time_point signTime,
149  NetClock::time_point seenTime)
150 {
151  // Because this can be called on untrusted, possibly
152  // malicious validations, we do our math in a way
153  // that avoids any chance of overflowing or underflowing
154  // the signing time.
155 
156  return (signTime > (now - p.validationCURRENT_EARLY)) &&
157  (signTime < (now + p.validationCURRENT_WALL)) &&
158  ((seenTime == NetClock::time_point{}) ||
159  (seenTime < (now + p.validationCURRENT_LOCAL)));

327  // Remove support of a validated ledger
328  void
329  removeTrie(
330  std::lock_guard<Mutex> const&,
331  NodeID const& nodeID,
332  Validation const& val)
333  {
334  {
335  auto it =
336  acquiring_.find(std::make_pair(val.seq(), val.ledgerID()));
337  if (it != acquiring_.end())
338  {
339  it->second.erase(nodeID);
340  if (it->second.empty())
341  acquiring_.erase(it);
342  }
343  }
344  {
345  auto it = lastLedger_.find(nodeID);
346  if (it != lastLedger_.end() && it->second.id() == val.ledgerID())
347  {
348  trie_.remove(it->second);
349  lastLedger_.erase(nodeID);
350  }
351  }

354  // Check if any pending acquire ledger requests are complete
355  void
356  checkAcquired(std::lock_guard<Mutex> const& lock)
357  {
358  for (auto it = acquiring_.begin(); it != acquiring_.end();)
359  {

361  adaptor_.acquire(it->first.second))
362  {
363  for (NodeID const& nodeID : it->second)
364  updateTrie(lock, nodeID, *ledger);
365 
366  it = acquiring_.erase(it);
367  }
368  else

402  void
403  updateTrie(
404  std::lock_guard<Mutex> const& lock,
405  NodeID const& nodeID,
406  Validation const& val,
407  boost::optional<std::pair<Seq, ID>> prior)
408  {
409  assert(val.trusted());
410 
411  // Clear any prior acquiring ledger for this node
412  if (prior)
413  {
414  auto it = acquiring_.find(*prior);
415  if (it != acquiring_.end())
416  {
417  it->second.erase(nodeID);
418  if (it->second.empty())
419  acquiring_.erase(it);
420  }
421  }
422 
423  checkAcquired(lock);
424 
425  std::pair<Seq, ID> valPair{val.seq(), val.ledgerID()};
426  auto it = acquiring_.find(valPair);
427  if (it != acquiring_.end())
428  {
429  it->second.insert(nodeID);
430  }
431  else
432  {
433  if (boost::optional<Ledger> ledger =
434  adaptor_.acquire(val.ledgerID()))
435  updateTrie(lock, nodeID, *ledger);
436  else
437  acquiring_[valPair].insert(nodeID);
438  }
439  }

480  template <class Pre, class F>
481  void
482  current(std::lock_guard<Mutex> const& lock, Pre&& pre, F&& f)
483  {
484  NetClock::time_point t = adaptor_.now();
485  pre(current_.size());
486  auto it = current_.begin();
487  while (it != current_.end())
488  {
489  // Check for staleness
490  if (!isCurrent(
491  parms_, t, it->second.signTime(), it->second.seenTime()))
492  {
493  removeTrie(lock, it->first, it->second);
494  it = current_.erase(it);
495  }
496  else
497  {
498  auto cit = typename decltype(current_)::const_iterator{it};
499  // contains a live record
500  f(cit->first, cit->second);
501  ++it;
502  }
503  }
504  }

591  ValStatus
592  add(NodeID const& nodeID, Validation const& val)
593  {
594  if (!isCurrent(parms_, adaptor_.now(), val.signTime(), val.seenTime()))
595  return ValStatus::stale;
596 
597  {
598  std::lock_guard lock{mutex_};
599 
600  // Check that validation sequence is greater than any non-expired
601  // validations sequence from that validator
602  auto const now = byLedger_.clock().now();
603  SeqEnforcer<Seq>& enforcer = seqEnforcers_[nodeID];
604  if (!enforcer(now, val.seq(), parms_))
605  return ValStatus::badSeq;
606 
607  byLedger_[val.ledgerID()].insert_or_assign(nodeID, val);
608 
609  auto const [it, inserted] = current_.emplace(nodeID, val);
610  if (!inserted)
611  {
612  // Replace existing only if this one is newer
613  Validation& oldVal = it->second;
614  if (val.signTime() > oldVal.signTime())
615  {
616  std::pair<Seq, ID> old(oldVal.seq(), oldVal.ledgerID());
617  it->second = val;
618  if (val.trusted())
619  updateTrie(lock, nodeID, val, old);
620  }
621  else
622  return ValStatus::stale;
623  }
624  else if (val.trusted())
625  {
626  updateTrie(lock, nodeID, val, boost::none);
627  }
628  }
629  return ValStatus::current;
630  }

708  boost::optional<std::pair<Seq, ID>>
709  getPreferred(Ledger const& curr)
710  {
711  std::lock_guard lock{mutex_};
712  boost::optional<SpanTip<Ledger>> preferred =
713  withTrie(lock, [this](LedgerTrie<Ledger>& trie) {
714  return trie.getPreferred(localSeqEnforcer_.largest());
715  });
716  // No trusted validations to determine branch
717  if (!preferred)
718  {
719  // fall back to majority over acquiring ledgers
720  auto it = std::max_element(
721  acquiring_.begin(),
722  acquiring_.end(),
723  [](auto const& a, auto const& b) {
724  std::pair<Seq, ID> const& aKey = a.first;
725  typename hash_set<NodeID>::size_type const& aSize =
726  a.second.size();
727  std::pair<Seq, ID> const& bKey = b.first;
728  typename hash_set<NodeID>::size_type const& bSize =
729  b.second.size();
730  // order by number of trusted peers validating that ledger
731  // break ties with ledger ID
732  return std::tie(aSize, aKey.second) <
733  std::tie(bSize, bKey.second);
734  });
735  if (it != acquiring_.end())
736  return it->first;
737  return boost::none;
738  }
739 
740  // If we are the parent of the preferred ledger, stick with our
741  // current ledger since we might be about to generate it
742  if (preferred->seq == curr.seq() + Seq{1} &&
743  preferred->ancestor(curr.seq()) == curr.id())
744  return std::make_pair(curr.seq(), curr.id());
745 
746  // A ledger ahead of us is preferred regardless of whether it is
747  // a descendant of our working ledger or it is on a different chain
748  if (preferred->seq > curr.seq())
749  return std::make_pair(preferred->seq, preferred->id);
750 
751  // Only switch to earlier or same sequence number
752  // if it is a different chain.
753  if (curr[preferred->seq] != preferred->id)
754  return std::make_pair(preferred->seq, preferred->id);
755 
756  // Stick with current ledger
757  return std::make_pair(curr.seq(), curr.id());
758  }

794  ID
795  getPreferredLCL(
796  Ledger const& lcl,
797  Seq minSeq,
798  hash_map<ID, std::uint32_t> const& peerCounts)
799  {
800  boost::optional<std::pair<Seq, ID>> preferred = getPreferred(lcl);
801 
802  // Trusted validations exist, but stick with local preferred ledger if
803  // preferred is in the past
804  if (preferred)
805  return (preferred->first >= minSeq) ? preferred->second : lcl.id();
806 
807  // Otherwise, rely on peer ledgers
808  auto it = std::max_element(
809  peerCounts.begin(), peerCounts.end(), [](auto& a, auto& b) {
810  // Prefer larger counts, then larger ids on ties
811  // (max_element expects this to return true if a < b)
812  return std::tie(a.second, a.first) <
813  std::tie(b.second, b.first);
814  });
815 
816  if (it != peerCounts.end())
817  return it->first;
818  return lcl.id();
819  }
