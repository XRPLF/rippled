90void
91hash_append(Hasher& h, beast::IP::Address const& addr) noexcept
92{
93 using beast::hash_append;
94 if (addr.is_v4())
95 hash_append(h, addr.to_v4().to_bytes());
96 else if (addr.is_v6())

99 UNREACHABLE("beast::hash_append : invalid address type");
