235bool
236transResultInfo(TER code, std::string& token, std::string& text)
237{
238 auto& results = transResults();
239
240 auto const r = results.find(TERtoInt(code));
241
242 if (r == results.end())
243 return false;
244
245 token = r->second.first;
246 text = r->second.second;
247 return true;
248}
249
250std::string
251transToken(TER code)
252{
253 std::string token;
254 std::string text;
255
256 return transResultInfo(code, token, text) ? token : "-";
257}
258
259std::string
260transHuman(TER code)
261{
262 std::string token;
263 std::string text;
264
265 return transResultInfo(code, token, text) ? text : "-";
266}

268std::optional<TER>
269transCode(std::string const& token)
270{
271 static auto const results = [] {
272 auto& byTer = transResults();
273 auto range = boost::make_iterator_range(byTer.begin(), byTer.end());
274 auto tRange = boost::adaptors::transform(range, [](auto const& r) {
275 return std::make_pair(r.second.first, r.first);
276 });
277 std::unordered_map<std::string, TERUnderlyingType> const byToken(
278 tRange.begin(), tRange.end());
279 return byToken;
280 }();
281
282 auto const r = results.find(token);
283
284 if (r == results.end())
285 return std::nullopt;
286
287 return TER::fromInt(r->second);
288}
