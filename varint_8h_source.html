53 // Returns: Number of bytes consumed or 0 on error,
54 // if the buffer was too small or t overflowed.
55 //
56 template <class = void>
57 std::size_t
58 read_varint (void const* buf,
59  std::size_t buflen, std::size_t& t)
60 {
61  t = 0;
62  std::uint8_t const* p =
63  reinterpret_cast<
64  std::uint8_t const*>(buf);
65  std::size_t n = 0;
66  while (p[n] & 0x80)
67  if (++n >= buflen)
68  return 0;
69  if (++n > buflen)
70  return 0;
71  // Special case for 0
72  if (n == 1 && *p == 0)
73  {
74  t = 0;
75  return 1;
76  }
77  auto const used = n;
78  while (n--)
79  {
80  auto const d = p[n];
81  auto const t0 = t;
82  t *= 127;
83  t += d & 0x7f;
84  if (t <= t0)
85  return 0; // overflow
86  }
87  return used;
88 }

106 template <class = void>
107 std::size_t
108 write_varint (void* p0, std::size_t v)
109 {
110  std::uint8_t* p = reinterpret_cast<
111  std::uint8_t*>(p0);
112  do
113  {
114  std::uint8_t d =
115  v % 127;
116  v /= 127;
117  if (v != 0)
118  d |= 0x80;
119  *p++ = d;
120  }
121  while (v != 0);
122  return p - reinterpret_cast<
123  std::uint8_t*>(p0);
124 }
