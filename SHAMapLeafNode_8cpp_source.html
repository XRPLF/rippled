69std::string
70SHAMapLeafNode::getString(const SHAMapNodeID& id) const
71{
72 std::string ret = SHAMapTreeNode::getString(id);
73
74 auto const type = getType();
75
76 if (type == SHAMapNodeType::tnTRANSACTION_NM)
77 ret += ",txn\n";
78 else if (type == SHAMapNodeType::tnTRANSACTION_MD)
79 ret += ",txn+md\n";
80 else if (type == SHAMapNodeType::tnACCOUNT_STATE)
81 ret += ",as\n";
82 else
83 ret += ",leaf\n";
84
85 ret += " Tag=";
86 ret += to_string(item_->key());
87 ret += "\n Hash=";
88 ret += to_string(hash_);
89 ret += "/";
90 ret += std::to_string(item_->size());
91 return ret;
92}
