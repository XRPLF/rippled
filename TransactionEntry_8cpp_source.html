35 Json::Value
36 doTransactionEntry(RPC::JsonContext& context)
37 {
38  std::shared_ptr<ReadView const> lpLedger;
39  Json::Value jvResult = RPC::lookupLedger(lpLedger, context);
40 
41  if (!lpLedger)
42  return jvResult;
43 
44  if (!context.params.isMember(jss::tx_hash))
45  {
46  jvResult[jss::error] = "fieldNotFoundTransaction";
47  }
48  else if (jvResult.get(jss::ledger_hash, Json::nullValue).isNull())
49  {
50  // We don't work on ledger current.
51 
52  // XXX We don't support any transaction yet.
53  jvResult[jss::error] = "notYetImplemented";
54  }
55  else
56  {
57  uint256 uTransID;
58  // XXX Relying on trusted WSS client. Would be better to have a strict
59  // routine, returning success or failure.
60  uTransID.SetHex(context.params[jss::tx_hash].asString());
61 
62  auto [sttx, stobj] = lpLedger->txRead(uTransID);
63  if (!sttx)
64  {
65  jvResult[jss::error] = "transactionNotFound";
66  }
67  else
68  {
69  jvResult[jss::tx_json] = sttx->getJson(JsonOptions::none);
70  if (stobj)
71  jvResult[jss::metadata] = stobj->getJson(JsonOptions::none);
72  // 'accounts'
73  // 'engine_...'
74  // 'ledger_...'
75  }
76  }
77 
78  return jvResult;
79 }
