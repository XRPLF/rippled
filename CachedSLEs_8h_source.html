68  {
69  {
70  std::lock_guard lock(mutex_);
71  auto iter = map_.find(digest);
72  if (iter != map_.end())
73  {
74  ++hit_;
75  map_.touch(iter);
76  return iter->second;
77  }
78  }
79  auto sle = h();
80  if (!sle)
81  return nullptr;
82  std::lock_guard lock(mutex_);
83  ++miss_;
84  auto const [it, inserted] = map_.emplace(digest, std::move(sle));
85  if (!inserted)
86  map_.touch(it);
87  return it->second;
88  }
