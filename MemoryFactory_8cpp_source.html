131  Status
132  fetch(void const* key, std::shared_ptr<NodeObject>* pObject) override
133  {
134  assert(db_);
135  uint256 const hash(uint256::fromVoid(key));
136 
137  std::lock_guard _(db_->mutex);
138 
139  Map::iterator iter = db_->table.find(hash);
140  if (iter == db_->table.end())
141  {
142  pObject->reset();
143  return notFound;
144  }
145  *pObject = iter->second;
146  return ok;
147  }

149  std::pair<std::vector<std::shared_ptr<NodeObject>>, Status>
150  fetchBatch(std::vector<uint256 const*> const& hashes) override
151  {
152  std::vector<std::shared_ptr<NodeObject>> results;
153  results.reserve(hashes.size());
154  for (auto const& h : hashes)
155  {
156  std::shared_ptr<NodeObject> nObj;
157  Status status = fetch(h->begin(), &nObj);
158  if (status != ok)
159  results.push_back({});
160  else
161  results.push_back(nObj);
162  }
163 
164  return {results, ok};
165  }
