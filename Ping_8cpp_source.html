31Json::Value
32doPing(RPC::JsonContext& context)
33{
34 Json::Value ret(Json::objectValue);
35 switch (context.role)
36 {
37 case Role::ADMIN:
38 ret[jss::role] = "admin";
39 break;
40 case Role::IDENTIFIED:
41 ret[jss::role] = "identified";
42 ret[jss::username] = std::string{context.headers.user};
43 if (context.headers.forwardedFor.size())
44 ret[jss::ip] = std::string{context.headers.forwardedFor};
45 break;
46 case Role::PROXY:
47 ret[jss::role] = "proxied";
48 ret[jss::ip] = std::string{context.headers.forwardedFor};
49 default:;
50 }
51
52 // This is only accessible on ws sessions.
53 if (context.infoSub)
54 {
55 if (context.infoSub->getConsumer().isUnlimited())
56 ret[jss::unlimited] = true;
57 }
58
59 return ret;
60}
