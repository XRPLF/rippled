100 bool
101 STValidation::isValid() const noexcept
102 {
103  if (!valid_)
104  {
105  assert(publicKeyType(getSignerPublic()) == KeyType::secp256k1);
106 
107  valid_ = verifyDigest(
108  getSignerPublic(),
109  getSigningHash(),
110  makeSlice(getFieldVL(sfSignature)),
111  getFlags() & vfFullyCanonicalSig);
112  }
113 
114  return valid_.value();
115 }
