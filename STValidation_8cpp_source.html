105bool
106STValidation::isValid() const noexcept
107{
108 if (!valid_)
109 {
110 XRPL_ASSERT(
111 publicKeyType(getSignerPublic()) == KeyType::secp256k1,
112 "ripple::STValidation::isValid : valid key type");
113
114 valid_ = verifyDigest(
115 getSignerPublic(),
116 getSigningHash(),
117 makeSlice(getFieldVL(sfSignature)),
118 getFlags() & vfFullyCanonicalSig);
119 }
120
121 return valid_.value();
122}
