86 bool
87 STValidation::isValid() const
88 {
89  try
90  {
91  if (publicKeyType(getSignerPublic()) != KeyType::secp256k1)
92  return false;
93 
94  return verifyDigest(
95  getSignerPublic(),
96  getSigningHash(),
97  makeSlice(getFieldVL(sfSignature)),
98  getFlags() & vfFullyCanonicalSig);
99  }
100  catch (std::exception const&)
101  {
102  JLOG(debugLog().error()) << "Exception validating validation";
103  return false;
104  }
105 }
