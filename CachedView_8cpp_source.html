33std::shared_ptr<SLE const>
34CachedViewImpl::read(Keylet const& k) const
35{
36 static CountedObjects::Counter hits{"CachedView::hit"};
37 static CountedObjects::Counter hitsexpired{"CachedView::hitExpired"};
38 static CountedObjects::Counter misses{"CachedView::miss"};
39 bool cacheHit = false;
40 bool baseRead = false;
41
42 auto const digest = [&]() -> std::optional<uint256> {
43 {
44 std::lock_guard lock(mutex_);
45 auto const iter = map_.find(k.key);
46 if (iter != map_.end())
47 {
48 cacheHit = true;
49 return iter->second;
50 }
51 }
52 return base_.digest(k.key);
53 }();
54 if (!digest)
55 return nullptr;
56 auto sle = cache_.fetch(*digest, [&]() {
57 baseRead = true;
58 return base_.read(k);
59 });
60 if (cacheHit && baseRead)
61 hitsexpired.increment();
62 else if (cacheHit)
63 hits.increment();
64 else
65 misses.increment();
66
67 if (!cacheHit)
68 {
69 // Avoid acquiring this lock unless necessary. It is only necessary if
70 // the key was not found in the map_. The lock is needed to add the key
71 // and digest.
72 std::lock_guard lock(mutex_);
73 map_.emplace(k.key, *digest);
74 }
75 if (!sle || !k.check(*sle))
76 return nullptr;
77 return sle;
78}
