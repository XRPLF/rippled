46// The preamble encapsulates the DER identifier and

92inline Preamble
93parsePreamble(Slice& s, std::error_code& ec)
94{
95 Preamble p;
96
97 if (s.size() < 2)
98 {
99 ec = error::short_preamble;
100 return p;
101 }
102
103 p.type = s[0] & 0xE0;
104 p.tag = s[0] & 0x1F;
105
106 s += 1;
107
108 if (p.tag == 0x1F)
109 { // Long tag form, which we do not support:
110 ec = error::long_tag;
111 return p;
112 }
113
114 p.length = s[0];
115 s += 1;
116
117 if (p.length & 0x80)
118 { // Long form length:
119 std::size_t const cnt = p.length & 0x7F;
120
121 if (cnt == 0)
122 {
123 ec = error::malformed_encoding;
124 return p;
125 }
126
127 if (cnt > sizeof(std::size_t))
128 {
129 ec = error::large_size;
130 return p;
131 }
132
133 if (cnt > s.size())
134 {
135 ec = error::short_preamble;
136 return p;
137 }
138
139 p.length = 0;
140
141 for (std::size_t i = 0; i != cnt; ++i)
142 p.length = (p.length << 8) + s[i];
143
144 s += cnt;
145
146 if (p.length == 0)
147 {
148 ec = error::malformed_encoding;
149 return p;
150 }
151 }
152
153 return p;
154}

156inline Buffer
157parseOctetString(Slice& s, std::uint32_t count, std::error_code& ec)
158{
159 if (count > s.size())
160 {
161 ec = error::buffer_underfull;
162 return {};
163 }
164
165 if (count > 65535)
166 {
167 ec = error::large_size;
168 return {};
169 }
170
171 Buffer b(s.data(), count);
172 s += count;
173 return b;
174}

176template <class Integer>
177Integer
178parseInteger(Slice& s, std::size_t count, std::error_code& ec)
179{
180 Integer v{0};
181
182 if (s.empty())
183 {
184 // can never have zero sized integers
185 ec = error::malformed_encoding;
186 return v;
187 }
188
189 if (count > s.size())
190 {
191 ec = error::buffer_underfull;
192 return v;
193 }
194
195 const bool isSigned = std::numeric_limits<Integer>::is_signed;
196 // unsigned types may have a leading zero octet
197 const size_t maxLength = isSigned ? sizeof(Integer) : sizeof(Integer) + 1;
198 if (count > maxLength)
199 {
200 ec = error::large_size;
201 return v;
202 }
203
204 if (!isSigned && (s[0] & (1 << 7)))
205 {
206 // trying to decode a negative number into a positive value
207 ec = error::malformed_encoding;
208 return v;
209 }
210
211 if (!isSigned && count == sizeof(Integer) + 1 && s[0])
212 {
213 // since integers are coded as two's complement, the first byte may
214 // be zero for unsigned reps
215 ec = error::malformed_encoding;
216 return v;
217 }
218
219 v = 0;
220 for (size_t i = 0; i < count; ++i)
221 v = (v << 8) | (s[i] & 0xff);
222
223 if (isSigned && (s[0] & (1 << 7)))
224 {
225 for (int i = count; i < sizeof(Integer); ++i)
226 v |= (Integer(0xff) << (8 * i));
227 }
228 s += count;
229 return v;
230}
