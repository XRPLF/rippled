134class ConsensusTimer

136 using time_point = std::chrono::steady_clock::time_point;
137 time_point start_;
138 std::chrono::milliseconds dur_;
139
140public:
141 std::chrono::milliseconds
142 read() const
143 {
144 return dur_;
145 }
146
147 void
148 tick(std::chrono::milliseconds fixed)
149 {
150 dur_ += fixed;
151 }
152
153 void
154 reset(time_point tp)
155 {
156 start_ = tp;
157 dur_ = std::chrono::milliseconds{0};
158 }
159

161 tick(time_point tp)
162 {
163 using namespace std::chrono;
164 dur_ = duration_cast<milliseconds>(tp - start_);

201struct ConsensusResult
202{
203 using Ledger_t = typename Traits::Ledger_t;
204 using TxSet_t = typename Traits::TxSet_t;
205 using NodeID_t = typename Traits::NodeID_t;
206
207 using Tx_t = typename TxSet_t::Tx;
208 using Proposal_t = ConsensusProposal<
209 NodeID_t,
210 typename Ledger_t::ID,
211 typename TxSet_t::ID>;
212 using Dispute_t = DisputedTx<Tx_t, NodeID_t>;
213
214 ConsensusResult(TxSet_t&& s, Proposal_t&& p)
215 : txns{std::move(s)}, position{std::move(p)}
216 {
217 XRPL_ASSERT(
218 txns.id() == position.position(),
219 "ripple::ConsensusResult : valid inputs");
220 }
221
223 TxSet_t txns;
224
226 Proposal_t position;
227
229 hash_map<typename Tx_t::ID, Dispute_t> disputes;
230
231 // Set of TxSet ids we have already compared/created disputes
232 hash_set<typename TxSet_t::ID> compares;
233
234 // Measures the duration of the establish phase for this consensus round
235 ConsensusTimer roundTime;
236
237 // Indicates state in which consensus ended. Once in the accept phase
238 // will be either Yes or MovedOn
239 ConsensusState state = ConsensusState::No;
240
241 // The number of peers proposing during the round
242 std::size_t proposers = 0;
243};
