299 {
300 using namespace std::chrono_literals;

302 if (when == 0ns)
303 what();

384 // Attempt to acquire the Ledger associated with the given ID
385 Ledger const*
386 acquireLedger(Ledger::ID const& ledgerID)
387 {
388 if (auto it = ledgers.find(ledgerID); it != ledgers.end())
389 {
390 return &(it->second);
391 }
392
393 // No peers
394 if (net.links(this).empty())
395 return nullptr;
396
397 // Don't retry if we already are acquiring it and haven't timed out
398 auto aIt = acquiringLedgers.find(ledgerID);
399 if (aIt != acquiringLedgers.end())
400 {
401 if (scheduler.now() < aIt->second)
402 return nullptr;
403 }
404
405 using namespace std::chrono_literals;
406 SimDuration minDuration{10s};
407 for (auto const link : net.links(this))
408 {
409 minDuration = std::min(minDuration, link.data.delay);
410
411 // Send a messsage to neighbors to find the ledger
412 net.send(
413 this, link.target, [to = link.target, from = this, ledgerID]() {
414 if (auto it = to->ledgers.find(ledgerID);
415 it != to->ledgers.end())
416 {
417 // if the ledger is found, send it back to the original
418 // requesting peer where it is added to the available
419 // ledgers
420 to->net.send(to, from, [from, ledger = it->second]() {
421 from->acquiringLedgers.erase(ledger.id());
422 from->ledgers.emplace(ledger.id(), ledger);
423 });
424 }
425 });
426 }
427 acquiringLedgers[ledgerID] = scheduler.now() + 2 * minDuration;
428 return nullptr;
429 }
430
431 // Attempt to acquire the TxSet associated with the given ID
432 TxSet const*
433 acquireTxSet(TxSet::ID const& setId)
434 {
435 if (auto it = txSets.find(setId); it != txSets.end())
436 {
437 return &(it->second);
438 }
439
440 // No peers
441 if (net.links(this).empty())
442 return nullptr;
443
444 // Don't retry if we already are acquiring it and haven't timed out
445 auto aIt = acquiringTxSets.find(setId);
446 if (aIt != acquiringTxSets.end())
447 {
448 if (scheduler.now() < aIt->second)
449 return nullptr;
450 }
451
452 using namespace std::chrono_literals;
453 SimDuration minDuration{10s};
454 for (auto const link : net.links(this))
455 {
456 minDuration = std::min(minDuration, link.data.delay);
457 // Send a message to neighbors to find the tx set
458 net.send(
459 this, link.target, [to = link.target, from = this, setId]() {
460 if (auto it = to->txSets.find(setId);
461 it != to->txSets.end())
462 {
463 // If the txSet is found, send it back to the original
464 // requesting peer, where it is handled like a TxSet
465 // that was broadcast over the network
466 to->net.send(to, from, [from, txSet = it->second]() {
467 from->acquiringTxSets.erase(txSet.id());
468 from->handle(txSet);
469 });
470 }
471 });
472 }
473 acquiringTxSets[setId] = scheduler.now() + 2 * minDuration;
474 return nullptr;
475 }

532 void
533 onAccept(
534 Result const& result,
535 Ledger const& prevLedger,
536 NetClock::duration const& closeResolution,
537 ConsensusCloseTimes const& rawCloseTimes,
538 ConsensusMode const& mode,
539 Json::Value&& consensusJson)
540 {
541 schedule(delays.ledgerAccept, [=, this]() {
542 const bool proposing = mode == ConsensusMode::proposing;
543 const bool consensusFail = result.state == ConsensusState::MovedOn;
544
545 TxSet const acceptedTxs = injectTxs(prevLedger, result.txns);
546 Ledger const newLedger = oracle.accept(
547 prevLedger,
548 acceptedTxs.txs(),
549 closeResolution,
550 result.position.closeTime());
551 ledgers[newLedger.id()] = newLedger;
552
553 issue(AcceptLedger{newLedger, lastClosedLedger});
554 prevProposers = result.proposers;
555 prevRoundTime = result.roundTime.read();
556 lastClosedLedger = newLedger;
557
558 auto const it = std::remove_if(
559 openTxs.begin(), openTxs.end(), [&](Tx const& tx) {
560 return acceptedTxs.exists(tx.id());
561 });
562 openTxs.erase(it, openTxs.end());
563
564 // Only send validation if the new ledger is compatible with our
565 // fully validated ledger
566 bool const isCompatible =
567 newLedger.isAncestor(fullyValidatedLedger);
568
569 // Can only send one validated ledger per seq
570 if (runAsValidator && isCompatible && !consensusFail &&
571 validations.canValidateSeq(newLedger.seq()))
572 {
573 bool isFull = proposing;
574
575 Validation v{
576 newLedger.id(),
577 newLedger.seq(),
578 now(),
579 now(),
580 key,
581 id,
582 isFull};
583 // share the new validation; it is trusted by the receiver
584 share(v);
585 // we trust ourselves
586 addTrustedValidation(v);
587 }
588
589 checkFullyValidated(newLedger);
590
591 // kick off the next round...
592 // in the actual implementation, this passes back through
593 // network ops
594 ++completedLedgers;
595 // startRound sets the LCL state, so we need to call it once after
596 // the last requested round completes
597 if (completedLedgers <= targetLedgers)
598 {
599 startRound();
600 }
601 });
602 }

672 bool
673 addTrustedValidation(Validation v)
674 {
675 v.setTrusted();
676 v.setSeen(now());
677 ValStatus const res = validations.add(v.nodeID(), v);
678
679 if (res == ValStatus::stale)
680 return false;
681
682 // Acquire will try to get from network if not already local
683 if (Ledger const* lgr = acquireLedger(v.ledgerID()))
684 checkFullyValidated(*lgr);
685 return true;
686 }
687
689 void
690 checkFullyValidated(Ledger const& ledger)
691 {
692 // Only consider ledgers newer than our last fully validated ledger
693 if (ledger.seq() <= fullyValidatedLedger.seq())
694 return;
695
696 std::size_t const count = validations.numTrustedForLedger(ledger.id());
697 std::size_t const numTrustedPeers = trustGraph.graph().outDegree(this);
698 quorum = static_cast<std::size_t>(std::ceil(numTrustedPeers * 0.8));
699 if (count >= quorum && ledger.isAncestor(fullyValidatedLedger))
700 {
701 issue(FullyValidateLedger{ledger, fullyValidatedLedger});
702 fullyValidatedLedger = ledger;
703 }
704 }

706 //-------------------------------------------------------------------------
707 // Peer messaging members
708
709 // Basic Sequence number router
710 // A message that will be flooded across the network is tagged with a
711 // sequence number by the origin node in a BroadcastMesg. Receivers will
712 // ignore a message as stale if they've already processed a newer sequence
713 // number, or will process and potentially relay the message along.
714 //
715 // The various bool handle(MessageType) members do the actual processing
716 // and should return true if the message should continue to be sent to
717 // peers.
718 //
719 // WARN: This assumes messages are received and processed in the order they
720 // are sent, so that a peer receives a message with seq 1 from node 0
721 // before seq 2 from node 0, etc.
722 // TODO: Break this out into a class and identify type interface to allow
723 // alternate routing strategies
724 template <class M>
725 struct BroadcastMesg
726 {
727 M mesg;
728 std::size_t seq;
729 PeerID origin;
730 };
731
732 struct Router
733 {
734 std::size_t nextSeq = 1;
735 bc::flat_map<PeerID, std::size_t> lastObservedSeq;
736 };
737
738 Router router;
739
740 // Send a broadcast message to all peers
741 template <class M>
742 void
743 send(BroadcastMesg<M> const& bm, PeerID from)
744 {
745 for (auto const link : net.links(this))
746 {
747 if (link.target->id != from && link.target->id != bm.origin)
748 {
749 // cheat and don't bother sending if we know it has already been
750 // used on the other end
751 if (link.target->router.lastObservedSeq[bm.origin] < bm.seq)
752 {
753 issue(Relay<M>{link.target->id, bm.mesg});
754 net.send(
755 this,
756 link.target,
757 [to = link.target, bm, id = this->id] {
758 to->receive(bm, id);
759 });
760 }
761 }
762 }
763 }
764
765 // Receive a shared message, process it and consider continuing to relay it
766 template <class M>
767 void
768 receive(BroadcastMesg<M> const& bm, PeerID from)
769 {
770 issue(Receive<M>{from, bm.mesg});
771 if (router.lastObservedSeq[bm.origin] < bm.seq)
772 {
773 router.lastObservedSeq[bm.origin] = bm.seq;
774 schedule(delays.onReceive(bm.mesg), [this, bm, from] {
775 if (handle(bm.mesg))
776 send(bm, from);
777 });
778 }
779 }
780
781 // Type specific receive handlers, return true if the message should
782 // continue to be broadcast to peers
783 bool
784 handle(Proposal const& p)
785 {
786 // Only relay untrusted proposals on the same ledger
787 if (!trusts(p.nodeID()))
788 return p.prevLedger() == lastClosedLedger.id();
789
790 // TODO: This always suppresses relay of peer positions already seen
791 // Should it allow forwarding if for a recent ledger ?
792 auto& dest = peerPositions[p.prevLedger()];
793 if (std::find(dest.begin(), dest.end(), p) != dest.end())
794 return false;
795
796 dest.push_back(p);
797
798 // Rely on consensus to decide whether to relay
799 return consensus.peerProposal(now(), Position{p});
800 }

802 bool
803 handle(TxSet const& txs)
804 {
805 bool const inserted =
806 txSets.insert(std::make_pair(txs.id(), txs)).second;
807 if (inserted)
808 consensus.gotTxSet(now(), txs);
809 // relay only if new
810 return inserted;
811 }

813 bool
814 handle(Tx const& tx)
815 {
816 // Ignore and suppress relay of transactions already in last ledger
817 TxSetType const& lastClosedTxs = lastClosedLedger.txs();
818 if (lastClosedTxs.find(tx) != lastClosedTxs.end())
819 return false;
820
821 // only relay if it was new to our open ledger
822 return openTxs.insert(tx).second;
823 }

825 bool
826 handle(Validation const& v)
827 {
828 // TODO: This is not relaying untrusted validations
829 if (!trusts(v.nodeID()))
830 return false;
831
832 // Will only relay if current
833 return addTrustedValidation(v);

897 // Called to begin the next round
898 void
899 startRound()
900 {
901 // Between rounds, we take the majority ledger
902 // In the future, consider taking peer dominant ledger if no validations
903 // yet
904 Ledger::ID bestLCL =
905 validations.getPreferred(lastClosedLedger, earliestAllowedSeq());
906 if (bestLCL == Ledger::ID{0})
907 bestLCL = lastClosedLedger.id();
908
909 issue(StartRound{bestLCL, lastClosedLedger});
910
911 // Not yet modeling dynamic UNL.
912 hash_set<PeerID> nowUntrusted;
913 consensus.startRound(
914 now(), bestLCL, lastClosedLedger, nowUntrusted, runAsValidator);
915 }

928 NetClock::time_point
929 now() const
930 {
931 // We don't care about the actual epochs, but do want the
932 // generated NetClock time to be well past its epoch to ensure
933 // any subtractions of two NetClock::time_point in the consensus
934 // code are positive. (e.g. proposeFRESHNESS)
935 using namespace std::chrono;
936 using namespace std::chrono_literals;
937 return NetClock::time_point(duration_cast<NetClock::duration>(
938 scheduler.now().time_since_epoch() + 86400s + clockSkew));
939 }
