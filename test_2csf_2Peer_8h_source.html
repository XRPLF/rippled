405  // Attempt to acquire the Ledger associated with the given ID
406  Ledger const*
407  acquireLedger(Ledger::ID const& ledgerID)
408  {
409  if (auto it = ledgers.find(ledgerID); it != ledgers.end())
410  {
411  return &(it->second);
412  }
413 
414  // No peers
415  if (net.links(this).empty())
416  return nullptr;
417 
418  // Don't retry if we already are acquiring it and haven't timed out
419  auto aIt = acquiringLedgers.find(ledgerID);
420  if (aIt != acquiringLedgers.end())
421  {
422  if (scheduler.now() < aIt->second)
423  return nullptr;
424  }
425 
426  using namespace std::chrono_literals;
427  SimDuration minDuration{10s};
428  for (auto const link : net.links(this))
429  {
430  minDuration = std::min(minDuration, link.data.delay);
431 
432  // Send a messsage to neighbors to find the ledger
433  net.send(
434  this, link.target, [to = link.target, from = this, ledgerID]() {
435  if (auto it = to->ledgers.find(ledgerID);
436  it != to->ledgers.end())
437  {
438  // if the ledger is found, send it back to the original
439  // requesting peer where it is added to the available
440  // ledgers
441  to->net.send(to, from, [from, ledger = it->second]() {
442  from->acquiringLedgers.erase(ledger.id());
443  from->ledgers.emplace(ledger.id(), ledger);
444  });
445  }
446  });
447  }
448  acquiringLedgers[ledgerID] = scheduler.now() + 2 * minDuration;
449  return nullptr;
450  }
451 
452  // Attempt to acquire the TxSet associated with the given ID
453  TxSet const*
454  acquireTxSet(TxSet::ID const& setId)
455  {
456  if (auto it = txSets.find(setId); it != txSets.end())
457  {
458  return &(it->second);
459  }
460 
461  // No peers
462  if (net.links(this).empty())
463  return nullptr;
464 
465  // Don't retry if we already are acquiring it and haven't timed out
466  auto aIt = acquiringTxSets.find(setId);
467  if (aIt != acquiringTxSets.end())
468  {
469  if (scheduler.now() < aIt->second)
470  return nullptr;
471  }
472 
473  using namespace std::chrono_literals;
474  SimDuration minDuration{10s};
475  for (auto const link : net.links(this))
476  {
477  minDuration = std::min(minDuration, link.data.delay);
478  // Send a message to neighbors to find the tx set
479  net.send(
480  this, link.target, [to = link.target, from = this, setId]() {
481  if (auto it = to->txSets.find(setId);
482  it != to->txSets.end())
483  {
484  // If the txSet is found, send it back to the original
485  // requesting peer, where it is handled like a TxSet
486  // that was broadcast over the network
487  to->net.send(to, from, [from, txSet = it->second]() {
488  from->acquiringTxSets.erase(txSet.id());
489  from->handle(txSet);
490  });
491  }
492  });
493  }
494  acquiringTxSets[setId] = scheduler.now() + 2 * minDuration;
495  return nullptr;
496  }

565  std::pair<CanonicalTxSet_t, Ledger_t>
566  buildAndValidate(
567  Result const& result,
568  Ledger_t const& prevLedger,
569  NetClock::duration const& closeResolution,
570  ConsensusMode const& mode,
571  Json::Value&& consensusJson)
572  {
573  schedule(delays.ledgerAccept, [=, this]() {
574  const bool proposing = mode == ConsensusMode::proposing;
575  const bool consensusFail = result.state == ConsensusState::MovedOn;
576 
577  TxSet const acceptedTxs = injectTxs(prevLedger, result.txns);
578  Ledger const newLedger = oracle.accept(
579  prevLedger,
580  acceptedTxs.txs(),
581  closeResolution,
582  result.position.closeTime());
583  ledgers[newLedger.id()] = newLedger;
584 
585  issue(AcceptLedger{newLedger, lastClosedLedger});
586  prevProposers = result.proposers;
587  prevRoundTime = result.roundTime.read();
588  lastClosedLedger = newLedger;
589 
590  auto const it = std::remove_if(
591  openTxs.begin(), openTxs.end(), [&](Tx const& tx) {
592  return acceptedTxs.exists(tx.id());
593  });
594  openTxs.erase(it, openTxs.end());
595 
596  // Only send validation if the new ledger is compatible with our
597  // fully validated ledger
598  bool const isCompatible =
599  newLedger.isAncestor(fullyValidatedLedger);
600 
601  // Can only send one validated ledger per seq
602  if (runAsValidator && isCompatible && !consensusFail &&
603  validations.canValidateSeq(newLedger.seq()))
604  {
605  bool isFull = proposing;
606 
607  Validation v{
608  newLedger.id(),
609  newLedger.seq(),
610  now(),
611  now(),
612  key,
613  id,
614  isFull};
615  // share the new validation; it is trusted by the receiver
616  share(v);
617  // we trust ourselves
618  addTrustedValidation(v);
619  }
620 
621  checkFullyValidated(newLedger);
622 
623  // kick off the next round...
624  // in the actual implementation, this passes back through
625  // network ops
626  ++completedLedgers;
627  // startRound sets the LCL state, so we need to call it once after
628  // the last requested round completes
629  if (completedLedgers <= targetLedgers)
630  {
631  startRound();
632  }
633  });
634 
635  return {};
636  }

722  void
723  checkFullyValidated(Ledger const& ledger)
724  {
725  // Only consider ledgers newer than our last fully validated ledger
726  if (ledger.seq() <= fullyValidatedLedger.seq())
727  return;
728 
729  std::size_t const count = validations.numTrustedForLedger(ledger.id());
730  std::size_t const numTrustedPeers = trustGraph.graph().outDegree(this);
731  q = static_cast<std::size_t>(std::ceil(numTrustedPeers * 0.8));
732  if (count >= q && ledger.isAncestor(fullyValidatedLedger))
733  {
734  issue(FullyValidateLedger{ledger, fullyValidatedLedger});
735  fullyValidatedLedger = ledger;
736  }

742  // Basic Sequence number router
743  // A message that will be flooded across the network is tagged with a
744  // sequence number by the origin node in a BroadcastMesg. Receivers will
745  // ignore a message as stale if they've already processed a newer sequence
746  // number, or will process and potentially relay the message along.
747  //
748  // The various bool handle(MessageType) members do the actual processing
749  // and should return true if the message should continue to be sent to
750  // peers.
751  //
752  // WARN: This assumes messages are received and processed in the order they
753  // are sent, so that a peer receives a message with seq 1 from node 0
754  // before seq 2 from node 0, etc.
755  // TODO: Break this out into a class and identify type interface to allow
756  // alternate routing strategies
757  template <class M>
758  struct BroadcastMesg
759  {
760  M mesg;
761  std::size_t seq;
762  PeerID origin;
763  };
764 
765  struct Router
766  {
767  std::size_t nextSeq = 1;
768  bc::flat_map<PeerID, std::size_t> lastObservedSeq;
769  };

771  Router router;

773  // Send a broadcast message to all peers
774  template <class M>
775  void
776  send(BroadcastMesg<M> const& bm, PeerID from)
777  {
778  for (auto const link : net.links(this))
779  {
780  if (link.target->id != from && link.target->id != bm.origin)
781  {
782  // cheat and don't bother sending if we know it has already been
783  // used on the other end
784  if (link.target->router.lastObservedSeq[bm.origin] < bm.seq)
785  {
786  issue(Relay<M>{link.target->id, bm.mesg});
787  net.send(
788  this,
789  link.target,
790  [to = link.target, bm, id = this->id] {
791  to->receive(bm, id);
792  });
793  }
794  }
795  }
796  }

798  // Receive a shared message, process it and consider continuing to relay it
799  template <class M>
800  void
801  receive(BroadcastMesg<M> const& bm, PeerID from)
802  {
803  issue(Receive<M>{from, bm.mesg});
804  if (router.lastObservedSeq[bm.origin] < bm.seq)
805  {
806  router.lastObservedSeq[bm.origin] = bm.seq;
807  schedule(delays.onReceive(bm.mesg), [this, bm, from] {
808  if (handle(bm.mesg))
809  send(bm, from);
810  });
811  }
812  }

814  // Type specific receive handlers, return true if the message should
815  // continue to be broadcast to peers
816  bool
817  handle(Proposal const& p)
818  {
819  // Only relay untrusted proposals on the same ledger
820  if (!trusts(p.nodeID()))
821  return p.prevLedger() == lastClosedLedger.id();
822 
823  // TODO: This always suppresses relay of peer positions already seen
824  // Should it allow forwarding if for a recent ledger ?
825  auto& dest = peerPositions[p.prevLedger()];
826  if (std::find(dest.begin(), dest.end(), p) != dest.end())
827  return false;
828 
829  dest.push_back(p);
830 
831  // Rely on consensus to decide whether to relay
832  return consensus.peerProposal(now(), Position{p});
833  }
