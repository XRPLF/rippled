39template <class T, class G>
40std::vector<T>
41random_weighted_shuffle(std::vector<T> v, std::vector<double> w, G& g)
42{
43 using std::swap;
44
45 for (int i = 0; i < v.size() - 1; ++i)
46 {
47 // pick a random item weighted by w
48 std::discrete_distribution<> dd(w.begin() + i, w.end());
49 auto idx = dd(g);
50 std::swap(v[i], v[idx]);
51 std::swap(w[i], w[idx]);
52 }
53 return v;
54}

64template <class RandomNumberDistribution, class Generator>
65std::vector<typename RandomNumberDistribution::result_type>
66sample(std::size_t size, RandomNumberDistribution dist, Generator& g)
67{
68 std::vector<typename RandomNumberDistribution::result_type> res(size);
69 std::generate(res.begin(), res.end(), [&dist, &g]() { return dist(g); });
70 return res;
71}

80template <class RAIter, class Generator>
81class Selector

97 std::vector<double> const& w,
98 Generator& g)
99 : first_{first}, last_{last}, dd_{w.begin(), w.end()}, g_{g}
100 {
101 using tag = typename std::iterator_traits<RAIter>::iterator_category;
102 static_assert(

151class PowerLawDistribution
152{
153 double xmin_;
154 double a_;
155 double inv_;
156 std::uniform_real_distribution<double> uf_{0, 1};
157
158public:
159 using result_type = double;
160
161 PowerLawDistribution(double xmin, double a) : xmin_{xmin}, a_{a}
162 {
163 inv_ = 1.0 / (1.0 - a_);
164 }
165
166 template <class Generator>
167 inline double
168 operator()(Generator& g)
169 {
170 // use inverse transform of CDF to sample
171 // CDF is P(X <= x): 1 - (x/xmin)^(1-a)
172 return xmin_ * std::pow(1 - uf_(g), inv_);
173 }
174};
