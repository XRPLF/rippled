30NotTEC
31NFTokenAcceptOffer::preflight(PreflightContext const& ctx)
32{
33 if (!ctx.rules.enabled(featureNonFungibleTokensV1))
34 return temDISABLED;
35
36 if (auto const ret = preflight1(ctx); !isTesSuccess(ret))
37 return ret;
38
39 if (ctx.tx.getFlags() & tfNFTokenAcceptOfferMask)
40 return temINVALID_FLAG;
41
42 auto const bo = ctx.tx[~sfNFTokenBuyOffer];
43 auto const so = ctx.tx[~sfNFTokenSellOffer];
44
45 // At least one of these MUST be specified
46 if (!bo && !so)
47 return temMALFORMED;
48
49 // The `BrokerFee` field must not be present in direct mode but may be
50 // present and greater than zero in brokered mode.
51 if (auto const bf = ctx.tx[~sfNFTokenBrokerFee])
52 {
53 if (!bo || !so)
54 return temMALFORMED;
55
56 if (*bf <= beast::zero)
57 return temMALFORMED;
58 }
59
60 return preflight2(ctx);
61}

63TER
64NFTokenAcceptOffer::preclaim(PreclaimContext const& ctx)
65{
66 auto const checkOffer = [&ctx](std::optional<uint256> id)
67 -> std::pair<std::shared_ptr<const SLE>, TER> {
68 if (id)
69 {
70 if (id->isZero())
71 return {nullptr, tecOBJECT_NOT_FOUND};
72
73 auto offerSLE = ctx.view.read(keylet::nftoffer(*id));
74
75 if (!offerSLE)
76 return {nullptr, tecOBJECT_NOT_FOUND};
77
78 if (hasExpired(ctx.view, (*offerSLE)[~sfExpiration]))
79 return {nullptr, tecEXPIRED};
80
81 // The initial implementation had a bug that allowed a negative
82 // amount. The fixNFTokenNegOffer amendment fixes that.
83 if ((*offerSLE)[sfAmount].negative() &&
84 ctx.view.rules().enabled(fixNFTokenNegOffer))
85 return {nullptr, temBAD_OFFER};
86
87 return {std::move(offerSLE), tesSUCCESS};
88 }
89 return {nullptr, tesSUCCESS};
90 };
91
92 auto const [bo, err1] = checkOffer(ctx.tx[~sfNFTokenBuyOffer]);
93 if (!isTesSuccess(err1))
94 return err1;
95 auto const [so, err2] = checkOffer(ctx.tx[~sfNFTokenSellOffer]);
96 if (!isTesSuccess(err2))
97 return err2;
98
99 if (bo && so)
100 {
101 // Brokered mode:
102 // The two offers being brokered must be for the same token:
103 if ((*bo)[sfNFTokenID] != (*so)[sfNFTokenID])
104 return tecNFTOKEN_BUY_SELL_MISMATCH;
105
106 // The two offers being brokered must be for the same asset:
107 if ((*bo)[sfAmount].issue() != (*so)[sfAmount].issue())
108 return tecNFTOKEN_BUY_SELL_MISMATCH;
109
110 // The two offers may not form a loop. A broker may not sell the
111 // token to the current owner of the token.
112 if (ctx.view.rules().enabled(fixNonFungibleTokensV1_2) &&
113 ((*bo)[sfOwner] == (*so)[sfOwner]))
114 return tecCANT_ACCEPT_OWN_NFTOKEN_OFFER;
115
116 // Ensure that the buyer is willing to pay at least as much as the
117 // seller is requesting:
118 if ((*so)[sfAmount] > (*bo)[sfAmount])
119 return tecINSUFFICIENT_PAYMENT;
120
121 // If the buyer specified a destination
122 if (auto const dest = bo->at(~sfDestination))
123 {
124 // Before this fix the destination could be either the seller or
125 // a broker. After, it must be whoever is submitting the tx.
126 if (ctx.view.rules().enabled(fixNonFungibleTokensV1_2))
127 {
128 if (*dest != ctx.tx[sfAccount])
129 return tecNO_PERMISSION;
130 }
131 else if (*dest != so->at(sfOwner) && *dest != ctx.tx[sfAccount])
132 return tecNFTOKEN_BUY_SELL_MISMATCH;
133 }
134
135 // If the seller specified a destination
136 if (auto const dest = so->at(~sfDestination))
137 {
138 // Before this fix the destination could be either the seller or
139 // a broker. After, it must be whoever is submitting the tx.
140 if (ctx.view.rules().enabled(fixNonFungibleTokensV1_2))
141 {
142 if (*dest != ctx.tx[sfAccount])
143 return tecNO_PERMISSION;
144 }
145 else if (*dest != bo->at(sfOwner) && *dest != ctx.tx[sfAccount])
146 return tecNFTOKEN_BUY_SELL_MISMATCH;
147 }
148
149 // The broker can specify an amount that represents their cut; if they
150 // have, ensure that the seller will get at least as much as they want
151 // to get *after* this fee is accounted for (but before the issuer's
152 // cut, if any).
153 if (auto const brokerFee = ctx.tx[~sfNFTokenBrokerFee])
154 {
155 if (brokerFee->issue() != (*bo)[sfAmount].issue())
156 return tecNFTOKEN_BUY_SELL_MISMATCH;
157
158 if (brokerFee >= (*bo)[sfAmount])
159 return tecINSUFFICIENT_PAYMENT;
160
161 if ((*so)[sfAmount] > (*bo)[sfAmount] - *brokerFee)
162 return tecINSUFFICIENT_PAYMENT;
163 }
164 }
165
166 if (bo)
167 {
168 if (((*bo)[sfFlags] & lsfSellNFToken) == lsfSellNFToken)
169 return tecNFTOKEN_OFFER_TYPE_MISMATCH;
170
171 // An account can't accept an offer it placed:
172 if ((*bo)[sfOwner] == ctx.tx[sfAccount])
173 return tecCANT_ACCEPT_OWN_NFTOKEN_OFFER;
174
175 // If not in bridged mode, the account must own the token:
176 if (!so &&
177 !nft::findToken(ctx.view, ctx.tx[sfAccount], (*bo)[sfNFTokenID]))
178 return tecNO_PERMISSION;
179
180 // If not in bridged mode...
181 if (!so)
182 {
183 // If the offer has a Destination field, the acceptor must be the
184 // Destination.
185 if (auto const dest = bo->at(~sfDestination);
186 dest.has_value() && *dest != ctx.tx[sfAccount])
187 return tecNO_PERMISSION;
188 }
189
190 // The account offering to buy must have funds:
191 //
192 // After this amendment, we allow an IOU issuer to buy an NFT with their
193 // own currency
194 auto const needed = bo->at(sfAmount);
195 if (ctx.view.rules().enabled(fixNonFungibleTokensV1_2))
196 {
197 if (accountFunds(
198 ctx.view, (*bo)[sfOwner], needed, fhZERO_IF_FROZEN, ctx.j) <
199 needed)
200 return tecINSUFFICIENT_FUNDS;
201 }
202 else if (
203 accountHolds(
204 ctx.view,
205 (*bo)[sfOwner],
206 needed.getCurrency(),
207 needed.getIssuer(),
208 fhZERO_IF_FROZEN,
209 ctx.j) < needed)
210 return tecINSUFFICIENT_FUNDS;
211 }
212
213 if (so)
214 {
215 if (((*so)[sfFlags] & lsfSellNFToken) != lsfSellNFToken)
216 return tecNFTOKEN_OFFER_TYPE_MISMATCH;
217
218 // An account can't accept an offer it placed:
219 if ((*so)[sfOwner] == ctx.tx[sfAccount])
220 return tecCANT_ACCEPT_OWN_NFTOKEN_OFFER;
221
222 // The seller must own the token.
223 if (!nft::findToken(ctx.view, (*so)[sfOwner], (*so)[sfNFTokenID]))
224 return tecNO_PERMISSION;
225
226 // If not in bridged mode...
227 if (!bo)
228 {
229 // If the offer has a Destination field, the acceptor must be the
230 // Destination.
231 if (auto const dest = so->at(~sfDestination);
232 dest.has_value() && *dest != ctx.tx[sfAccount])
233 return tecNO_PERMISSION;
234 }
235
236 // The account offering to buy must have funds:
237 auto const needed = so->at(sfAmount);
238 if (!ctx.view.rules().enabled(fixNonFungibleTokensV1_2))
239 {
240 if (accountHolds(
241 ctx.view,
242 ctx.tx[sfAccount],
243 needed.getCurrency(),
244 needed.getIssuer(),
245 fhZERO_IF_FROZEN,
246 ctx.j) < needed)
247 return tecINSUFFICIENT_FUNDS;
248 }
249 else if (!bo)
250 {
251 // After this amendment, we allow buyers to buy with their own
252 // issued currency.
253 //
254 // In the case of brokered mode, this check is essentially
255 // redundant, since we have already confirmed that buy offer is >
256 // than the sell offer, and that the buyer can cover the buy
257 // offer.
258 //
259 // We also _must not_ check the tx submitter in brokered
260 // mode, because then we are confirming that the broker can
261 // cover what the buyer will pay, which doesn't make sense, causes
262 // an unnecessary tec, and is also resolved with this amendment.
263 if (accountFunds(
264 ctx.view,
265 ctx.tx[sfAccount],
266 needed,
267 fhZERO_IF_FROZEN,
268 ctx.j) < needed)
269 return tecINSUFFICIENT_FUNDS;
270 }
271 }
272
273 // Fix a bug where the transfer of an NFToken with a transfer fee could
274 // give the NFToken issuer an undesired trust line.
275 if (ctx.view.rules().enabled(fixEnforceNFTokenTrustline))
276 {
277 std::shared_ptr<SLE const> const& offer = bo ? bo : so;
278 if (!offer)
279 // Should be caught in preflight.
280 return tecINTERNAL;
281
282 uint256 const& tokenID = offer->at(sfNFTokenID);
283 STAmount const& amount = offer->at(sfAmount);
284 if (nft::getTransferFee(tokenID) != 0 &&
285 (nft::getFlags(tokenID) & nft::flagCreateTrustLines) == 0 &&
286 !amount.native())
287 {
288 auto const issuer = nft::getIssuer(tokenID);
289 // Issuer doesn't need a trust line to accept their own currency.
290 if (issuer != amount.getIssuer() &&
291 !ctx.view.read(keylet::line(issuer, amount.issue())))
292 return tecNO_LINE;
293 }
294 }
295 return tesSUCCESS;
296}

298TER
299NFTokenAcceptOffer::pay(
300 AccountID const& from,
301 AccountID const& to,
302 STAmount const& amount)
303{
304 // This should never happen, but it's easy and quick to check.
305 if (amount < beast::zero)
306 return tecINTERNAL;
307
308 auto const result = accountSend(view(), from, to, amount, j_);
309
310 // After this amendment, if any payment would cause a non-IOU-issuer to
311 // have a negative balance, or an IOU-issuer to have a positive balance in
312 // their own currency, we know that something went wrong. This was
313 // originally found in the context of IOU transfer fees. Since there are
314 // several payouts in this tx, just confirm that the end state is OK.
315 if (!view().rules().enabled(fixNonFungibleTokensV1_2))
316 return result;
317 if (result != tesSUCCESS)
318 return result;
319 if (accountFunds(view(), from, amount, fhZERO_IF_FROZEN, j_).signum() < 0)
320 return tecINSUFFICIENT_FUNDS;
321 if (accountFunds(view(), to, amount, fhZERO_IF_FROZEN, j_).signum() < 0)
322 return tecINSUFFICIENT_FUNDS;
323 return tesSUCCESS;
324}

326TER
327NFTokenAcceptOffer::transferNFToken(
328 AccountID const& buyer,
329 AccountID const& seller,
330 uint256 const& nftokenID)
331{
332 auto tokenAndPage = nft::findTokenAndPage(view(), seller, nftokenID);
333
334 if (!tokenAndPage)
335 return tecINTERNAL;
336
337 if (auto const ret = nft::removeToken(
338 view(), seller, nftokenID, std::move(tokenAndPage->page));
339 !isTesSuccess(ret))
340 return ret;
341
342 auto const sleBuyer = view().read(keylet::account(buyer));
343 if (!sleBuyer)
344 return tecINTERNAL;
345
346 std::uint32_t const buyerOwnerCountBefore =
347 sleBuyer->getFieldU32(sfOwnerCount);
348
349 auto const insertRet =
350 nft::insertToken(view(), buyer, std::move(tokenAndPage->token));
351
352 // if fixNFTokenReserve is enabled, check if the buyer has sufficient
353 // reserve to own a new object, if their OwnerCount changed.
354 //
355 // There was an issue where the buyer accepts a sell offer, the ledger
356 // didn't check if the buyer has enough reserve, meaning that buyer can get
357 // NFTs free of reserve.
358 if (view().rules().enabled(fixNFTokenReserve))
359 {
360 // To check if there is sufficient reserve, we cannot use mPriorBalance
361 // because NFT is sold for a price. So we must use the balance after
362 // the deduction of the potential offer price. A small caveat here is
363 // that the balance has already deducted the transaction fee, meaning
364 // that the reserve requirement is a few drops higher.
365 auto const buyerBalance = sleBuyer->getFieldAmount(sfBalance);
366
367 auto const buyerOwnerCountAfter = sleBuyer->getFieldU32(sfOwnerCount);
368 if (buyerOwnerCountAfter > buyerOwnerCountBefore)
369 {
370 if (auto const reserve =
371 view().fees().accountReserve(buyerOwnerCountAfter);
372 buyerBalance < reserve)
373 return tecINSUFFICIENT_RESERVE;
374 }
375 }
376
377 return insertRet;
378}

380TER
381NFTokenAcceptOffer::acceptOffer(std::shared_ptr<SLE> const& offer)
382{
383 bool const isSell = offer->isFlag(lsfSellNFToken);
384 AccountID const owner = (*offer)[sfOwner];
385 AccountID const& seller = isSell ? owner : account_;
386 AccountID const& buyer = isSell ? account_ : owner;
387
388 auto const nftokenID = (*offer)[sfNFTokenID];
389
390 if (auto amount = offer->getFieldAmount(sfAmount); amount != beast::zero)
391 {
392 // Calculate the issuer's cut from this sale, if any:
393 if (auto const fee = nft::getTransferFee(nftokenID); fee != 0)
394 {
395 auto const cut = multiply(amount, nft::transferFeeAsRate(fee));
396
397 if (auto const issuer = nft::getIssuer(nftokenID);
398 cut != beast::zero && seller != issuer && buyer != issuer)
399 {
400 if (auto const r = pay(buyer, issuer, cut); !isTesSuccess(r))
401 return r;
402 amount -= cut;
403 }
404 }
405
406 // Send the remaining funds to the seller of the NFT
407 if (auto const r = pay(buyer, seller, amount); !isTesSuccess(r))
408 return r;
409 }
410
411 // Now transfer the NFT:
412 return transferNFToken(buyer, seller, nftokenID);
413}

415TER
416NFTokenAcceptOffer::doApply()
417{
418 auto const loadToken = [this](std::optional<uint256> const& id) {
419 std::shared_ptr<SLE> sle;
420 if (id)
421 sle = view().peek(keylet::nftoffer(*id));
422 return sle;
423 };
424
425 auto bo = loadToken(ctx_.tx[~sfNFTokenBuyOffer]);
426 auto so = loadToken(ctx_.tx[~sfNFTokenSellOffer]);
427
428 if (bo && !nft::deleteTokenOffer(view(), bo))
429 {
430 JLOG(j_.fatal()) << "Unable to delete buy offer '"
431 << to_string(bo->key()) << "': ignoring";
432 return tecINTERNAL;
433 }
434
435 if (so && !nft::deleteTokenOffer(view(), so))
436 {
437 JLOG(j_.fatal()) << "Unable to delete sell offer '"
438 << to_string(so->key()) << "': ignoring";
439 return tecINTERNAL;
440 }
441
442 // Bridging two different offers
443 if (bo && so)
444 {
445 AccountID const buyer = (*bo)[sfOwner];
446 AccountID const seller = (*so)[sfOwner];
447
448 auto const nftokenID = (*so)[sfNFTokenID];
449
450 // The amount is what the buyer of the NFT pays:
451 STAmount amount = (*bo)[sfAmount];
452
453 // Three different folks may be paid. The order of operations is
454 // important.
455 //
456 // o The broker is paid the cut they requested.
457 // o The issuer's cut is calculated from what remains after the
458 // broker is paid. The issuer can take up to 50% of the remainder.
459 // o Finally, the seller gets whatever is left.
460 //
461 // It is important that the issuer's cut be calculated after the
462 // broker's portion is already removed. Calculating the issuer's
463 // cut before the broker's cut is removed can result in more money
464 // being paid out than the seller authorized. That would be bad!
465
466 // Send the broker the amount they requested.
467 if (auto const cut = ctx_.tx[~sfNFTokenBrokerFee];
468 cut && cut.value() != beast::zero)
469 {
470 if (auto const r = pay(buyer, account_, cut.value());
471 !isTesSuccess(r))
472 return r;
473
474 amount -= cut.value();
475 }
476
477 // Calculate the issuer's cut, if any.
478 if (auto const fee = nft::getTransferFee(nftokenID);
479 amount != beast::zero && fee != 0)
480 {
481 auto cut = multiply(amount, nft::transferFeeAsRate(fee));
482
483 if (auto const issuer = nft::getIssuer(nftokenID);
484 seller != issuer && buyer != issuer)
485 {
486 if (auto const r = pay(buyer, issuer, cut); !isTesSuccess(r))
487 return r;
488
489 amount -= cut;
490 }
491 }
492
493 // And send whatever remains to the seller.
494 if (amount > beast::zero)
495 {
496 if (auto const r = pay(buyer, seller, amount); !isTesSuccess(r))
497 return r;
498 }
499
500 // Now transfer the NFT:
501 return transferNFToken(buyer, seller, nftokenID);
502 }
503
504 if (bo)
505 return acceptOffer(bo);
506
507 if (so)
508 return acceptOffer(so);
509
510 return tecINTERNAL;
511}
