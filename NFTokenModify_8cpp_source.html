30NotTEC
31NFTokenModify::preflight(PreflightContext const& ctx)
32{
33 if (!ctx.rules.enabled(featureNonFungibleTokensV1_1) ||
34 !ctx.rules.enabled(featureDynamicNFT))
35 return temDISABLED;
36
37 if (NotTEC const ret = preflight1(ctx); !isTesSuccess(ret))
38 return ret;
39
40 if (ctx.tx.getFlags() & tfUniversalMask)
41 return temINVALID_FLAG;
42
43 if (auto owner = ctx.tx[~sfOwner]; owner == ctx.tx[sfAccount])
44 return temMALFORMED;
45
46 if (auto uri = ctx.tx[~sfURI])
47 {
48 if (uri->length() == 0 || uri->length() > maxTokenURILength)
49 return temMALFORMED;
50 }
51
52 return preflight2(ctx);
53}

56NFTokenModify::preclaim(PreclaimContext const& ctx)
57{
58 AccountID const account = ctx.tx[sfAccount];
59 AccountID const owner =
60 ctx.tx[ctx.tx.isFieldPresent(sfOwner) ? sfOwner : sfAccount];
61
62 if (!nft::findToken(ctx.view, owner, ctx.tx[sfNFTokenID]))
63 return tecNO_ENTRY;
64
65 // Check if the NFT is mutable
66 if (!(nft::getFlags(ctx.tx[sfNFTokenID]) & nft::flagMutable))
67 return tecNO_PERMISSION;
68
69 // Verify permissions for the issuer
70 if (AccountID const issuer = nft::getIssuer(ctx.tx[sfNFTokenID]);
71 issuer != account)
72 {
73 auto const sle = ctx.view.read(keylet::account(issuer));
74 if (!sle)
75 return tecINTERNAL; // LCOV_EXCL_LINE
76 if (auto const minter = (*sle)[~sfNFTokenMinter]; minter != account)
77 return tecNO_PERMISSION;
78 }
79
80 return tesSUCCESS;
81}
