41  void fetch (Results& results, beast::Journal journal) override
42  {
43  results.addresses.resize (0);

45  for (int i = 0; i < m_strings.size (); ++i)
46  {
47  beast::IP::Endpoint ep (beast::IP::Endpoint::from_string (m_strings [i]));
48  if (is_unspecified (ep))

50  if (! is_unspecified (ep))

52  }
