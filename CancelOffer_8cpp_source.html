28 CancelOffer::preflight(PreflightContext const& ctx)
29 {
30  auto const ret = preflight1(ctx);
31  if (!isTesSuccess(ret))
32  return ret;
33 
34  auto const uTxFlags = ctx.tx.getFlags();
35 
36  if (uTxFlags & tfUniversalMask)
37  {
38  JLOG(ctx.j.trace()) << "Malformed transaction: "
39  << "Invalid flags set.";

41  }
42 
43  auto const seq = ctx.tx.getFieldU32(sfOfferSequence);
44  if (!seq)
45  {
46  JLOG(ctx.j.trace()) << "CancelOffer::preflight: missing sequence";
47  return temBAD_SEQUENCE;
48  }
49 
50  return preflight2(ctx);
51 }

56 CancelOffer::preclaim(PreclaimContext const& ctx)
57 {
58  auto const id = ctx.tx[sfAccount];
59  auto const offerSequence = ctx.tx[sfOfferSequence];
60 
61  auto const sle = ctx.view.read(keylet::account(id));
62  if (!sle)
63  return terNO_ACCOUNT;
64 
65  if ((*sle)[sfSequence] <= offerSequence)
66  {
67  JLOG(ctx.j.trace()) << "Malformed transaction: "
68  << "Sequence " << offerSequence << " is invalid.";
69  return temBAD_SEQUENCE;
70  }
71 
72  return tesSUCCESS;
73 }

78 CancelOffer::doApply()
79 {
80  auto const offerSequence = ctx_.tx[sfOfferSequence];
81 
82  auto const sle = view().read(keylet::account(account_));
83  if (!sle)
84  return tefINTERNAL;
85 
86  if (auto sleOffer = view().peek(keylet::offer(account_, offerSequence)))
87  {
88  JLOG(j_.debug()) << "Trying to cancel offer #" << offerSequence;
89  return offerDelete(view(), sleOffer, ctx_.app.journal("View"));
90  }
91 
92  JLOG(j_.debug()) << "Offer #" << offerSequence << " can't be found.";
93  return tesSUCCESS;
94 }
