69template <class Buffers>
70ZeroCopyInputStream<Buffers>::ZeroCopyInputStream(Buffers const& buffers)
71 : last_(buffers.end())
72 , first_(buffers.begin())
73 , pos_((first_ != last_) ? *first_ : const_buffer(nullptr, 0))
74{
75}

77template <class Buffers>
78bool
79ZeroCopyInputStream<Buffers>::Next(const void** data, int* size)
80{
81 *data = boost::asio::buffer_cast<void const*>(pos_);
82 *size = boost::asio::buffer_size(pos_);
83 if (first_ == last_)
84 return false;
85 count_ += *size;
86 pos_ = (++first_ != last_) ? *first_ : const_buffer(nullptr, 0);
87 return true;
88}
89
90template <class Buffers>
91void
92ZeroCopyInputStream<Buffers>::BackUp(int count)
93{
94 --first_;
95 pos_ = *first_ + (boost::asio::buffer_size(*first_) - count);
96 count_ -= count;
97}
98
99template <class Buffers>
100bool
101ZeroCopyInputStream<Buffers>::Skip(int count)
102{
103 if (first_ == last_)
104 return false;
105 while (count > 0)
106 {
107 auto const size = boost::asio::buffer_size(pos_);
108 if (count < size)
109 {
110 pos_ = pos_ + count;
111 count_ += count;
112 return true;
113 }
114 count_ += size;
115 if (++first_ == last_)
116 return false;
117 count -= size;
118 pos_ = *first_;
119 }
120 return true;
121}

175template <class Streambuf>
176ZeroCopyOutputStream<Streambuf>::~ZeroCopyOutputStream()
177{
178 if (commit_ != 0)
179 streambuf_.commit(commit_);
180}

182template <class Streambuf>
183bool
184ZeroCopyOutputStream<Streambuf>::Next(void** data, int* size)
185{
186 if (commit_ != 0)
187 {
188 streambuf_.commit(commit_);
189 count_ += commit_;
190 }
191
192 if (pos_ == buffers_.end())
193 {
194 buffers_ = streambuf_.prepare(blockSize_);
195 pos_ = buffers_.begin();
196 }
197
198 *data = boost::asio::buffer_cast<void*>(*pos_);
199 *size = boost::asio::buffer_size(*pos_);
200 commit_ = *size;
201 ++pos_;
202 return true;
203}

205template <class Streambuf>
206void
207ZeroCopyOutputStream<Streambuf>::BackUp(int count)
208{
209 XRPL_ASSERT(
210 count <= commit_, "ripple::ZeroCopyOutputStream::BackUp : valid input");
211 auto const n = commit_ - count;
212 streambuf_.commit(n);
213 count_ += n;
214 commit_ = 0;
215}
