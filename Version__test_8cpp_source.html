82  void testGetAPIVersionNumber()
83  {
84  testcase ("test getAPIVersionNumber function");
85 
86  unsigned int versionIfUnspecified =
87  RPC::APIVersionIfUnspecified < RPC::ApiMinimumSupportedVersion ?
88  RPC::APIInvalidVersion : RPC::APIVersionIfUnspecified;
89 
90  Json::Value j_array = Json::Value (Json::arrayValue);
91  Json::Value j_null = Json::Value (Json::nullValue);
92  BEAST_EXPECT(RPC::getAPIVersionNumber(j_array) == versionIfUnspecified);
93  BEAST_EXPECT(RPC::getAPIVersionNumber(j_null) == versionIfUnspecified);
94 
95  Json::Value j_object = Json::Value (Json::objectValue);
96  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == versionIfUnspecified);
97  j_object[jss::api_version] = RPC::APIVersionIfUnspecified;
98  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == versionIfUnspecified);
99 
100  j_object[jss::api_version] = RPC::ApiMinimumSupportedVersion;
101  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == RPC::ApiMinimumSupportedVersion);
102  j_object[jss::api_version] = RPC::ApiMaximumSupportedVersion;
103  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == RPC::ApiMaximumSupportedVersion);
104 
105  j_object[jss::api_version] = RPC::ApiMinimumSupportedVersion - 1;
106  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == RPC::APIInvalidVersion);
107  j_object[jss::api_version] = RPC::ApiMaximumSupportedVersion + 1;
108  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == RPC::APIInvalidVersion);
109 
110  j_object[jss::api_version] = RPC::APIInvalidVersion;
111  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == RPC::APIInvalidVersion);
112  j_object[jss::api_version] = "a";
113  BEAST_EXPECT(RPC::getAPIVersionNumber(j_object) == RPC::APIInvalidVersion);
114  }
