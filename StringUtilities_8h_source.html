39 inline static std::string sqlEscape (Blob const& vecSrc)
40 {
41  size_t size = vecSrc.size ();
42 
43  if (size == 0)
44  return "X''";
45 
46  std::string j (size * 2 + 3, 0);
47 
48  unsigned char* oPtr = reinterpret_cast<unsigned char*> (&*j.begin ());
49  const unsigned char* iPtr = &vecSrc[0];
50 
51  *oPtr++ = 'X';
52  *oPtr++ = '\'';
53 
54  for (int i = size; i != 0; --i)
55  {
56  unsigned char c = *iPtr++;
57  *oPtr++ = charHex (c >> 4);
58  *oPtr++ = charHex (c & 15);
59  }
60 
61  *oPtr++ = '\'';
62  return j;
63 }

67 template <class Iterator>
68 boost::optional<Blob>
69 strUnHex(std::size_t strSize, Iterator begin, Iterator end)
70 {
71  Blob out;
72 
73  out.reserve((strSize + 1) / 2);
74 
75  auto iter = begin;
76 
77  if (strSize & 1)
78  {
79  int c = charUnHex(*iter);
80 
81  if (c < 0)
82  return {};
83 
84  out.push_back(c);
85  ++iter;
86  }
87 
88  while (iter != end)
89  {
90  int cHigh = charUnHex(*iter);
91  ++iter;
92 
93  if (cHigh < 0)
94  return {};
95 
96  int cLow = charUnHex(*iter);
97  ++iter;
98 
99  if (cLow < 0)
100  return {};
101 
102  out.push_back(static_cast<unsigned char>((cHigh << 4) | cLow));
103  }
104 
105  return {std::move(out)};
106 }
