31// Macros below are copied from antithesis_sdk.h and slightly simplified
32// The duplication is because Visual Studio 2019 cannot compile that header
33// even with the option -Zc:__cplusplus added.
34#define ALWAYS(cond, message, ...) assert((message) && (cond))
35#define ALWAYS_OR_UNREACHABLE(cond, message, ...) assert((message) && (cond))
36#define SOMETIMES(cond, message, ...)
37#define REACHABLE(message, ...)
38#define UNREACHABLE(message, ...) assert((message) && false)
39#endif
40
41#define XRPL_ASSERT ALWAYS_OR_UNREACHABLE
42
43// How to use the instrumentation macros:
44//
45// * XRPL_ASSERT if cond must be true but the line might not be reached during
46// fuzzing. Same like `assert` in normal use.
47// * ALWAYS if cond must be true _and_ the line must be reached during fuzzing.
48// Same like `assert` in normal use.
49// * REACHABLE if the line must be reached during fuzzing
50// * SOMETIMES a hint for the fuzzer to try to make the cond true
51// * UNREACHABLE if the line must not be reached (in fuzzing or in normal use).
52// Same like `assert(false)` in normal use.
53//
54// NOTE: XRPL_ASSERT has similar semantics as C `assert` macro, with only minor
55// differences:
56// * XRPL_ASSERT must have an unique name (naming convention in CONTRIBUTING.md)
57// * during fuzzing, the program will continue execution past failed XRPL_ASSERT
58//
59// We continue to use regular C `assert` inside unit tests and inside constexpr
60// functions.
61//
62// NOTE: UNREACHABLE does *not* have the same semantics as std::unreachable.
63// The program will continue execution past an UNREACHABLE in a Release build
64// and during fuzzing (similar to failed XRPL_ASSERT).
65// Also, the naming convention in UNREACHABLE is subtly different from other
66// instrumentation macros - its name describes the condition which was _not_
67// meant to happen, while name in other macros describes the condition that is
68// meant to happen (e.g. as in "assert that this happens").
