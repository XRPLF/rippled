263/* Extract 'length' bits from the char array 's'
264 starting with bit 'start' */
265unsigned long
266RFC1751::extract(char const* s, int start, int length)
267{
268 unsigned char cl;
269 unsigned char cc;
270 unsigned char cr;
271 unsigned long x;
272
273 XRPL_ASSERT(length <= 11, "ripple::RFC1751::extract : maximum length");
274 XRPL_ASSERT(start >= 0, "ripple::RFC1751::extract : minimum start");
275 XRPL_ASSERT(length >= 0, "ripple::RFC1751::extract : minimum length");
276 XRPL_ASSERT(
277 start + length <= 66,
278 "ripple::RFC1751::extract : maximum start + length");
279
280 int const shiftR = 24 - (length + (start % 8));
281 cl = s[start / 8]; // get components
282 cc = (shiftR < 16) ? s[start / 8 + 1] : 0;
283 cr = (shiftR < 8) ? s[start / 8 + 2] : 0;
284
285 x = ((long)(cl << 8 | cc) << 8 | cr); // Put bits together
286 x = x >> shiftR; // Right justify number
287 x = (x & (0xffff >> (16 - length))); // Trim extra bits.
288
289 return x;
290}

292// Encode 8 bytes in 'c' as a string of English words.
293// Returns a pointer to a static buffer
294void
295RFC1751::btoe(std::string& strHuman, std::string const& strData)
296{
297 char caBuffer[9]; /* add in room for the parity 2 bits*/
298 int p, i;
299
300 memcpy(caBuffer, strData.c_str(), 8);
301
302 // compute parity: merely add groups of two bits.
303 for (p = 0, i = 0; i < 64; i += 2)
304 p += extract(caBuffer, i, 2);
305
306 caBuffer[8] = char(p) << 6;
307
308 strHuman = std::string() + s_dictionary[extract(caBuffer, 0, 11)] + " " +
309 s_dictionary[extract(caBuffer, 11, 11)] + " " +
310 s_dictionary[extract(caBuffer, 22, 11)] + " " +
311 s_dictionary[extract(caBuffer, 33, 11)] + " " +
312 s_dictionary[extract(caBuffer, 44, 11)] + " " +
313 s_dictionary[extract(caBuffer, 55, 11)];
314}

316void
317RFC1751::insert(char* s, int x, int start, int length)
318{
319 unsigned char cl;
320 unsigned char cc;
321 unsigned char cr;
322 unsigned long y;
323 int shift;
324
325 XRPL_ASSERT(length <= 11, "ripple::RFC1751::insert : maximum length");
326 XRPL_ASSERT(start >= 0, "ripple::RFC1751::insert : minimum start");
327 XRPL_ASSERT(length >= 0, "ripple::RFC1751::insert : minimum length");
328 XRPL_ASSERT(
329 start + length <= 66,
330 "ripple::RFC1751::insert : maximum start + length");
331
332 shift = ((8 - ((start + length) % 8)) % 8);
333 y = (long)x << shift;
334 cl = (y >> 16) & 0xff;
335 cc = (y >> 8) & 0xff;
336 cr = y & 0xff;
337
338 if (shift + length > 16)
339 {
340 s[start / 8] |= cl;
341 s[start / 8 + 1] |= cc;
342 s[start / 8 + 2] |= cr;
343 }
344 else if (shift + length > 8)
345 {
346 s[start / 8] |= cc;
347 s[start / 8 + 1] |= cr;
348 }
349 else
350 {
351 s[start / 8] |= cr;
352 }
353}

371// Binary search of dictionary.
372int
373RFC1751::wsrch(std::string const& strWord, int iMin, int iMax)
374{
375 int iResult = -1;
376
377 while (iResult < 0 && iMin != iMax)
378 {
379 // Have a range to search.
380 int iMid = iMin + (iMax - iMin) / 2;
381 int iDir = strWord.compare(s_dictionary[iMid]);
382
383 if (!iDir)
384 {
385 iResult = iMid; // Found it.
386 }
387 else if (iDir < 0)
388 {
389 iMax = iMid; // key < middle, middle is new max.
390 }
391 else
392 {
393 iMin = iMid + 1; // key > middle, new min is past the middle.
394 }
395 }
396
397 return iResult;
398}

400// Convert 6 words to binary.
401//
402// Returns 1 OK - all good words and parity is OK
403// 0 word not in data base
404// -1 badly formed in put ie > 4 char word
405// -2 words OK but parity is wrong
406int
407RFC1751::etob(std::string& strData, std::vector<std::string> vsHuman)
408{
409 if (6 != vsHuman.size())
410 return -1;
411
412 int i, p = 0;
413 char b[9] = {0};
414
415 for (auto& strWord : vsHuman)
416 {
417 int l = strWord.length();
418
419 if (l > 4 || l < 1)
420 return -1;
421
422 standard(strWord);
423
424 auto v = wsrch(strWord, l < 4 ? 0 : 571, l < 4 ? 570 : 2048);
425
426 if (v < 0)
427 return 0;
428
429 insert(b, v, p, 11);
430 p += 11;
431 }
432
433 /* now check the parity of what we got */
434 for (p = 0, i = 0; i < 64; i += 2)
435 p += extract(b, i, 2);
436
437 if ((p & 3) != extract(b, 64, 2))
438 return -2;
439
440 strData.assign(b, 8);
441
442 return 1;
443}

497std::string
498RFC1751::getWordFromBlob(void const* blob, size_t bytes)
499{
500 // This is a simple implementation of the Jenkins one-at-a-time hash
501 // algorithm:
502 // http://en.wikipedia.org/wiki/Jenkins_hash_function#one-at-a-time
503 unsigned char const* data = static_cast<unsigned char const*>(blob);
504 std::uint32_t hash = 0;
505
506 for (size_t i = 0; i < bytes; ++i)
507 {
508 hash += data[i];
509 hash += (hash << 10);
510 hash ^= (hash >> 6);
511 }
512
513 hash += (hash << 3);
514 hash ^= (hash >> 11);
515 hash += (hash << 15);
516
517 return s_dictionary
518 [hash % (sizeof(s_dictionary) / sizeof(s_dictionary[0]))];
519}
