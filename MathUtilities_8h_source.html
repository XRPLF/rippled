43constexpr std::size_t
44calculatePercent(std::size_t count, std::size_t total)
45{
46 assert(total != 0); // NOTE No XRPL_ASSERT here, because constexpr
47 return ((std::min(count, total) * 100) + total - 1) / total;
48}
