39// Support for SignerEntries that is needed by a few Transactors.
40//
41// SignerEntries is represented as a std::vector<SignerEntries::SignerEntry>.
42// There is no direct constructor for SignerEntries.
43//
44// o A std::vector<SignerEntries::SignerEntry> is a SignerEntries.
45// o More commonly, SignerEntries are extracted from an STObject by
46// calling SignerEntries::deserialize().
47class SignerEntries
48{
49public:
50 explicit SignerEntries() = delete;
51
52 struct SignerEntry
53 {
54 AccountID account;
55 std::uint16_t weight;
56 std::optional<uint256> tag;
57
58 SignerEntry(
59 AccountID const& inAccount,
60 std::uint16_t inWeight,
61 std::optional<uint256> inTag)
62 : account(inAccount), weight(inWeight), tag(inTag)
63 {
64 }
65
66 // For sorting to look for duplicate accounts
67 friend bool
68 operator<(SignerEntry const& lhs, SignerEntry const& rhs)
69 {
70 return lhs.account < rhs.account;
71 }
72
73 friend bool
74 operator==(SignerEntry const& lhs, SignerEntry const& rhs)
75 {
76 return lhs.account == rhs.account;
77 }
78 };
79
80 // Deserialize a SignerEntries array from the network or from the ledger.
81 //
82 // obj Contains a SignerEntries field that is an STArray.
83 // journal For reporting error conditions.
84 // annotation Source of SignerEntries, like "ledger" or "transaction".
85 static Expected<std::vector<SignerEntry>, NotTEC>
86 deserialize(
87 STObject const& obj,
88 beast::Journal journal,
89 std::string_view annotation);
90};
