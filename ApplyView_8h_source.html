29 enum ApplyFlags : std::uint32_t {

31 
32  // This is a local transaction with the
33  // fail_hard flag set.
34  tapFAIL_HARD = 0x10,
35 
36  // This is not the transaction's last pass

38  tapRETRY = 0x20,
