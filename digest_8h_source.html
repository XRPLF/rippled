170struct basic_sha512_half_hasher
171{
172private:

175public:

184

187 {
188 h_(data, size);

190
191 explicit
192 operator result_type() noexcept
193 {
194 auto const digest = sha512_hasher::result_type(h_);
195 return result_type::fromVoid(digest.data());

198private:
199 inline void
200 erase(std::false_type)
201 {
202 }
203
204 inline void
205 erase(std::true_type)
206 {
207 secure_erase(&h_, sizeof(h_));
208 }
209};
