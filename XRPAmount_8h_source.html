266 inline XRPAmount
267 mulRatio(
268  XRPAmount const& amt,
269  std::uint32_t num,
270  std::uint32_t den,
271  bool roundUp)
272 {
273  using namespace boost::multiprecision;
274 
275  if (!den)
276  Throw<std::runtime_error>("division by zero");
277 
278  int128_t const amt128(amt.drops());
279  auto const neg = amt.drops() < 0;
280  auto const m = amt128 * num;
281  auto r = m / den;
282  if (m % den)
283  {
284  if (!neg && roundUp)
285  r += 1;
286  if (neg && !roundUp)
287  r -= 1;
288  }
289  if (r > std::numeric_limits<XRPAmount::value_type>::max())
290  Throw<std::overflow_error>("XRP mulRatio overflow");
291  return XRPAmount(r.convert_to<XRPAmount::value_type>());
292 }
