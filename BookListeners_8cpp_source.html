40void
41BookListeners::publish(
42 MultiApiJson const& jvObj,
43 hash_set<std::uint64_t>& havePublished)
44{
45 std::lock_guard sl(mLock);
46 auto it = mListeners.cbegin();
47
48 while (it != mListeners.cend())
49 {
50 InfoSub::pointer p = it->second.lock();
51
52 if (p)
53 {
54 // Only publish jvObj if this is the first occurence
55 if (havePublished.emplace(p->getSeq()).second)
56 {
57 jvObj.visit(
58 p->getApiVersion(), //
59 [&](Json::Value const& jv) { p->send(jv, true); });
60 }
61 ++it;
62 }
63 else
64 it = mListeners.erase(it);
65 }
66}
