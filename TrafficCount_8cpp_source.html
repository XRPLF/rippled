24TrafficCount::category
25TrafficCount::categorize(
26 ::google::protobuf::Message const& message,
27 int type,
28 bool inbound)
29{
30 if ((type == protocol::mtPING) || (type == protocol::mtSTATUS_CHANGE))
31 return TrafficCount::category::base;
32
33 if (type == protocol::mtCLUSTER)
34 return TrafficCount::category::cluster;
35
36 if (type == protocol::mtMANIFESTS)
37 return TrafficCount::category::manifests;
38
39 if (type == protocol::mtENDPOINTS)
40 return TrafficCount::category::overlay;
41
42 if (type == protocol::mtTRANSACTION)
43 return TrafficCount::category::transaction;
44
45 if (type == protocol::mtVALIDATORLIST ||
46 type == protocol::mtVALIDATORLISTCOLLECTION)
47 return TrafficCount::category::validatorlist;
48
49 if (type == protocol::mtVALIDATION)
50 return TrafficCount::category::validation;
51
52 if (type == protocol::mtPROPOSE_LEDGER)
53 return TrafficCount::category::proposal;
54
55 if (type == protocol::mtHAVE_SET)
56 return inbound ? TrafficCount::category::get_set
57 : TrafficCount::category::share_set;
58
59 if (auto msg = dynamic_cast<protocol::TMLedgerData const*>(&message))
60 {
61 if (msg->type() == protocol::liTS_CANDIDATE)
62 return (inbound && !msg->has_requestcookie())
63 ? TrafficCount::category::ld_tsc_get
64 : TrafficCount::category::ld_tsc_share;
65
66 if (msg->type() == protocol::liTX_NODE)
67 return (inbound && !msg->has_requestcookie())
68 ? TrafficCount::category::ld_txn_get
69 : TrafficCount::category::ld_txn_share;
70
71 if (msg->type() == protocol::liAS_NODE)
72 return (inbound && !msg->has_requestcookie())
73 ? TrafficCount::category::ld_asn_get
74 : TrafficCount::category::ld_asn_share;
75
76 return (inbound && !msg->has_requestcookie())
77 ? TrafficCount::category::ld_get
78 : TrafficCount::category::ld_share;
79 }
80
81 if (auto msg = dynamic_cast<protocol::TMGetLedger const*>(&message))
82 {
83 if (msg->itype() == protocol::liTS_CANDIDATE)
84 return (inbound || msg->has_requestcookie())
85 ? TrafficCount::category::gl_tsc_share
86 : TrafficCount::category::gl_tsc_get;
87
88 if (msg->itype() == protocol::liTX_NODE)
89 return (inbound || msg->has_requestcookie())
90 ? TrafficCount::category::gl_txn_share
91 : TrafficCount::category::gl_txn_get;
92
93 if (msg->itype() == protocol::liAS_NODE)
94 return (inbound || msg->has_requestcookie())
95 ? TrafficCount::category::gl_asn_share
96 : TrafficCount::category::gl_asn_get;
97
98 return (inbound || msg->has_requestcookie())
99 ? TrafficCount::category::gl_share
100 : TrafficCount::category::gl_get;
101 }
102
103 if (auto msg = dynamic_cast<protocol::TMGetObjectByHash const*>(&message))
104 {
105 if (msg->type() == protocol::TMGetObjectByHash::otLEDGER)
106 return (msg->query() == inbound)
107 ? TrafficCount::category::share_hash_ledger
108 : TrafficCount::category::get_hash_ledger;
109
110 if (msg->type() == protocol::TMGetObjectByHash::otTRANSACTION)
111 return (msg->query() == inbound)
112 ? TrafficCount::category::share_hash_tx
113 : TrafficCount::category::get_hash_tx;
114
115 if (msg->type() == protocol::TMGetObjectByHash::otTRANSACTION_NODE)
116 return (msg->query() == inbound)
117 ? TrafficCount::category::share_hash_txnode
118 : TrafficCount::category::get_hash_txnode;
119
120 if (msg->type() == protocol::TMGetObjectByHash::otSTATE_NODE)
121 return (msg->query() == inbound)
122 ? TrafficCount::category::share_hash_asnode
123 : TrafficCount::category::get_hash_asnode;
124
125 if (msg->type() == protocol::TMGetObjectByHash::otCAS_OBJECT)
126 return (msg->query() == inbound)
127 ? TrafficCount::category::share_cas_object
128 : TrafficCount::category::get_cas_object;
129
130 if (msg->type() == protocol::TMGetObjectByHash::otFETCH_PACK)
131 return (msg->query() == inbound)
132 ? TrafficCount::category::share_fetch_pack
133 : TrafficCount::category::get_fetch_pack;
134
135 if (msg->type() == protocol::TMGetObjectByHash::otTRANSACTIONS)
136 return TrafficCount::category::get_transactions;
137
138 return (msg->query() == inbound) ? TrafficCount::category::share_hash
139 : TrafficCount::category::get_hash;
140 }
141
142 if (type == protocol::mtPROOF_PATH_REQ)
143 return TrafficCount::category::proof_path_request;
144
145 if (type == protocol::mtPROOF_PATH_RESPONSE)
146 return TrafficCount::category::proof_path_response;
147
148 if (type == protocol::mtREPLAY_DELTA_REQ)
149 return TrafficCount::category::replay_delta_request;
150
151 if (type == protocol::mtREPLAY_DELTA_RESPONSE)
152 return TrafficCount::category::replay_delta_response;
153
154 if (type == protocol::mtHAVE_TRANSACTIONS)
155 return TrafficCount::category::have_transactions;
156
157 if (type == protocol::mtTRANSACTIONS)
158 return TrafficCount::category::requested_transactions;
159
160 return TrafficCount::category::unknown;
161}
