51void
52TimeoutCounter::setTimer(ScopedLockType& sl)
53{
54 if (isDone())
55 return;
56 timer_.expires_after(timerInterval_);
57 timer_.async_wait(
58 [wptr = pmDowncast()](boost::system::error_code const& ec) {
59 if (ec == boost::asio::error::operation_aborted)
60 return;
61
62 if (auto ptr = wptr.lock())
63 {
64 ScopedLockType sl(ptr->mtx_);
65 ptr->queueJob(sl);
66 }
67 });
68}

70void
71TimeoutCounter::queueJob(ScopedLockType& sl)
72{
73 if (isDone())
74 return;
75 if (queueJobParameter_.jobLimit &&
76 app_.getJobQueue().getJobCountTotal(queueJobParameter_.jobType) >=
77 queueJobParameter_.jobLimit)
78 {
79 JLOG(journal_.debug()) << "Deferring " << queueJobParameter_.jobName
80 << " timer due to load";
81 setTimer(sl);
82 return;
83 }
84
85 app_.getJobQueue().addJob(
86 queueJobParameter_.jobType,
87 queueJobParameter_.jobName,
88 [wptr = pmDowncast()]() {
89 if (auto sptr = wptr.lock(); sptr)
90 sptr->invokeOnTimer();
91 });
92}

94void
95TimeoutCounter::invokeOnTimer()
96{
97 ScopedLockType sl(mtx_);
98
99 if (isDone())
100 return;
101
102 if (!progress_)
103 {
104 ++timeouts_;
105 JLOG(journal_.debug())
106 << "Timeout(" << timeouts_ << ") " << " acquiring " << hash_;
107 onTimer(false, sl);
108 }
109 else
110 {
111 progress_ = false;
112 onTimer(true, sl);
113 }
114
115 if (!isDone())
116 setTimer(sl);
117}
