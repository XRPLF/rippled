328struct LedgerHistoryHelper
329{
330 LedgerOracle oracle;
331 Tx::ID nextTx{0};
332 std::unordered_map<std::string, Ledger> ledgers;
333 std::set<char> seen;
334
335 LedgerHistoryHelper()
336 {
337 ledgers[""] = Ledger{Ledger::MakeGenesis{}};
338 }
339
345 Ledger const&
346 operator[](std::string const& s)
347 {
348 auto it = ledgers.find(s);
349 if (it != ledgers.end())
350 return it->second;
351
352 // enforce that the new suffix has never been seen
353 assert(seen.emplace(s.back()).second);
354
355 Ledger const& parent = (*this)[s.substr(0, s.size() - 1)];
356 return ledgers.emplace(s, oracle.accept(parent, Tx{++nextTx}))
357 .first->second;
358 }
359};
